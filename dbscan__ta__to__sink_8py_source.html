30def make_moo_record(conf_dict,name,path='temptypes'):
31 fields = []
32 for pname,pvalue in conf_dict.items():
33 typename = None
34 if type(pvalue) == int:
35 typename = 'temptypes.temp_integer'
36 elif type(pvalue) == float:
37 typename = 'temptypes.temp_float'
38 elif type(pvalue) == str:
39 typename = 'temptypes.temp_string'
40 else:
41 raise Exception(f'Invalid config argument type: {type(value)}')
42 fields.append(dict(name=pname,item=typename))
43 moo.otypes.make_type(schema='record', fields=fields, name=name, path=path)
