363{

365 throw dunedaq::conffwk::Generic( ERS_HERE, "nothing to unload" );
366
367 std::lock_guard<std::mutex> scoped_lock1(m_tmpl_mutex); // always lock template objects mutex first
368 std::lock_guard<std::mutex> scoped_lock2(m_impl_mutex);
369
370 // call conffwk actions if any
371 {
372 std::lock_guard<std::mutex> scoped_lock(m_actn_mutex);
373 for(auto & i : m_actions)
374 {
375 i->unload();
376 }
377 }
378
379 for(auto & i : m_cache_map)
380 {
381 delete i.second;
382 }
383
384 m_cache_map.clear();
385
386 {
387 std::lock_guard<std::mutex> scoped_lock3(m_else_mutex);
388
389 for(auto& cb : m_callbacks)
390 delete cb;
391
392 for(auto& cb : m_pre_callbacks)
393 delete cb;
394
395 m_callbacks.clear();
396 m_pre_callbacks.clear();
397
398 m_impl->unsubscribe();
399
400 for(auto& l : m_convert_map)
401 {
402 for(auto& a : *l.second)
403 delete a;
404
405 delete l.second;
406 }
407
408 m_convert_map.clear();
409 }
410
411 p_superclasses.clear();
412
413 for(auto& j : p_direct_classes_desc_cache)
414 delete j.second;
415
416 for(auto& j : p_all_classes_desc_cache)
417 delete j.second;
418
419 p_direct_classes_desc_cache.clear();

422 m_impl->close_db();

1269{
1270 // check that there is no at least one subscription
1271 // if NO, then unsubscribe
1272
1273 if (m_callbacks.empty())
1274 {
1275 m_impl->unsubscribe();
1276 return;
1277 }
1278
1279 // prepare subscription criteria
1280
1281 ConfigurationSubscriptionCriteria::ObjectMap obj_subscriptions;
1282 std::set<std::string> class_subscriptions;
1283

1285
1286 bool found_subscribe_all = false;
1287
1288 for (const auto &i : m_callbacks)
1289 if (i->m_criteria.get_classes_subscription().empty() && i->m_criteria.get_objects_subscription().empty())
1290 {
1291 found_subscribe_all = true;
1292 break;
1293 }
1294
1295 if (found_subscribe_all == false)
1296 {
1297 // build list of all classes for which there is a subscription
1298 for (const auto &i : m_callbacks)
1299 for (const auto &j : i->m_criteria.get_classes_subscription())
1300 class_subscriptions.insert(j);
1301
1302 // build list of all objects for which there is a subscription (if there is no such class)
1303 for (const auto &i : m_callbacks)
1304 for (const auto &j : i->m_criteria.get_objects_subscription())
1305 {
1306 const std::string &obj_class_name = j.first;
1307 if (class_subscriptions.find(obj_class_name) == class_subscriptions.end())
1308 for (const auto &k : j.second)
1309 obj_subscriptions[obj_class_name].insert(k);
1310 }
1311 }
1312
1313 m_impl->subscribe(class_subscriptions, obj_subscriptions, system_cb, system_pre_cb);
1314}

1391{

1393
1394 // Remove deleted and update modified implementation objects first
1395 for (const auto& i : changes)
1396 {
1397 const std::string * class_name = &DalFactory::instance().get_known_class_name_ref(i->get_class_name());
1398
1399 update_impl_objects(m_impl->m_impl_objects, *i, class_name);
1400

1402 conffwk::fmap<conffwk::fset>::const_iterator sc = p_superclasses.find(class_name);

1405 for (const auto &c : sc->second)
1406 update_impl_objects(m_impl->m_impl_objects, *i, c);
1407
1408 // delete/update implementation objects defined in subclasses
1409 sc = p_subclasses.find(class_name);
1410
1411 if (sc != p_subclasses.end())
1412 for (const auto &c : sc->second)
1413 update_impl_objects(m_impl->m_impl_objects, *i, c);
1414 }
1415
1416 for (const auto& i : changes)
1417 {
1418 const std::string * class_name = &DalFactory::instance().get_known_class_name_ref(i->get_class_name()); // FIXME: optimise with above
1419
1420 // invoke configuration update if there are template objects of given class
1421
1422 {
1423 conffwk::fmap<CacheBase*>::iterator j = m_cache_map.find(class_name);
1424
1425 if (j != m_cache_map.end())
1426 {
1427 TLOG_DEBUG(3) << " * call update on \'" << j->first << "\' template objects";
1428 j->second->m_functions.m_update_fn(*this, i);
1429 }
1430 }
1431
1432
1433 // invoke configuration update if there are template objects in super-classes
1434
1435 {
1436 conffwk::fmap<conffwk::fset>::const_iterator sc = p_superclasses.find(class_name);
1437
1438 if (sc != p_superclasses.end())
1439 {
1440 for (const auto& c : sc->second)
1441 {
1442 conffwk::fmap<CacheBase*>::iterator j = m_cache_map.find(c);
1443
1444 if (j != m_cache_map.end())
1445 {
1446 TLOG_DEBUG(3) << " * call update on \'" << j->first << "\' template objects (as super-class of \'" << *class_name << "\')";
1447 j->second->m_functions.m_update_fn(*this, i);
1448 }
1449 }
1450 }
1451 }
1452
1453
1454 // invoke configuration update if there are template objects in sub-classes
1455
1456 {
1457 conffwk::fmap<conffwk::fset>::const_iterator sc = p_subclasses.find(class_name);
1458
1459 if (sc != p_subclasses.end())
1460 {
1461 for (const auto& c : sc->second)
1462 {
1463 conffwk::fmap<CacheBase*>::iterator j = m_cache_map.find(c);
1464
1465 if (j != m_cache_map.end())
1466 {
1467 TLOG_DEBUG(3) << " * call update on \'" << j->first << "\' template objects (as sub-class of \'" << *class_name << "\')";
1468 j->second->m_functions.m_update_fn(*this, i);
1469 }
1470 }
1471 }
1472 }
1473
1474 }

1479void
1480Configuration::system_cb(std::vector<ConfigurationChange *>& changes, Configuration * conf) noexcept
1481{
1482
1483 TLOG_DEBUG(3) <<
1484 "*** Enter Configuration::system_cb()\n"
1485 "*** Number of user subscriptions: " << conf->m_callbacks.size()
1486 ;
1487
1488 // call conffwk actions if any
1489 {
1490 std::lock_guard<std::mutex> scoped_lock(conf->m_impl_mutex);
1491 std::lock_guard<std::mutex> scoped_lock2(conf->m_actn_mutex);
1492 for (auto & i : conf->m_actions)
1493 i->notify(changes);
1494 }
1495
1496
1497 // update template objects in cache
1498 {
1499 std::lock_guard<std::mutex> scoped_lock(conf->m_tmpl_mutex); // always lock template objects mutex first
1500 std::lock_guard<std::mutex> scoped_lock2(conf->m_impl_mutex);
1501 conf->update_cache(changes);
1502 }
1503
1504
1505 // user removed all subscriptions
1506 if(conf->m_callbacks.empty()) return;
1507
1508 // note, one cannot lock m_tmpl_mutex or m_impl_mutex here,
1509 // since user callback may call arbitrary get() methods to access conffwk
1510 // and template objects locking above two mutexes
1511 std::lock_guard<std::mutex> scoped_lock(conf->m_else_mutex);
1512
1513 // check if there is only one subscription
1514 if (conf->m_callbacks.size() == 1)
1515 {
1516 auto j = *conf->m_callbacks.begin();
1517 (*(j->m_cb))(changes, j->m_param);
1518 }
1519 // may need to calculate the changes for each subscription
1520 else
1521 {
1522 for (const auto &j : conf->m_callbacks)
1523 {
1524
1525 if (ers::debug_level() >= 3)
1526 {
1527 std::ostringstream text;
1528
1529 text <<
1530 "*** Process subscription " << (void*) j << "\n"
1531 " class subscription done for " << j->m_criteria.get_classes_subscription().size() << " classes:\n";
1532
1533 for (const auto &i1 : j->m_criteria.get_classes_subscription())
1534 text << " * class \"" << i1 << "\"\n";
1535
1536 text << " object subscription done in " << j->m_criteria.get_objects_subscription().size() << " classes:\n";
1537
1538 for (const auto &i2 : j->m_criteria.get_objects_subscription())
1539 {
1540 text << " * class \"" << (i2.first) << "\":\n";
1541 for (const auto &i3 : i2.second)
1542 text << " - \"" << i3 << "\"\n";
1543 }
1544
1545 TLOG_DEBUG(3) << text.str();
1546 }
1547
1548 if (j->m_criteria.get_classes_subscription().empty() && j->m_criteria.get_objects_subscription().empty())
1549 {
1550 try
1551 {
1552 TLOG_DEBUG(3) << "*** Invoke callback " << (void *)j << " with\n" << changes;
1553 (*(j->m_cb))(changes, j->m_param);
1554 }
1555 catch (const ers::Issue &ex)
1556 {
1557 ers::error(dunedaq::conffwk::Generic( ERS_HERE, "user callback thrown ers exception", ex));
1558 }
1559 catch (const std::exception &ex)
1560 {
1561 ers::error(dunedaq::conffwk::Generic( ERS_HERE, "user callback thrown std exception", ex));
1562 }
1563 catch (...)
1564 {
1565 ers::error(dunedaq::conffwk::Generic( ERS_HERE, "user callback thrown unknown exception"));
1566 }
1567 }
1568 else
1569 {
1570 std::vector<ConfigurationChange*> changes1;
1571
1572 for (const auto &i : changes)
1573 {
1574 const std::string &cname = i->get_class_name();
1575 ConfigurationChange *class_changes = nullptr;
1576
1577 ConfigurationSubscriptionCriteria::ObjectMap::const_iterator p = j->m_criteria.get_objects_subscription().find(cname);
1578 const bool found_obj_subscription(p != j->m_criteria.get_objects_subscription().end());
1579 const bool found_class_subscription(j->m_criteria.get_classes_subscription().find(cname) != j->m_criteria.get_classes_subscription().end());
1580
1581 if (found_class_subscription || found_obj_subscription)
1582 class_changes = new ConfigurationChange(cname);
1583
1584 if (found_class_subscription)
1585 {
1586 for (const auto &k : i->m_modified)
1587 class_changes->m_modified.push_back(k);
1588
1589 for (const auto &k : i->m_created)
1590 class_changes->m_created.push_back(k);
1591
1592 for (const auto &k : i->m_removed)
1593 class_changes->m_removed.push_back(k);
1594 }
1595
1596 if (found_obj_subscription)
1597 {
1598 for (const auto &obj_id : i->m_modified)
1599 if (p->second.find(obj_id) != p->second.end())
1600 class_changes->m_modified.push_back(obj_id);
1601
1602 for (const auto &obj_id : i->m_removed)
1603 if (p->second.find(obj_id) != p->second.end())
1604 class_changes->m_removed.push_back(obj_id);
1605 }
1606
1607 // the changes for given class can be empty if there is subscription on objects of given class, but no such objects were modified
1608 if (class_changes)
1609 {
1610 if (class_changes->m_modified.empty() && class_changes->m_created.empty() && class_changes->m_removed.empty())
1611 delete class_changes;
1612 else
1613 changes1.push_back(class_changes);
1614 }
1615 }
1616
1617 if (!changes1.empty())
1618 {
1619 TLOG_DEBUG(3) << "*** Invoke callback " << (void *)j << " with\n" << changes1;
1620
1621 try
1622 {
1623 (*(j->m_cb))(changes1, j->m_param);
1624 }
1625 catch (const ers::Issue &ex)
1626 {
1627 ers::error(dunedaq::conffwk::Generic( ERS_HERE, "user callback thrown ers exception", ex));
1628 }
1629 catch (const std::exception &ex)
1630 {
1631 ers::error(dunedaq::conffwk::Generic( ERS_HERE, "user callback thrown std exception", ex));
1632 }
1633 catch (...)
1634 {
1635 ers::error(dunedaq::conffwk::Generic( ERS_HERE, "user callback thrown unknown exception"));
1636 }
1637
1638 for (const auto &i : changes1)
1639 delete i;
1640 }
1641 }
1642 }
1643 }
1644
1645 TLOG_DEBUG(3) <<"*** Leave Configuration::system_cb()";
1646}
