78{
79 write_soft_reset_register();
80
81 millisleep(1000);
82
83 // Upload config file to PLL
84 configure_pll(clock_config_file);
85
86 // Reset mmcm
87 getNode("csr.ctrl.rst").write(0x1);
88 getNode("csr.ctrl.rst").write(0x0);
89
90 getNode("io_select.csr.ctrl.amc_out").write(0xfff);
91 getNode("io_select.csr.ctrl.amc_in").write(0x0);
92 getNode("io_select.csr.ctrl.usfp_src").write(0x0);
93
94 getClient().dispatch();
95
96 TLOG() << "Reset done";
97}

146std::string
147MIBIONode::get_sfp_status(uint32_t sfp_id, bool print_out) const { // NOLINT(build/unsigned)
148 std::stringstream status;
149
150 validate_sfp_id(sfp_id);
151
152 // enable i2c path for sfp
153 auto i2c_switch = get_i2c_device<I2C9546SwitchSlave>("i2c", "TCA9546_Switch");
154 i2c_switch->set_channels_states(1UL << sfp_id);
155
156 auto sfp = get_i2c_device<I2CSFPSlave>(m_sfp_i2c_buses.at(0), "SFP_EEProm");
157
158 status << "SFP " << sfp_id << ":" << std::endl;
159
160 try
161 {
162 status << sfp->get_status();
163 }
164 catch(...)
165 {
166 i2c_switch->set_channels_states(8);
167 throw;
168 }
169
170 i2c_switch->set_channels_states(8);
171
172 if (print_out)
173 TLOG() << status.str();
174
175 return status.str();
176}
