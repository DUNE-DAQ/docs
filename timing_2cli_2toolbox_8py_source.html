30class IntRange(click.types.IntParamType):
31 """A parameter that works similar to :data:`click.INT` but restricts
32 the value to fit into a range. The default behavior is to fail if the
33 value falls outside the range, but it can also be silently clamped
34 between the two edges.
35
36 See :ref:`ranges` for an example.
37 """
38 name = 'integer range'
39
40 def __init__(self, min=None, max=None, clamp=False):
41 self.min = min
42 self.max = max
43 self.clamp = clamp
44
45 def convert(self, value, param, ctx):
46
47 if type(value) == str:
48 if value.startswith('0x'):
49 base = 16
50 elif value.startswith('0o'):
51 bae = 8
52 elif value.startswith('0b'):
53 base = 2
54 else:
55 base = 10
56 rv = int(value, base)
57 else:
58 rv = int(value)
59 if self.clamp:
60 if self.min is not None and rv < self.min:
61 return self.min
62 if self.max is not None and rv > self.max:
63 return self.max
64 if self.min is not None and rv < self.min or \
65 self.max is not None and rv > self.max:
66 if self.min is None:
67 self.fail('%s is bigger than the maximum valid value '
68 '%s.' % (rv, self.max), param, ctx)
69 elif self.max is None:
70 self.fail('%s is smaller than the minimum valid value '
71 '%s.' % (rv, self.min), param, ctx)
72 else:
73 self.fail('%s is not in the valid range of %s to %s.'
74 % (rv, self.min, self.max), param, ctx)
75 return rv
76
77 def __repr__(self):
78 return 'IntRange(%r, %r)' % (self.min, self.max)

182def split_ints(ctx, param, value):
183
184 sep = ','
185 dash = '-'
186
187 if value is None:
188 return []
189
190 numbers = []
191 for item in value.split(sep):
192 nums = item.split(dash)
193 if len(nums) == 1:
194 # single entry
195 numbers.append(__str2int__(item))
196 elif len(nums) == 2:
197 # range
198 i, j = __str2int__(nums[0]), __str2int__(nums[1])
199 if i > j:
200 click.ClickException('Invalid interval '+item)
201 numbers.extend(list(range(i,j+1)))
202 else:
203 click.ClickException('Malformed option (comma separated list expected): {}'.format(value))
204
205 return numbers
