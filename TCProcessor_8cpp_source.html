261dfmessages::TriggerDecision
262TCProcessor::create_decision(const PendingTD& pending_td)
263{
264 m_earliest_tc_index = get_earliest_tc_index(pending_td);
265 TLOG_DEBUG(5) << "earliest TC index: " << m_earliest_tc_index;
266
267 if (pending_td.contributing_tcs.size() > 1) {
268 TLOG_DEBUG(5) << "!!! TD created from " << pending_td.contributing_tcs.size() << " TCs !!!";
269 }
270
271 dfmessages::TriggerDecision decision;
272 decision.trigger_number = 0; // filled by MLT
273 decision.run_number = 0; // filled by MLT
274 decision.trigger_timestamp = pending_td.contributing_tcs[m_earliest_tc_index].time_candidate;
275 decision.readout_type = dfmessages::ReadoutType::kLocalized;
276
277 if (m_hsi_passthrough == true) {
278 if (pending_td.contributing_tcs[m_earliest_tc_index].type == TCType::kTiming) {
279 decision.trigger_type = pending_td.contributing_tcs[m_earliest_tc_index].detid & 0xff;
280 } else {
281 m_trigger_type_shifted = (static_cast<int>(pending_td.contributing_tcs[m_earliest_tc_index].type) << 8);
282 decision.trigger_type = m_trigger_type_shifted;
283 }
284 } else {
285 m_TD_bitword = get_TD_bitword(pending_td);
286 TLOG_DEBUG(5) << "[MLT] TD has bitword: " << m_TD_bitword << " "
287 << static_cast<dfmessages::trigger_type_t>(m_TD_bitword.to_ulong());
288 decision.trigger_type = static_cast<dfmessages::trigger_type_t>(m_TD_bitword.to_ulong()); // m_trigger_type;
289
290 //decision.trigger_type = 1; // m_trigger_type;
291 }
292
293 TLOG_DEBUG(3) << "HSI passthrough: " << m_hsi_passthrough
294 << ", TC detid: " << pending_td.contributing_tcs[m_earliest_tc_index].detid
295 << ", TC type: " << static_cast<int>(pending_td.contributing_tcs[m_earliest_tc_index].type)
296 << ", TC cont number: " << pending_td.contributing_tcs.size()
297 << ", DECISION trigger type: " << decision.trigger_type
298 << ", DECISION timestamp: " << decision.trigger_timestamp
299 << ", request window begin: " << pending_td.readout_start
300 << ", request window end: " << pending_td.readout_end;
301
302 std::vector<dfmessages::ComponentRequest> requests =
303 create_all_decision_requests(m_mandatory_links, pending_td.readout_start, pending_td.readout_end);
304 add_requests_to_decision(decision, requests);
305
306 if (!m_use_roi_readout) {
307 for (const auto& [key, value] : m_group_links) {
308 std::vector<dfmessages::ComponentRequest> group_requests =
309 create_all_decision_requests(value, pending_td.readout_start, pending_td.readout_end);
310 add_requests_to_decision(decision, group_requests);
311 }
312 } else { // using ROI readout
313 roi_readout_make_requests(decision);
314 }
315
316 m_tds_created_count++;
317 m_tds_created_tc_count += pending_td.contributing_tcs.size();
318
319 return decision;
320}

323void
324TCProcessor::send_trigger_decisions() {
325 // A unique lock that can be locked and unlocked
326 std::unique_lock<std::mutex> lock(m_td_vector_mutex);
327
328 while (m_running_flag) {
329 // Either there are pending TDs, or wait for a bit
330 m_cv.wait(lock, [this] {
331 return !m_pending_tds.empty() || !m_running_flag;
332 });
333 auto ready_tds = get_ready_tds(m_pending_tds);
334 TLOG_DEBUG(10) << "ready tds: " << ready_tds.size() << ", updated pending tds: " << m_pending_tds.size();
335
336 for (std::vector<PendingTD>::iterator it = ready_tds.begin(); it != ready_tds.end();) {
337 call_tc_decision(*it);
338 ++it;
339 }
340 }

343void
344TCProcessor::call_tc_decision(const TCProcessor::PendingTD& pending_td)
345{
346
347 if (m_use_bitwords) {
348 // Check trigger bitwords
349 m_TD_bitword = get_TD_bitword(pending_td);
350 m_bitword_check = check_trigger_bitwords();
351 if (m_bitword_check == false) {
352 m_tds_failed_bitword_count++;
353 m_tds_failed_bitword_tc_count += pending_td.contributing_tcs.size();
354 }
355 }
356
357 if ((!m_use_bitwords) || (m_bitword_check)) {
358
359 dfmessages::TriggerDecision decision = create_decision(pending_td);
360 auto tn = decision.trigger_number;
361 auto td_ts = decision.trigger_timestamp;
362
363 if (m_latency_monitoring.load()) m_latency_instance.update_latency_out( pending_td.contributing_tcs.front().time_start );
364 if(!m_td_sink->try_send(std::move(decision), iomanager::Sender::s_no_block)) {
365 ers::warning(TDDropped(ERS_HERE, tn, td_ts));
366 m_tds_dropped_count++;
367 m_tds_dropped_tc_count += pending_td.contributing_tcs.size();
368 }
369 else {
370 m_tds_sent_count++;
371 m_tds_sent_tc_count += pending_td.contributing_tcs.size();
372 }
373 }

377void
378TCProcessor::add_tc(const triggeralgs::TriggerCandidate tc)
379{
380 bool tc_dealt = false;
381 int64_t tc_wallclock_arrived =
382 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now().time_since_epoch()).count();
383
384 if (m_tc_merging || m_ignore_tc_pileup) {
385
386 for (std::vector<PendingTD>::iterator it = m_pending_tds.begin(); it != m_pending_tds.end();) {
387 // Don't deal with TC here if there's no overlap
388 if (!check_overlap(tc, *it)) {
389 it++;
390 continue;
391 }
392
393 // If overlap and ignoring, we drop the TC and flag it as dealt with.
394 if (m_ignore_tc_pileup) {
395 m_tds_dropped_tc_count++;
396 tc_dealt = true;
397 TLOG_DEBUG(3) << "TC overlapping with a previous TD, dropping!";
398 break;
399 }
400
401 // If we're here, TC merging must be on, in which case we're actually
402 // going to merge the TC into the TD.
403 it->contributing_tcs.push_back(tc);
404 if ( (m_use_readout_map) && (m_readout_window_map.count(tc.type)) ){
405 TLOG_DEBUG(3) << "TC with start/end times " << tc.time_candidate - m_readout_window_map[tc.type].first << "/"
406 << tc.time_candidate + m_readout_window_map[tc.type].second
407 << " overlaps with pending TD with start/end times " << it->readout_start << "/"
408 << it->readout_end;
409 it->readout_start = ((tc.time_candidate - m_readout_window_map[tc.type].first) >= it->readout_start)
410 ? it->readout_start
411 : (tc.time_candidate - m_readout_window_map[tc.type].first);
412 it->readout_end = ((tc.time_candidate + m_readout_window_map[tc.type].second) >= it->readout_end)
413 ? (tc.time_candidate + m_readout_window_map[tc.type].second)
414 : it->readout_end;
415 } else {
416 TLOG_DEBUG(3) << "TC with start/end times " << tc.time_start << "/" << tc.time_end
417 << " overlaps with pending TD with start/end times " << it->readout_start << "/"
418 << it->readout_end;
419 it->readout_start = (tc.time_start >= it->readout_start) ? it->readout_start : tc.time_start;
420 it->readout_end = (tc.time_end >= it->readout_end) ? tc.time_end : it->readout_end;
421 }
422 it->walltime_expiration = tc_wallclock_arrived + m_buffer_timeout;
423 tc_dealt = true;
424 break;
425 }
426 }
427
428 // Don't do anything else if we've already dealt with the TC
429 if (tc_dealt) {
430 return;
431 }
432
433 // Create a new TD out of the TC
434 PendingTD td_candidate;
435 td_candidate.contributing_tcs.push_back(tc);
436 if ( (m_use_readout_map) && (m_readout_window_map.count(tc.type)) ){
437 td_candidate.readout_start = tc.time_candidate - m_readout_window_map[tc.type].first;
438 td_candidate.readout_end = tc.time_candidate + m_readout_window_map[tc.type].second;
439 } else {
440 td_candidate.readout_start = tc.time_start;
441 td_candidate.readout_end = tc.time_end;
442 }
443 td_candidate.walltime_expiration = tc_wallclock_arrived + m_buffer_timeout;
444 m_pending_tds.push_back(td_candidate);
445}

447void
448TCProcessor::add_tc_ignored(const triggeralgs::TriggerCandidate tc)
449{
450 for (std::vector<PendingTD>::iterator it = m_pending_tds.begin(); it != m_pending_tds.end();) {
451 if (check_overlap(tc, *it)) {
452 if ( (m_use_readout_map) && (m_readout_window_map.count(tc.type)) ) {
453 TLOG_DEBUG(3) << "!Ignored! TC with start/end times " << tc.time_candidate - m_readout_window_map[tc.type].first
454 << "/" << tc.time_candidate + m_readout_window_map[tc.type].second
455 << " overlaps with pending TD with start/end times " << it->readout_start << "/"
456 << it->readout_end;
457 } else {
458 TLOG_DEBUG(3) << "!Ignored! TC with start/end times " << tc.time_start << "/" << tc.time_end
459 << " overlaps with pending TD with start/end times " << it->readout_start << "/"
460 << it->readout_end;
461 }
462 it->contributing_tcs.push_back(tc);
463 break;
464 }
465 ++it;
466 }
467 return;

470bool
471TCProcessor::check_overlap(const triggeralgs::TriggerCandidate& tc, const PendingTD& pending_td)
472{
473 if ( (m_use_readout_map) && (m_readout_window_map.count(tc.type)) ) {
474 return !(((tc.time_candidate + m_readout_window_map[tc.type].second) < pending_td.readout_start) ||
475 ((tc.time_candidate - m_readout_window_map[tc.type].first > pending_td.readout_end)));
476 } else {
477 return !((tc.time_end < pending_td.readout_start) || (tc.time_start > pending_td.readout_end));
478 }

481std::vector<TCProcessor::PendingTD>
482TCProcessor::get_ready_tds(std::vector<PendingTD>& pending_tds)
483{
484 std::vector<PendingTD> return_tds;
485 for (std::vector<PendingTD>::iterator it = pending_tds.begin(); it != pending_tds.end();) {
486 auto timestamp_now =
487 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now().time_since_epoch())
488 .count();
489 if (timestamp_now >= it->walltime_expiration) {
490 return_tds.push_back(*it);
491 it = pending_tds.erase(it);
492 } else if (check_td_readout_length(*it)) { // Also pass on TDs with (too) long readout window
493 return_tds.push_back(*it);
494 it = pending_tds.erase(it);
495 } else {
496 ++it;
497 }
498 }
499 return return_tds;
500}

502int
503TCProcessor::get_earliest_tc_index(const PendingTD& pending_td)
504{
505 int earliest_tc_index = -1;
506 triggeralgs::timestamp_t earliest_tc_time;
507 for (int i = 0; i < static_cast<int>(pending_td.contributing_tcs.size()); i++) {
508 if (earliest_tc_index == -1) {
509 earliest_tc_time = pending_td.contributing_tcs[i].time_candidate;
510 earliest_tc_index = i;
511 } else {
512 if (pending_td.contributing_tcs[i].time_candidate < earliest_tc_time) {
513 earliest_tc_time = pending_td.contributing_tcs[i].time_candidate;
514 earliest_tc_index = i;
515 }
516 }
517 }
518 return earliest_tc_index;

532void
533TCProcessor::clear_td_vectors()
534{
535 std::lock_guard<std::mutex> lock(m_td_vector_mutex);
536 m_tds_cleared_count += m_pending_tds.size();
537 // Use std::accumulate to sum up the sizes of all contributing_tcs vectors
538 size_t tds_cleared_tc_count = std::accumulate(
539 m_pending_tds.begin(), m_pending_tds.end(), 0,
540 [](size_t sum, const PendingTD& ptd) {
541 return sum + ptd.contributing_tcs.size();

772void
773TCProcessor::roi_readout_make_requests(dfmessages::TriggerDecision& decision)
774{
775 // Get configuration at random (weighted)
776 int group_pick = pick_roi_group_conf();
777 if (group_pick != -1) {
778 roi_group this_group = m_roi_conf[m_roi_conf_ids[group_pick]];
779 std::vector<dfmessages::SourceID> links;
780
781 // If mode is random, pick groups to request at random
782 if (this_group.mode == "kRandom") {
783 TLOG_DEBUG(10) << "RAND";
784 std::set<int> groups;
785 while (static_cast<int>(groups.size()) < this_group.n_links) {
786 groups.insert(get_random_num_int());
787 }
788 for (auto r_id : groups) {
789 links.insert(links.end(), m_group_links[r_id].begin(), m_group_links[r_id].end());
790 }
791 // Otherwise, read sequntially by IDs, starting at 0
792 } else {
793 TLOG_DEBUG(10) << "SEQ";
794 int r_id = 0;
795 while (r_id < this_group.n_links) {
796 links.insert(links.end(), m_group_links[r_id].begin(), m_group_links[r_id].end());
797 r_id++;
798 }
799 }
800
801 TLOG_DEBUG(10) << "TD timestamp: " << decision.trigger_timestamp;
802 TLOG_DEBUG(10) << "group window: " << this_group.time_window;
803
804 // Once the components are prepared, create requests and append them to decision
805 std::vector<dfmessages::ComponentRequest> requests =
806 create_all_decision_requests(links, decision.trigger_timestamp - this_group.time_window,
807 decision.trigger_timestamp + this_group.time_window);
808 add_requests_to_decision(decision, requests);
809 links.clear();
810 }
811 return;
812}

815TCProcessor::get_TD_bitword(const PendingTD& ready_td)
816{
817 // get only unique types
818 std::vector<int> tc_types;
819 for (auto tc : ready_td.contributing_tcs) {
820 tc_types.push_back(static_cast<int>(tc.type));

824 // form TD bitword

826 for (auto tc_type : tc_types) {
827 td_bitword.set(tc_type);
828 }
829 return td_bitword;
