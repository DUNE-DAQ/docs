71void
72TAProcessor::conf(const appmodel::DataHandlerModule* conf)
73{
74 for (auto output : conf->get_outputs()) {
75 try {
76 if (output->get_data_type() == "TriggerCandidate") {
77 m_tc_sink = get_iom_sender<triggeralgs::TriggerCandidate>(output->UID());
78 }
79 } catch (const ers::Issue& excpt) {
80 ers::error(datahandlinglibs::ResourceQueueError(ERS_HERE, "tc", "DefaultRequestHandlerModel", excpt));
81 }
82 }
83
84 m_sourceid.id = conf->get_source_id();
85 m_sourceid.subsystem = trigger::TAWrapper::subsystem;
86 std::vector<const appmodel::TCAlgorithm*> tc_algorithms;
87 auto dp = conf->get_module_configuration()->get_data_processor();
88 auto proc_conf = dp->cast<appmodel::TADataProcessor>();
89 if (proc_conf != nullptr && m_post_processing_enabled ) {
90 tc_algorithms = proc_conf->get_algorithms();
91 }
92
93 for (auto algo : tc_algorithms) {
94 TLOG() << "Selected TC algorithm: " << algo->UID();
95 std::shared_ptr<triggeralgs::TriggerCandidateMaker> maker = make_tc_maker(algo->class_name());
96 nlohmann::json algo_json = algo->to_json(true);
97 maker->configure(algo_json[algo->UID()]);
98 inherited::add_postprocess_task(std::bind(&TAProcessor::find_tc, this, std::placeholders::_1, maker));
99 m_tcms.push_back(maker);
100 }
101 m_latency_monitoring.store( dp->get_latency_monitoring() );
102 inherited::conf(conf);
103}

130void
131TAProcessor::find_tc(const TAWrapper* ta, std::shared_ptr<triggeralgs::TriggerCandidateMaker> tca)
132{
133 //time_activity gave 0 :/
134 if (m_latency_monitoring.load()) m_latency_instance.update_latency_in( ta->activity.time_start );
135 m_ta_received_count++;
136 std::vector<triggeralgs::TriggerCandidate> tcs;
137 tca->operator()(ta->activity, tcs);
138 for (auto tc : tcs) {
139 m_tc_made_count++;
140 if (m_latency_monitoring.load()) m_latency_instance.update_latency_out( tc.time_candidate );
141 if(!m_tc_sink->try_send(std::move(tc), iomanager::Sender::s_no_block)) {
142 ers::warning(TCDropped(ERS_HERE, tc.time_start, m_sourceid.id));
143 m_tc_failed_sent_count++;
144 } else {
145 m_tc_sent_count++;
146 }
147 }
148 return;
149}
