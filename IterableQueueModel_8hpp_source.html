249 Iterator(IterableQueueModel<T>& queue, uint32_t index) // NOLINT(build/unsigned)
250 : m_queue(queue)
251 , m_index(index)
252 {}
253
254 reference operator*() const { return m_queue.records_[m_index]; }
255 pointer operator->() { return &m_queue.records_[m_index]; }
256 Iterator& operator++() // NOLINT(runtime/increment_decrement) :)
257 {
258 if (good()) {
259 m_index++;
260 if (m_index == m_queue.size_) {
261 m_index = 0;
262 }
263 }
264 if (!good()) {
265 m_index = std::numeric_limits<uint32_t>::max(); // NOLINT(build/unsigned)
266 }
267 return *this;
268 }
269 Iterator operator++(int amount) // NOLINT(runtime/increment_decrement) :)
270 {
271 Iterator tmp = *this;
272 for (int i = 0; i < amount; ++i) {
273 ++(*this);
274 }
275 return tmp;
276 }
277 friend bool operator==(const Iterator& a, const Iterator& b) { return a.m_index == b.m_index; }
278 friend bool operator!=(const Iterator& a, const Iterator& b) { return a.m_index != b.m_index; }
279
280 bool good()
281 {
282 auto const currentRead = m_queue.readIndex_.load(std::memory_order_relaxed);
283 auto const currentWrite = m_queue.writeIndex_.load(std::memory_order_relaxed);
284 return (*this != m_queue.end()) &&
285 ((m_index >= currentRead && m_index < currentWrite) ||
286 (m_index >= currentRead && currentWrite < currentRead) ||
287 (currentWrite < currentRead && m_index < currentRead && m_index < currentWrite));
288 }
289
290 uint32_t get_index() { return m_index; } // NOLINT(build/unsigned)
291
292 private:
293 IterableQueueModel<T>& m_queue;
294 uint32_t m_index; // NOLINT(build/unsigned)
295 };

297 Iterator begin()
298 {
299 auto const currentRead = readIndex_.load(std::memory_order_relaxed);
300 if (currentRead == writeIndex_.load(std::memory_order_acquire)) {
301 // queue is empty
302 return end();
303 }
304 return Iterator(*this, currentRead);
305 }

307 Iterator end()
308 {
309 return Iterator(*this, std::numeric_limits<uint32_t>::max()); // NOLINT(build/unsigned)
