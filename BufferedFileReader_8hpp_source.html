101 TLOG_DEBUG(TLVL_WORK_STEPS) << "Using zlib compression" << std::endl;

104 TLOG_DEBUG(TLVL_WORK_STEPS) << "Running without compression" << std::endl;
105 } else {

110 m_input_stream.push(io_source, m_buffer_size);
