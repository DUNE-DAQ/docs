78void
79Application::execute(const dataobj_t& cmd_data)
80{
81 auto rc_cmd = cmd_data.get<rcif::cmd::RCCommand>();
82 std::string cmdname = rc_cmd.id;
83 if (!is_cmd_valid(cmd_data)) {
84 throw InvalidCommand(ERS_HERE, cmdname, get_state(), m_error.load(), m_busy.load());
85 }
86
87 m_busy.store(true);
88
89 if (cmdname == "start") {
90 auto cmd_obj = rc_cmd.data.get<cmd::CmdObj>();
91
92 for (const auto& addressed : cmd_obj.modules) {
93 dataobj_t startpars = addressed.data;
94 auto rc_startpars = startpars.get<rcif::cmd::StartParams>();
95 m_runinfo.set_run_number(rc_startpars.run);
96 break;
97 }
98
99 m_run_start_time = std::chrono::steady_clock::now();
100 m_runinfo.set_running(true);
101 m_runinfo.set_run_time(0);
102 }
103 else if (cmdname == "stop") {
104 m_run_start_time = std::chrono::steady_clock::time_point();
105 m_runinfo.set_running(false);
106 m_runinfo.set_run_number(0);
107 m_runinfo.set_run_time(0);
108 }
109
110 try {
111 m_mod_mgr.execute(cmdname, rc_cmd.data);
112 m_busy.store(false);
113 if (rc_cmd.exit_state != "ANY")
114 set_state(rc_cmd.exit_state);
115 } catch (ers::Issue& ex) {
116 m_busy.store(false);
117 m_error.store(true);
118 throw;
119 }
120}

122void
123Application::generate_opmon_data()
124{
125 opmon::AppInfo ai;
126 ai.set_state(get_state());
127 ai.set_busy(m_busy.load());
128 ai.set_error(m_error.load());
129
130 char hostname[256];
131 auto res = gethostname(hostname, 256);
132 if (res < 0)
133 ai.set_host("Unknown");
134 else
135 ai.set_host (std::string(hostname));
136
137 publish(std::move(ai), {}, opmonlib::to_level(opmonlib::EntryOpMonLevel::kTopPriority));
138
139 if ( m_run_start_time.time_since_epoch().count() != 0 ) {
140 auto now = std::chrono::steady_clock::now();
141 m_runinfo.set_run_time(std::chrono::duration_cast<std::chrono::seconds>(now - m_run_start_time).count() );
142 }
143
144 publish( decltype(m_runinfo)(m_runinfo) );
145}

147bool
148Application::is_cmd_valid(const dataobj_t& cmd_data)
149{
150 if (m_busy.load() || m_error.load())
151 return false;
152
153 std::string state = get_state();
154 std::string entry_state = cmd_data.get<rcif::cmd::RCCommand>().entry_state;
155 if (entry_state == "ANY" || state == entry_state)
156 return true;
157
158 return false;
159}
