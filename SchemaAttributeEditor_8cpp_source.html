40void dbse::SchemaAttributeEditor::FillInfo()

42 setWindowTitle ( QString ( "Attribute Editor : %1" ).arg (
43 SchemaAttribute->get_name().c_str() ) );
44 setObjectName ( QString::fromStdString ( SchemaAttribute->get_name() ) );
45 ui->AttributeNameLineEdit->setText ( QString::fromStdString (
46 SchemaAttribute->get_name() ) );
47 ui->AttributeTypeComboBox->setCurrentIndex (
48 ui->AttributeTypeComboBox->findData ( QString::fromStdString (
49 SchemaAttribute->get_type() ),
50 Qt::DisplayRole ) );
51
52 if ( SchemaAttribute->get_is_multi_values() )
53 ui->AttributeIsMultivariable->setCurrentIndex (
54 0 );
55 else
56 {
57 ui->AttributeIsMultivariable->setCurrentIndex ( 1 );
58 }
59
60 if ( SchemaAttribute->get_is_no_null() )
61 {
62 ui->AttributeIsNotNull->setCurrentIndex ( 0 );
63 }
64 else
65 {
66 ui->AttributeIsNotNull->setCurrentIndex ( 1 );
67 }
68
69 ui->AttributeDescriptionTextBox->setPlainText (
70 QString::fromStdString ( SchemaAttribute->get_description() ) );
71 ui->AttributeRangeLineEdit->setText ( QString::fromStdString (
72 SchemaAttribute->get_range() ) );
73 ui->AttributeInitialValue->setText (
74 QString::fromStdString ( SchemaAttribute->get_init_value() ) );
75
76 int Index = ui->AttributeTypeComboBox->currentIndex();
77
78 if ( Index == 0 || Index >= 9 )
79 {
80 ui->FormatLayout->setEnabled ( false );
81 ui->FormatLabel->hide();
82 ui->AttributeFormatComboBox->hide();
83 }
84 else
85 {
86 ui->FormatLayout->setEnabled ( true );
87 ui->FormatLabel->show();

94 ui->AttributeFormatComboBox->setCurrentIndex (

139void dbse::SchemaAttributeEditor::ParseToSave()
140{
141 bool Changed = false;
142 std::string AttributeName = ui->AttributeNameLineEdit->text().toStdString();
143 std::string AttributeType = ui->AttributeTypeComboBox->currentText().toStdString();
144 char AttributeFormatChar = ( ui->AttributeFormatComboBox->currentText().toStdString() ) [0];

148 switch ( AttributeFormatChar )

156 break;
157

160 break;
161
162 default:
163 AttributeFormat = OksAttribute::Format::Dec;
164 break;
165 }
166
168 int Index = ui->AttributeTypeComboBox->currentIndex();
169
170 if ( Index == 0 || Index >= 9 )
171 {
172 AttributeFormat = OksAttribute::Format::Dec;
173 }
174

176
177 switch ( ui->AttributeIsMultivariable->currentIndex() )

179 case 0:
180 IsMultivariable = true;

191

193

196 case 0:
197 IsNotNull = true;
198 break;
199
200 case 1:
201 IsNotNull = false;
202 break;
203
204 default:
205 IsNotNull = false;
206 break;
207 }
208
209 std::string AttributeDescription =
210 ui->AttributeDescriptionTextBox->toPlainText().toStdString();
211 std::string AttributeRange = ui->AttributeRangeLineEdit->text().toStdString();
212 std::string AttributeInitial = ui->AttributeInitialValue->text().toStdString();
213
214 if ( AttributeName != SchemaAttribute->get_name() )
215 {
216 KernelWrapper::GetInstance().PushSetAttributeNameCommand ( SchemaClass, SchemaAttribute, AttributeName );
217 Changed = true;
218 }
219
220 if ( AttributeType != SchemaAttribute->get_type() )
221 {
222 KernelWrapper::GetInstance().PushSetAttributeTypeCommand ( SchemaClass, SchemaAttribute, AttributeType );
223 Changed = true;
224 }
225

228 KernelWrapper::GetInstance().PushSetAttributeFormatCommand ( SchemaClass,
229 SchemaAttribute,
230 AttributeFormat );
231 Changed = true;
232 }
233
234 if ( IsMultivariable != SchemaAttribute->get_is_multi_values() )

237 SchemaAttribute,
238 IsMultivariable );

240 }
241

249 {
250 KernelWrapper::GetInstance().PushSetAttributeDescriptionCommand ( SchemaClass,
251 SchemaAttribute,
252 AttributeDescription );

254 }
255
256 if ( AttributeRange != SchemaAttribute->get_range() )
257 {
258 KernelWrapper::GetInstance().PushSetAttributeRangeCommand ( SchemaClass,
259 SchemaAttribute,
260 AttributeRange );
261 Changed = true;
262 }
263
264 if ( AttributeInitial != SchemaAttribute->get_init_value() )
265 {
266 KernelWrapper::GetInstance().PushSetAttributeInitialValuesCommand ( SchemaClass,
267 SchemaAttribute,
268 AttributeInitial );
269 Changed = true;
270 }

272 if ( Changed )
