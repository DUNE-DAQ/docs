39std::vector<double>
40firwin(int N, double cutoff)
41{
42 int alpha = N / 2;
43 std::vector<double> window = hamming(N);
44 std::vector<double> ret;
45 double sum = 0;
46 for (int m = 0; m < N; ++m) {
47 double val = window[m] * sinc(cutoff * (m - alpha));
48 ret.push_back(val);
49 sum += val;
50 }
51 for (int m = 0; m < N; ++m) {
52 ret[m] /= sum;
53 }
54 return ret;
55}
