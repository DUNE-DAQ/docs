53make_parents_list(

56 std::vector<const dunedaq::confmodel::Component *> & p_list,

79static void
80make_parents_list(
81 const ConfigObjectImpl * child,
82 const dunedaq::confmodel::Segment * segment,
83 std::vector<const dunedaq::confmodel::Component *> & p_list,
84 std::list<std::vector<const dunedaq::confmodel::Component *> >& out,
85 bool is_segment,
86 dunedaq::confmodel::TestCircularDependency& cd_fuse)
87{
88 dunedaq::confmodel::AddTestOnCircularDependency add_fuse_test(cd_fuse, segment);
89
90 // add the segment to the path
91 p_list.push_back(segment);
92
93 // check if the application is in the nested segment
94 for (const auto& seg : segment->get_segments()) {
95 if (seg->config_object().implementation() == child)
96 out.push_back(p_list);
97 else
98 make_parents_list(child, seg, p_list, out, is_segment, cd_fuse);
99 }
100 if (!is_segment) {
101 for (const auto& app : segment->get_applications()) {
102 if (app->config_object().implementation() == child)
103 out.push_back(p_list);
104 else if (const auto resource_set = app->cast<dunedaq::confmodel::ResourceSet>())
105 make_parents_list(child, resource_set, p_list, out, cd_fuse);
106 }
107 }
108
109 // remove the segment from the path
110
111 p_list.pop_back();
112}

161 static std::vector<const Application*> getSegmentApps(const Segment* segment,
162 const Session* session,
163 bool enabled_only) {
164 std::vector<const Application*> apps;
165 auto segapps = segment->get_applications();
166 if (enabled_only) {
167 for (auto app : segapps) {
168 auto comp = app->cast<Component>();
169 if (comp == nullptr || !comp->disabled(*session)) {
170 apps.insert(apps.end(), app);
171 }
172 }
173 }
174 else {
175 apps.swap(segapps);
176 }
177 for (auto seg : segment->get_segments()) {
178 if (!enabled_only || !seg->disabled(*session)) {
179 auto segapps = getSegmentApps(seg, session, enabled_only);
180 apps.insert(apps.end(), segapps.begin(),segapps.end());
181 }
182 }
183 return apps;
184}

215nlohmann::json get_json_config(conffwk::Configuration& confdb,
216 const std::string& class_name,
217 const std::string& uid,
218 bool direct_only) {
219 using nlohmann::json;
220 using namespace conffwk;
221 TLOG_DBG(9) << "Getting attributes for " << uid << " of class " << class_name;
222 json attributes;
223 auto class_info = confdb.get_class_info(class_name);
224 ConfigObject obj;
225 confdb.get(class_name, uid, obj);
226 for (auto attr : class_info.p_attributes) {
227 if (attr.p_type == type_t::u8_type) {
228 add_json_value<uint8_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
229 }
230 else if (attr.p_type == type_t::u16_type) {
231 add_json_value<uint16_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
232 }
233 else if (attr.p_type == type_t::u32_type) {
234 add_json_value<uint32_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
235 }
236 else if (attr.p_type == type_t::u64_type) {
237 add_json_value<uint64_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
238 }
239 else if (attr.p_type == type_t::s8_type) {
240 add_json_value<int8_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
241 }
242 else if (attr.p_type == type_t::s16_type) {
243 add_json_value<int16_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
244 }
245 else if (attr.p_type == type_t::s32_type ||
246 attr.p_type == type_t::s16_type) {
247 add_json_value<int32_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
248 }
249 else if (attr.p_type == type_t::s64_type) {
250 add_json_value<int64_t>(obj, attr.p_name, attr.p_is_multi_value, attributes);
251 }
252 else if (attr.p_type == type_t::float_type) {
253 add_json_value<float>(obj, attr.p_name, attr.p_is_multi_value, attributes);
254 }
255 else if (attr.p_type == type_t::double_type) {
256 add_json_value<double>(obj, attr.p_name, attr.p_is_multi_value, attributes);
257 }
258 else if (attr.p_type == type_t::bool_type) {
259 add_json_value<bool>(obj, attr.p_name, attr.p_is_multi_value, attributes);
260 }
261 else if ((attr.p_type == type_t::string_type) ||
262 (attr.p_type == type_t::enum_type) ||
263 (attr.p_type == type_t::date_type) ||
264 (attr.p_type == type_t::time_type)) {
265 add_json_value<std::string>(obj, attr.p_name, attr.p_is_multi_value, attributes);
266 }
267 }
268 if (!direct_only) {
269 TLOG_DBG(9) << "Processing relationships";
270 for (auto iter: class_info.p_relationships) {
271 std::string rel_name = iter.p_name;
272 if (iter.p_cardinality == cardinality_t::zero_or_one ||
273 iter.p_cardinality == cardinality_t::only_one) {
274 ConfigObject rel_obj;
275 obj.get(rel_name, rel_obj);
276 if (!rel_obj.is_null()) {
277 TLOG_DBG(9) << "Getting attibute of relationship " << rel_name;
278 attributes[rel_name] = get_json_config(confdb, rel_obj.class_name(),
279 rel_obj.UID(),
280 direct_only);
281 }
282 else {
283 TLOG_DBG(9) << "Relationship " << rel_name << " not set";
284 }
285 }
286 else {
287 TLOG_DBG(9) << "Relationship " << rel_name << " is multi value. "
288 << "Getting attibutes for relationship.";
289 std::vector<ConfigObject> rel_vec;
290 obj.get(rel_name, rel_vec);
291 std::vector<json> configs;
292 for (auto rel_obj : rel_vec) {
293 TLOG_DBG(9) << "Getting attibute of relationship " << rel_obj.UID();
294 auto rel_conf = get_json_config(confdb, rel_obj.class_name(), rel_obj.UID(),
295 direct_only);
296 configs.push_back(rel_conf);
297 }
298 attributes[rel_name] = configs;
299 }
300 }
301 }
302 json json_config;
303 json_config[uid] = attributes;
304 return json_config;
305}

351std::vector<const confmodel::DetDataSender*> DetectorToDaqConnection::get_senders() const {
352 std::vector<const confmodel::DetDataSender*> senders;
353
354 for ( auto d2d_res : this->get_contains() ) {
355 // Maybe senders not in a resource set so check for direct containment
356 auto sender = d2d_res->cast<confmodel::DetDataSender>();
357 if ( sender != nullptr ) {
358 senders.push_back(sender);
359 }
360 else {
361 // Look for a resource set containing senders
362 auto rs = d2d_res->cast<confmodel::ResourceSet>();
363 if (rs != nullptr) {
364 // Look for senders in resource set
365 for (auto res : rs->get_contains()) {
366 auto sender = res->cast<confmodel::DetDataSender>();
367 if ( sender != nullptr ) {
368 senders.push_back(sender);
369 }
370 }
371 }
372 }
373 }
374
375 return senders;
376}

379const confmodel::DetDataReceiver* DetectorToDaqConnection::get_receiver() const {
380
381 std::vector<const confmodel::DetDataReceiver*> receivers;
382
383 for ( auto d2d_res : this->get_contains() ) {
384 auto r = d2d_res->cast<confmodel::DetDataReceiver>();
385 if ( r == nullptr )
386 continue;
387
388 receivers.push_back(r);
389 }
390
391 if (receivers.size() != 1) {
392 throw(ConfigurationError(ERS_HERE, "DetectorToDaqConnection : expected 1 receiver in D2d conection {name of connection}, found {number found}"));

394
395 // Receiver identified

401std::vector<const confmodel::DetectorStream*> DetectorToDaqConnection::get_streams() const {
402
403 std::vector<const confmodel::DetectorStream*> streams;
404 // Loop over senders

412
413 streams.push_back(stream->cast<confmodel::DetectorStream>());
414 }
415 }
416
417 return streams;
