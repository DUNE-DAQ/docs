104 // Set the initial state from the window starting at first_msg_p
105 template<size_t N>
106 void setState(const RegisterArray<N>& first_tick_registers)
107 {
108 static_assert(N >= NREGISTERS, "Wrong array size");
109
110 // AAA: Loop through all the registers, loop through all the channels, look at the
111 // first message of the superchunk and read the ADC value. This will be used as the
112 // pedestal for the channel state
113 for (size_t j = 0; j < NREGISTERS * SAMPLES_PER_REGISTER; ++j) {
114 const size_t register_offset = j % SAMPLES_PER_REGISTER;
115 const size_t register_index = j / SAMPLES_PER_REGISTER;
116 const size_t register_t0_start = register_index * SAMPLES_PER_REGISTER * FRAMES_PER_MSG;
117
118 int16_t ped=0;
119 for (size_t itime = 0; itime < timeWindowNumFrames; ++itime) {
120 const size_t msg_index = itime / timeWindowNumFrames;
121 const size_t msg_time_offset = itime % timeWindowNumFrames;
122 // The index in uint16_t of the start of the message we want.
123 const size_t msg_start_index = msg_index * (swtpg_wibeth::ADCS_SIZE) / sizeof(uint16_t); // NOLINT
124 const size_t offset_within_msg = register_t0_start + SAMPLES_PER_REGISTER * msg_time_offset + register_offset;
125 const size_t index = msg_start_index + offset_within_msg;
126
127 const uint16_t* input16 = first_tick_registers.data(); // NOLINT
128
129 ped = input16[index];
130
131 break; // breaking in order to select only the first entry
132 }
133
134 // Set the pedestals and the 25/75-percentiles
135 chanState.pedestals[j] = ped;
136 chanState.pedestalsRS[j] = 0;
137 chanState.RS[j] = 0;
138 // AAA: Quantiles are set to the pedestal value +/- 20 so that the IQR
139 // becomes above the RMS value of the input ADCs. We use the frugal
140 // streaming on the 25th/75th quantiles so that the IQR becomes
141 // a good estimate of the RMS of the input.
142 chanState.quantile25[j] = ped-20;
143 chanState.quantile75[j] = ped+20;
144 }
145 }

148 size_t timeWindowNumFrames;
149 uint8_t first_register; // NOLINT
150 uint8_t last_register; // NOLINT
151 uint16_t* __restrict__ output; // NOLINT
152 const int16_t* __restrict__ taps;
153 int16_t ntaps;
154 uint8_t tap_exponent; // NOLINT
155 uint16_t threshold; // NOLINT
156 int16_t multiplier;
157 int16_t adcMax;
158 size_t nhits;
159 uint16_t absTimeModNTAPS; // NOLINT
160 ChanState<NREGISTERS> chanState;
161};
