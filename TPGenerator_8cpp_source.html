29__m256i
30TPGenerator::expand_frame(const __m256i& regi) {
31 // Refer to the diagram and documentation on frame expansion for details.
32
33 // Prepare even (2,4,6,8), odd (1,3,5,7) rows in 64-bit sense.
34 __m256i odd = _mm256_permutevar8x32_epi32(regi, _mm256_setr_epi32(1, 0, 1, 2, 3, 4, 5, 6));
35
36 // Shift into place.
37 __m256i even = _mm256_sllv_epi64(regi, _mm256_setr_epi64x(6, 14, 22, 30));
38 odd = _mm256_srlv_epi64(odd, _mm256_setr_epi64x(30, 22, 14, 6));
39
40 // Everything is center aligned in 32-bit. Mask and right-align the right side.
41 __m256i both = _mm256_blend_epi32(even, odd, 0b01010101);
42 __m256i right = _mm256_and_si256(_mm256_set1_epi32(0xFFFFu), both);
43 __m256i left = _mm256_and_si256(_mm256_set1_epi32(0x3FFF0000u), both);
44
45 right = _mm256_srli_epi32(right, 2);
46 return _mm256_or_si256(left, right);
47}

49__m256i
50TPGenerator::old_expand_frame(const __m256i& regi) {
51 // Refer to the diagram and documentation on frame expansion for details.
52
53 // Rearrange original with row 3 doubled.
54 __m256i idx = _mm256_set_epi32(6, 5, 4, 3, 3, 2, 1, 0);
55 __m256i shuf1 = _mm256_permutevar8x32_epi32(regi, idx);
56
57 // Left shift each row.
58 __m256i count = _mm256_set_epi32(12, 8, 4, 0, 14, 10, 6, 2);
59 __m256i high_half = _mm256_sllv_epi32(shuf1, count);
60 high_half = _mm256_and_si256(high_half, _mm256_set1_epi32(0x3FFF0000u)); // Mask out the low half.
61
62 // Left shift for low half later.
63 count = _mm256_set_epi32(10, 6, 2, 0, 12, 8, 4, 0);
64 __m256i shift2 = _mm256_sllv_epi32(shuf1, count);
65
66 // Rearrange original and doubled rows 2 and 0.
67 idx = _mm256_set_epi32(5, 4, 3, 2, 2, 1, 0, 0);
68 __m256i shuf2 = _mm256_permutevar8x32_epi32(regi, idx);
69
70 // Right shift each row.
71 count = _mm256_set_epi32(22, 26, 30, 0, 20, 24, 28, 0);
72 __m256i shift3 = _mm256_srlv_epi32(shuf2, count);
73
74 // "Complete" the low half. Still more.
75 __m256i low_half = _mm256_or_si256(shift2, shift3);
76 low_half = _mm256_and_si256(low_half, _mm256_set1_epi32(0x3FFFu)); // Mask out the high half.
77
78 // Combine halves and clear space for an odd entry.
79 __m256i both = _mm256_or_si256(low_half, high_half);
80 both = _mm256_andnot_si256(_mm256_set_epi32(0, 0, 0, 0xFFFFu, 0, 0, 0, 0), both);
81
82 // There is a specific 16-bit entry that needs special handling.
83 // Align it.
84 __m256i shift4 = _mm256_srli_epi32(regi, 18);
85 // Mask it.
86 shift4 = _mm256_and_si256(_mm256_set_epi32(0, 0x3FFFu, 0, 0, 0, 0, 0, 0), shift4);
87
88 // Permute into the right spot
89 idx = _mm256_set_epi32(0, 0, 0, 6, 0, 0, 0, 0);
90 __m256i shuf3 = _mm256_permutevar8x32_epi32(shift4, idx);
91
92 // Add it in.
93 both = _mm256_or_si256(both, shuf3);
94 return both;
95}
