169void dbse::SchemaMainWindow::RemoveClass()
170{
171 QModelIndex Index = ui->ClassTableView->currentIndex();
172 QModelIndex proxyIndex = proxyModel->mapToSource( Index );
173 QStringList Row = TableModel->getRowFromIndex ( proxyIndex );
174 OksClass * SchemaClass = KernelWrapper::GetInstance().FindClass ( Row.at (
175 0 ).toStdString() );
176
177 if ( SchemaClass->all_sub_classes()->size() != 0 )
178 {
179 QMessageBox::warning ( 0, "Schema editor",
180 QString ( "Cannot delete class because it has sub-classes." ) );
181 return;
182 }
183 else if ( KernelWrapper::GetInstance().AnyClassReferenceThis ( SchemaClass ) )
184 {
185 QMessageBox::warning ( 0, "Schema editor",
186 QString ( "Cannot delete class because some classes references it." ) );
187 return;
188 }
189 else
190 {
191 QString Name = QString::fromStdString ( SchemaClass->get_name() );
192 KernelWrapper::GetInstance().PushRemoveClassCommand ( SchemaClass, SchemaClass->get_name(),
193 SchemaClass->get_description(),
194 SchemaClass->get_is_abstract() );
195 BuildFileModel();
196 }
197}

408void dbse::SchemaMainWindow::SaveView()

410 SchemaTab * CurrentTab = dynamic_cast<SchemaTab *> ( ui->TabWidget->currentWidget() );
411
412 if ( CurrentTab->GetScene()->items().size() != 0 )
413 {
414 QString FileName = QFileDialog::getSaveFileName ( this, tr ( "Save View" ) );

417 {
418 FileName.append ( ".view" );

422 ViewFile.open ( QIODevice::WriteOnly );
423
424 for ( QGraphicsItem * Item : CurrentTab->GetScene()->items() )

426 if ( dynamic_cast<SchemaGraphicObject *> ( Item ) )

429 QString ObjectDescription = SchemaObject->GetClassName() + ","
430 + QString::number ( SchemaObject->scenePos().x() ) + ","

433 }

440void dbse::SchemaMainWindow::LoadView()
441{
442 QFileDialog FileDialog ( this, tr ( "Open File" ), ".", tr ( "View files (*.view)" ) );
443 FileDialog.setAcceptMode ( QFileDialog::AcceptOpen );
444 FileDialog.setFileMode ( QFileDialog::AnyFile );
445 FileDialog.setViewMode ( QFileDialog::Detail );
446 QStringList FilesSelected;
447 QString ViewPath;
448
449 if ( FileDialog.exec() )
450 {
451 FilesSelected = FileDialog.selectedFiles();
452 }
453
454 if ( FilesSelected.size() )
455 {
456 ViewPath = FilesSelected.value ( 0 );
457 }
458
459 if ( !ViewPath.isEmpty() )
460 {
461 SchemaTab * CurrentTab = dynamic_cast<SchemaTab *> ( ui->TabWidget->currentWidget() );
462
463 if ( CurrentTab->GetScene()->items().size() != 0 )
464 {
465 int UserAnswer =
466 QMessageBox::question (
467 0,
468 tr ( "DBE" ),
469 QString (
470 "If you load this view the current view will be lost.\n\nDo you want to load it anyway?\n" ),
471 QMessageBox::Save | QMessageBox::Cancel, QMessageBox::Save );
472
473 if ( UserAnswer == QMessageBox::Cancel )
474 {
475 return;
476 }
477 }
478
479 CurrentTab->GetScene()->clear();
480
481 QFile ViewFile ( ViewPath );
482 ViewFile.open ( QIODevice::ReadOnly );
483
484 QStringList ClassesNames;
485 QList<QPointF> Positions;
486
487 while ( !ViewFile.atEnd() )
488 {
489 QString Line ( ViewFile.readLine() );
490 QStringList ObjectDescription = Line.split ( "," );
491 ClassesNames.append ( ObjectDescription.at ( 0 ) );
492 QPointF Position;
493 Position.setX ( ObjectDescription.at ( 1 ).toInt() );
494 Position.setY ( ObjectDescription.at ( 2 ).toInt() );
495 Positions.append ( Position );
496 }
497
498 CurrentTab->GetScene()->CleanItemMap();
499 CurrentTab->GetScene()->AddItemToScene ( ClassesNames, Positions );
500 ViewFile.close();
501 }

505{
506 QModelIndex proxyIndex = proxyModel->mapToSource( Index );
507 QStringList Row = TableModel->getRowFromIndex ( proxyIndex );
508
509 if ( !Row.isEmpty() )
510 {
511 bool WidgetFound = false;
512 QString ClassName = Row.at ( 0 );
513 OksClass * ClassInfo = KernelWrapper::GetInstance().FindClass ( ClassName.toStdString() );
514
515 for ( QWidget * Editor : QApplication::allWidgets() )
516 {
517 SchemaClassEditor * Widget = dynamic_cast<SchemaClassEditor *> ( Editor );
518
519 if ( Widget != nullptr )
520 {
521 if ( ( Widget->objectName() ).compare ( ClassName ) == 0 )
522 {
523 Widget->raise();
524 Widget->setVisible ( true );
525 Widget->activateWindow();
526 WidgetFound = true;
527 }
528 }
529 }
530
531 if ( !WidgetFound )
532 {
533 SchemaClassEditor * Editor = new SchemaClassEditor ( ClassInfo );
534 Editor->show();
535 }
536 }
