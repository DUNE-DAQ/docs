42void
43DAQModuleManager::initialize(std::shared_ptr<ConfigurationManager> cfgMgr, opmonlib::OpMonManager& opm)
44{
45 m_module_configuration = std::make_shared<ModuleConfiguration>(cfgMgr);
46 get_iomanager()->configure(cfgMgr->session()->UID(),
47 m_module_configuration->queues(),
48 m_module_configuration->networkconnections(),
49 m_module_configuration->connectivity_service(),
50 opm);
51 init_modules(m_module_configuration->modules(), opm);
52
53 for (auto& plan_pair : m_module_configuration->action_plans()) {
54 auto cmd = plan_pair.first;
55
56 for (auto& step : plan_pair.second->get_steps()) {
57 auto byType = step->cast<confmodel::DaqModulesGroupByType>();
58 auto byMod = step->cast<confmodel::DaqModulesGroupById>();
59 if (byType != nullptr) {
60 for (auto& mod_type : byType->get_modules()) {
61 check_mod_has_cmd(cmd, mod_type);
62 }
63 } else if (byMod != nullptr) {
64 for (auto& mod : byMod->get_modules()) {
65 check_mod_has_cmd(cmd, mod->class_name(), mod->UID());
66 }
67 } else {
68 throw ActionPlanValidationFailed(ERS_HERE, cmd, "", "Invalid subclass of DaqModulesGroup encountered!");
69 }
70 }
71 }
72 this->m_initialized = true;
73}

75void
76DAQModuleManager::check_mod_has_cmd(const std::string& cmd, const std::string& mod_class, const std::string& mod_id)
77{
78
79 if (!m_modules_by_type.count(mod_class) || m_modules_by_type[mod_class].size() == 0) {
80 auto issue = ActionPlanValidationFailed(ERS_HERE, cmd, mod_class, "Module does not exist");
81 if (mod_id == "") {
82 ers::info(issue);
83 return;
84 } else {
85 throw issue;
86 }
87 }
88
89 auto module_test = m_module_map[m_modules_by_type[mod_class][0]];
90 if (mod_id != "") {
91 bool match = false;
92 for (auto& mod_name : m_modules_by_type[mod_class]) {
93 if (mod_id == mod_name) {
94 module_test = m_module_map[mod_name];
95 match = true;
96 break;
97 }
98 }
99 if (!match) {
100 throw ActionPlanValidationFailed(ERS_HERE, cmd, mod_class, "No module with id " + mod_id + " found.");
101 }
102 }
103
104 if (!module_test->has_command(cmd)) {
105 throw ActionPlanValidationFailed(ERS_HERE, cmd, mod_class, "Module does not have method " + cmd);
106 }

172void
173DAQModuleManager::execute_action_plan_step(std::string const& cmd,
174 const confmodel::DaqModulesGroup* step,
175 const dataobj_t& cmd_data,
176 bool execution_mode_is_serial)
177{
178 std::string failed_mod_names("");
179 std::unordered_map<std::string, std::future<bool>> futures;
180
181 auto byType = step->cast<confmodel::DaqModulesGroupByType>();
182 auto byMod = step->cast<confmodel::DaqModulesGroupById>();
183 if (byType != nullptr) {
184 for (auto& mod_class : byType->get_modules()) {
185 auto modules = m_modules_by_type[mod_class];
186 for (auto& mod_name : modules) {
187 auto data_obj = get_dataobj_for_module(mod_name, cmd_data);
188 TLOG_DEBUG(1) << "Executing action " << cmd << " on module " << mod_name << " (class " << mod_class << ")";
189 futures[mod_name] =
190 std::async(std::launch::async, &DAQModuleManager::execute_action, this, mod_name, cmd, data_obj);
191 if (execution_mode_is_serial)
192 futures[mod_name].wait();
193 }
194 }
195 } else if (byMod != nullptr) {
196 for (auto& mod : byMod->get_modules()) {
197 auto mod_name = mod->UID();
198 auto data_obj = get_dataobj_for_module(mod_name, cmd_data);
199 TLOG_DEBUG(1) << "Executing action " << cmd << " on module " << mod_name << " (class " << mod->class_name()
200 << ")";
201 futures[mod_name] =
202 std::async(std::launch::async, &DAQModuleManager::execute_action, this, mod_name, cmd, data_obj);
203 if (execution_mode_is_serial)
204 futures[mod_name].wait();
205 }
206 } else {
207 throw CommandDispatchingFailed(ERS_HERE, cmd, "Could not get DaqModulesGroup!");
208 }
209
210 for (auto& future : futures) {
211 future.second.wait();
212 auto ret = future.second.get();
213 if (!ret) {
214 failed_mod_names.append(future.first);
215 failed_mod_names.append(", ");
216 }
217 }
218 // Throw if any dispatching failed
219 if (!failed_mod_names.empty()) {
220 throw CommandDispatchingFailed(ERS_HERE, cmd, failed_mod_names);
221 }
222}

237void
238DAQModuleManager::check_cmd_data(const std::string& id, const dataobj_t& cmd_data)
239{
240 // This method ensures that each module is only matched once per command.
241 // If multiple matches are found, an ers::Issue is thrown
242 // Disclaimenr for the occasional reader: this is the first implementation of the
243 // multiple-matches detection logic. The author is painfully aware that it can be
244 // vastly improved, in style if not in performance.
245
246 auto cmd_obj = cmd_data.get<cmd::CmdObj>();
247 const dataobj_t dummy{};
248
249 // Make a convenience array with module names that have the requested command
250 std::vector<std::string> cmd_mod_names = get_modnames_by_cmdid(id);
251
252 // containers for error tracking
253 std::map<std::string, std::vector<std::string>> mod_to_re;
254
255 if (!cmd_obj.modules.empty()) {
256 for (const auto& addressed : cmd_obj.modules) {
257 if (!addressed.match.empty()) {
258 // Find module names matching the regex
259 for (const std::string& mod_name : cmd_mod_names) {
260 // match module name with regex
261 if (std::regex_match(mod_name, std::regex(addressed.match))) {
262 mod_to_re[mod_name].push_back(addressed.match);
263 }
264 }
265 }
266 }
267
268 // Select modules with multiple matches
269 for (auto i = mod_to_re.begin(), last = mod_to_re.end(); i != last;) {
270 if (i->second.size() == 1) {
271 i = mod_to_re.erase(i);
272 } else {
273 ++i;
274 }
275 }
276
277 // Catch cases
278 if (mod_to_re.size() > 0) {
279 std::string mod_names;
280 for (const auto& [mod_name, matched_re] : mod_to_re) {
281 mod_names += mod_name + ", ";
282 }
283 throw ConflictingCommandMatching(ERS_HERE, id, mod_names);
284 }
285 }
286}

288void
289DAQModuleManager::execute(const std::string& cmd, const dataobj_t& cmd_data)
290{
291
292 TLOG_DEBUG(1) << "Command id:" << cmd;
293
294 if (!m_initialized) {
295 throw DAQModuleManagerNotInitialized(ERS_HERE, cmd);
296 }
297
298 check_cmd_data(cmd, cmd_data);
299
300 auto action_plan = m_module_configuration->action_plan(cmd);
301 if (action_plan == nullptr) {
302#if 0
303 throw ActionPlanNotFound(ERS_HERE, cmd, "Throwing exception");
304#elif 0
305 ers::warning(ActionPlanNotFound(ERS_HERE, cmd, "Returning without executing actions"));
306 return;
307#else
308 // Emulate old behavior
309 TLOG_DEBUG(1) << ActionPlanNotFound(ERS_HERE, cmd, "Executing action on all modules in parallel");
310 std::string failed_mod_names("");
311 std::unordered_map<std::string, std::future<bool>> futures;
312
313 auto mods = get_modnames_by_cmdid(cmd);
314 for (auto& mod : mods) {
315 TLOG_DEBUG(1) << "Executing action " << cmd << " on module " << mod;
316 auto data_obj = get_dataobj_for_module(mod, cmd_data);
317 futures[mod] = std::async(std::launch::async, &DAQModuleManager::execute_action, this, mod, cmd, data_obj);
318 }
319
320 for (auto& future : futures) {
321 future.second.wait();
322 auto ret = future.second.get();
323 if (!ret) {
324 failed_mod_names.append(future.first);
325 failed_mod_names.append(", ");
326 }
327 }
328 // Throw if any dispatching failed
329 if (!failed_mod_names.empty()) {
330 throw CommandDispatchingFailed(ERS_HERE, cmd, failed_mod_names);
331 }
332#endif
333 } else {
334 auto execution_policy = action_plan->get_execution_policy();
335 auto serial_execution = execution_policy == "modules-in-series";
336
337 // We validated the action plans already
338 for (auto& step : action_plan->get_steps()) {
339 execute_action_plan_step(cmd, step, cmd_data, serial_execution);
340 }
341 }
342
343 // Shutdown IOManager at scrap
344 if (cmd == "scrap") {
345 get_iomanager()->shutdown();
346 }
347}
