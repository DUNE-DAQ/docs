247void
248dunedaq::sspmodules::EthernetDevice::SendReceive(dunedaq::sspmodules::CtrlPacket& tx,
249 dunedaq::sspmodules::CtrlPacket& rx,
250 unsigned int txSize,
251 unsigned int rxSizeExpected,
252 unsigned int retryCount)
253{
254 unsigned int timesTried = 0;
255 bool success = false;
256
257 while (!success) {
258 try {
259 SendEthernet(tx, txSize);
260 // Insert small delay between send and receive on Linux
261 usleep(100);
262 ReceiveEthernet(rx, rxSizeExpected);
263 usleep(2000);
264 success = true;
265 } catch (ETCPError&) {
266 if (timesTried < retryCount) {
267 DevicePurgeComm();
268 ++timesTried;
269 // dune::DAQLogger::LogWarning("SSP_EthernetDevice")<<"Send/receive failed "<<timesTried<<" times on Ethernet
270 // link, retrying..."<<std::endl;
271 } else {
272 // dune::DAQLogger::LogError("SSP_EthernetDevice")<<"Send/receive failed on Ethernet link, giving
273 // up."<<std::endl;
274 throw;
275 }
276 }
277 }

298void
299dunedaq::sspmodules::EthernetDevice::DevicePurge(boost::asio::ip::tcp::socket& socket)
300{
301 bool done = false;
302 unsigned int bytesQueued = 0;
303 unsigned int sleepTime = 0;
304
305 // Keep getting data from channel until queue is empty
306 do {
307 bytesQueued = socket.available();
308
309 // Read data from device, up to 256 bytes
310 if (bytesQueued != 0) {
311 sleepTime = 0;
312 unsigned int bytesToGet = std::min((unsigned int)256, bytesQueued);
313 std::vector<char> junkBuf(bytesToGet);
314 socket.read_some(boost::asio::buffer(junkBuf, bytesToGet));
315 } else { // If queue is empty, wait a bit and check that it hasn't filled up again, then return
316 usleep(1000); // 1ms
317 sleepTime += 1000;
318 bytesQueued = socket.available();
319 if (bytesQueued == 0 && sleepTime > 1000000) {
320 done = 1;
321 }
322 }
323 } while (!done);
324}
