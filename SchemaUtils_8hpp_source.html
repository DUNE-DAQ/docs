89inline std::string
90get_uri_for_connection(const confmodel::NetworkConnection* netCon)
91{
92 std::string uri = "";
93 if (netCon) {
94 TLOG() << "Getting URI for network connection " << netCon->UID();
95 auto service = netCon->get_associated_service();
96 if (service->get_protocol() == "tcp") {
97
98 std::string port = "*";
99 if (service->get_port() && service->get_port() != 0) {
100 port = std::to_string(service->get_port());
101 }
102 std::string ipaddr = "0.0.0.0";
103 char hostname[256];
104 if (gethostname(&hostname[0], 256) == 0) {
105 ipaddr = std::string(hostname);
106 }
107 auto iface = service->get_eth_device_name();
108 if (iface != "") {
109 // Work out which ip address goes with this device
110 struct ifaddrs* ifaddr;
111 getifaddrs(&ifaddr);
112 for (auto ifa = ifaddr; ifa != NULL; ifa = ifa->ifa_next) {
113 if (ifa->ifa_addr == NULL) {
114 continue;
115 }
116 if (std::string(ifa->ifa_name) == iface) {
117 char ip[NI_MAXHOST];
118 int status =
119 getnameinfo(ifa->ifa_addr, sizeof(struct sockaddr_in), ip, NI_MAXHOST, NULL, 0, NI_NUMERICHOST);
120 if (status != 0) {
121 continue;
122 }
123 ipaddr = std::string(ip);
124 break;
125 }
126 }
127 }
128 uri = std::string(service->get_protocol() + "://" + ipaddr + ":" + port);
129 } else if (service->get_protocol() == "inproc") {
130 uri = std::string(service->get_protocol() + "://" + service->get_path());
131 }
132 }
133 return uri;
134}
