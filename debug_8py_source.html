271 def asciidecode( v ):
272 return (''.join([chr(c) for c in v])).rstrip()
273
274 def tempdecode( v ):
275 lSign = -1 if ((v[0]>>7) & 0x1) else 1
276 x = lSign*((v[0] & 0x7f) + v[1]/float(0xff))
277 return '{:.3f} C'.format(x)
278
279 def vccdecode( v ):
280 return '{:.4f} V'.format(((v[0] << 8) + v[1])/float(10000))
281
282 def biascurdecode( v ):
283 return '{:.3f} mA'.format(((v[0] << 8) + v[1])*2e-3)
284
285 def powerdecode( v ):
286 return '{:.3f} mW'.format(((v[0] << 8) + v[1])*1e-3)
287
