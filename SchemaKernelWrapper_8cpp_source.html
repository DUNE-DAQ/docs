144bool dbse::KernelWrapper::AnyClassReferenceThis ( OksClass * SchemaClass )
145{
146 std::vector<OksClass *> ClassList;
147 GetClassList ( ClassList );
148
149 for ( OksClass * ClassInfo : ClassList )
150 {
151 const std::list<OksRelationship *> * RelationshipList = ClassInfo->direct_relationships();
152
153 if ( RelationshipList != nullptr )
154 {
155 for ( OksRelationship * RelationshipInfo : *RelationshipList )
156 {
157 if ( RelationshipInfo->get_class_type()->get_name() == SchemaClass->get_name() )
158 {
159 return true;
160 }
161 }
162 }
163 }
164
165 return false;
166}

168QString dbse::KernelWrapper::GetCardinalityStringRelationship (
169 OksRelationship * SchemaRelationship ) const
170{
171 QString RelationshipLowCc (
172 OksRelationship::card2str ( SchemaRelationship->get_low_cardinality_constraint() ) );
173 QString RelationshipHighCc (
174 OksRelationship::card2str ( SchemaRelationship->get_high_cardinality_constraint() ) );
175
176 if ( RelationshipLowCc == "zero" )
177 {
178 RelationshipLowCc = "0";
179 }
180 else if ( RelationshipLowCc == "one" )
181 {
182 RelationshipLowCc = "1";
183 }
184 else if ( RelationshipLowCc == "many" )
185 {
186 RelationshipLowCc = "n";
187 }
188
189 if ( RelationshipHighCc == "zero" )
190 {
191 RelationshipHighCc = "0";
192 }
193 else if ( RelationshipHighCc == "one" )
194 {
195 RelationshipHighCc = "1";
196 }
197 else if ( RelationshipHighCc == "many" )
198 {
199 RelationshipHighCc = "n";
200 }
201
202 return QString ( RelationshipLowCc + ':' + RelationshipHighCc );
203}
