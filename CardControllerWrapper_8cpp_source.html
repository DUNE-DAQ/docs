197void
198CardControllerWrapper::check_alignment(const felixcardcontroller::LogicalUnit & lu_cfg, const uint64_t & aligned)
199{
200 TLOG_DEBUG(TLVL_WORK_STEPS) << "Checking link alignment for " << lu_cfg.log_unit_id;
201 std::vector<uint32_t> alignment_mask = lu_cfg.ignore_alignment_mask;
202 // check the alingment on a logical unit
203 for(auto li : lu_cfg.links) {
204 // here we want to print out a log message when the links do not appear to be aligned.
205 // for WIB readout link_id 5 is always reserved for tp links, so alignemnt is not expected fort these
206 bool is_aligned = aligned & (1<<li.link_id);
207 auto found_link = std::find(std::begin(alignment_mask), std::end(alignment_mask), li.link_id);
208 if(found_link == std::end(alignment_mask)) {
209 if(!lu_cfg.emu_fanout && !is_aligned) {
210 ers::error(flxlibs::ChannelAlignment(ERS_HERE, li.link_id));
211 }
212 }
213 }
