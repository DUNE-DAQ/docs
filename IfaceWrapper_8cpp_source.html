213//-----------------------------------------------------------------------------
214void
215IfaceWrapper::setup_flow_steering()
216{
217 // Flow steering setup
218 TLOG() << "Configuring Flow steering rules for iface=" << m_iface_id;
219 struct rte_flow_error error;
220 struct rte_flow *flow;
221 TLOG() << "Attempt to flush previous flow rules...";
222 rte_flow_flush(m_iface_id, &error);
223#warning RS: FIXME -> Check for flow flush return!
224 for (auto const& [lcoreid, rxqs] : m_rx_core_map) {
225 for (auto const& [rxqid, srcip] : rxqs) {
226 // Put the IP numbers temporarily in a vector, so they can be converted easily to uint32_t
227 TLOG() << "Creating flow rule for src_ip=" << srcip << " assigned to rxq=" << rxqid;
228 size_t ind = 0, current_ind = 0;
229 std::vector<uint8_t> v;
230 for (int i = 0; i < 4; ++i) {
231 v.push_back(std::stoi(srcip.substr(current_ind, srcip.size() - current_ind), &ind));
232 current_ind += ind + 1;
233 }
234
235 flow = generate_ipv4_flow(m_iface_id, rxqid,
236 RTE_IPV4(v[0], v[1], v[2], v[3]), 0xffffffff, 0, 0, &error);
237
238 if (not flow) { // ers::fatal
239 TLOG() << "Flow can't be created for " << rxqid
240 << " Error type: " << (unsigned)error.type
241 << " Message: " << error.message;
242 ers::fatal(dunedaq::datahandlinglibs::InitializationError(
243 ERS_HERE, "Couldn't create Flow API rules!"));
244 rte_exit(EXIT_FAILURE, "error in creating flow");
245 }
246 }
247 }
248
249 return;
250}

310//-----------------------------------------------------------------------------
311void
312IfaceWrapper::generate_opmon_data() {
313
314 opmon::EthStats s;
315 s.set_ipackets( m_iface_xstats.m_eth_stats.ipackets );
316 s.set_opackets( m_iface_xstats.m_eth_stats.opackets );
317 s.set_ibytes( m_iface_xstats.m_eth_stats.ibytes );
318 s.set_obytes( m_iface_xstats.m_eth_stats.obytes );
319 s.set_imissed( m_iface_xstats.m_eth_stats.imissed );
320 s.set_ierrors( m_iface_xstats.m_eth_stats.ierrors );
321 s.set_oerrors( m_iface_xstats.m_eth_stats.oerrors );
322 s.set_rx_nombuf( m_iface_xstats.m_eth_stats.rx_nombuf );
323 publish( std::move(s) );
324
325 // Poll stats from HW
326 m_iface_xstats.poll();
327
328 // loop over all the xstats information
329 opmon::EthXStatsInfo xinfos;
330 opmon::EthXStatsErrors xerrs;
331 std::map<std::string, opmon::QueueEthXStats> xq;
332
333 for (int i = 0; i < m_iface_xstats.m_len; ++i) {
334
335 std::string name(m_iface_xstats.m_xstats_names[i].name);
336
337 // first we select the info from the queue
338 static std::regex queue_regex(R"((rx|tx)_q(\d+)_([^_]+))");
339 std::smatch match;
340
341 if ( std::regex_match(name, match, queue_regex) ) {
342 auto queue_name = match[1].str() + '-' + match[2].str();
343 auto & entry = xq[queue_name];
344 try {
345 opmonlib::set_value( entry, match[3], m_iface_xstats.m_xstats_values[i] );
346 } catch ( const ers::Issue & e ) {
347 ers::warning( MetricPublishFailed( ERS_HERE, name, e) );
348 }
349 continue;
350 }
351
352 google::protobuf::Message * metric_p = nullptr;
353 static std::regex err_regex(R"(.+error.*)");
354 if ( std::regex_match( name, err_regex ) ) metric_p = & xerrs;
355 else metric_p = & xinfos;
356
357 try {
358 opmonlib::set_value(*metric_p, name, m_iface_xstats.m_xstats_values[i]);
359 } catch ( const ers::Issue & e ) {
360 ers::warning( MetricPublishFailed( ERS_HERE, name, e) );
361 }
362
363 } // loop over xstats
364
365 // Reset HW counters
366 m_iface_xstats.reset_counters();
367
368 // finally we publish the information
369 publish( std::move(xinfos) );
370 publish( std::move(xerrs) );
371 for ( auto [id, stat] : xq ) {
372 publish( std::move(stat), {{"queue", id}} );
373 }
374
375 for( const auto& [src_rx_q,_] : m_num_frames_rxq) {
376 opmon::QueueInfo i;
377 i.set_packets_received( m_num_frames_rxq[src_rx_q].load() );
378 i.set_bytes_received( m_num_bytes_rxq[src_rx_q].load() );
379 i.set_full_rx_burst( m_num_full_bursts[src_rx_q].load() );
380 i.set_max_burst_size( m_max_burst_size[src_rx_q].exchange(0) );
381
382 publish( std::move(i), {{"queue", std::to_string(src_rx_q)}} );
383 }
384}

401//-----------------------------------------------------------------------------
402void
403IfaceWrapper::handle_eth_payload(int src_rx_q, char* payload, std::size_t size)
404{
405 // Get DAQ Header and its StreamID
406 auto* daq_header = reinterpret_cast<dunedaq::detdataformats::DAQEthHeader*>(payload);
407 auto src_id = m_stream_id_to_source_id[src_rx_q][(unsigned)daq_header->stream_id];
408
409 if ( auto src_it = m_sources.find(src_id); src_it != m_sources.end()) {
410 src_it->second->handle_payload(payload, size);
411 } else {
412 // Really bad -> unexpeced StreamID in UDP Payload.
413 // This check is needed in order to avoid dynamically add thousands
414 // of Sources on the fly, in case the data corruption is extremely severe.
415 if (m_num_unexid_frames.count(src_id) == 0) {
416 m_num_unexid_frames[src_id] = 0;
417 }
418 m_num_unexid_frames[src_id]++;
419 }
420}
