13AVXPipeline::save_state(const __m256i& processed_signal) {
14 __m256i active = _mm256_cmpgt_epi16(processed_signal, _mm256_setzero_si256());
15 __m256i inactive = _mm256_cmpeq_epi16(processed_signal, _mm256_setzero_si256());
16 __m256i was_inactive = _mm256_cmpeq_epi16(m_adc_integral, _mm256_setzero_si256());
17
18 // If it was *not* inactive and is now inactive, then it must be a new TP.
19 __m256i new_tps = _mm256_andnot_si256(was_inactive, inactive);
20
21 m_adc_integral = _mm256_adds_epu16(m_adc_integral, processed_signal);
22
23 // If it is saturated, then we want to close early.
24 __m256i is_saturated = _mm256_cmpeq_epi16(m_adc_integral, _mm256_set1_epi16(-1));
25 new_tps = _mm256_or_si256(new_tps, is_saturated);
26
27 __m256i above_peak = _mm256_cmpgt_epi16(processed_signal, m_adc_peak);
28
29 m_adc_peak = _mm256_max_epi16(m_adc_peak, processed_signal);
30 m_time_peak = _mm256_blendv_epi8(m_time_peak, m_time_over_threshold, above_peak);
31
32 __m256i time_add = _mm256_blendv_epi8(_mm256_setzero_si256(), _mm256_set1_epi16(1), active);
33 m_time_over_threshold = _mm256_adds_epi16(m_time_over_threshold, time_add);
34
35 return new_tps;
36}

39AVXPipeline::check_for_tps(const __m256i& tp_mask) {
40 // tp_mask & 0xFFFF = 0 -> tp_mask == 0.
41 // True => tp_mask is all zeros and has no TPs.
42 // Negate!
43 return !_mm256_testz_si256(tp_mask, _mm256_set1_epi16(-1));

46std::vector<dunedaq::trgdataformats::TriggerPrimitive>
47AVXPipeline::generate_tps(const __m256i& tp_mask) {
48 // Mask everything that's relevant.
49 __m256i time_over_threshold = _mm256_blendv_epi8(_mm256_setzero_si256(), m_time_over_threshold, tp_mask);
50 __m256i adc_integral = _mm256_blendv_epi8(_mm256_setzero_si256(), m_adc_integral, tp_mask);
51 __m256i adc_peak = _mm256_blendv_epi8(_mm256_setzero_si256(), m_adc_peak, tp_mask);
52 __m256i time_peak = _mm256_blendv_epi8(_mm256_setzero_si256(), m_time_peak, tp_mask);
53
54 // Convert to uint16_t.
55 uint16_t tp_tot[16], tp_integral[16], tp_adc_peak[16], tp_time_peak[16];
56 _mm256_storeu_si256(reinterpret_cast<__m256i*>(tp_tot), time_over_threshold);
57 _mm256_storeu_si256(reinterpret_cast<__m256i*>(tp_integral), adc_integral);
58 _mm256_storeu_si256(reinterpret_cast<__m256i*>(tp_adc_peak), adc_peak);
59 _mm256_storeu_si256(reinterpret_cast<__m256i*>(tp_time_peak), time_peak);
60
61 std::vector<dunedaq::trgdataformats::TriggerPrimitive> tps;
62 for (int i = 0; i < 16; i++) {
63 if (tp_tot[i] == 0) continue; // Don't track non-TPs.
64 dunedaq::trgdataformats::TriggerPrimitive tp;
65 tp.adc_integral = tp_integral[i];
66 tp.adc_peak = tp_adc_peak[i];
67 tp.channel = m_channels[i];
68 tp.time_peak = tp_time_peak[i];
69 tp.time_over_threshold = tp_tot[i]; // TOT was incremented by 1 in AVX. Need to convert in TPGenerator.hpp.
70 tp.type = tp.Type::kTPC;
71 tp.algorithm = tp.Algorithm::kUnknown; // TODO: Choose a separate way to represent the mixed processors.
72
73 // time_start is handled at the next level up, since it is aware of the true and relative times.
74 tps.push_back(tp);
75 }
76
77 // Reset the channels that generated tps.
78 m_time_over_threshold = _mm256_blendv_epi8(m_time_over_threshold, _mm256_setzero_si256(), tp_mask);
79 m_adc_integral = _mm256_blendv_epi8(m_adc_integral, _mm256_setzero_si256(), tp_mask);
80 m_adc_peak = _mm256_blendv_epi8(m_adc_peak, _mm256_setzero_si256(), tp_mask);
81 m_time_peak = _mm256_blendv_epi8(m_time_peak, _mm256_setzero_si256(), tp_mask);
82
83 // Finalize.
84 return tps;
85}
