94void
95GIBIONode::reset(const std::string& clock_config_file) const
96{
97
98 write_soft_reset_register();
99
100 // Reset I2C switch and expander, active low
101 getNode("csr.ctrl.i2c_sw_rst").write(0x0);
102 getNode("csr.ctrl.i2c_exten_rst").write(0x0);
103 getNode("csr.ctrl.clk_gen_rst").write(0x0);
104 getClient().dispatch();
105
106 millisleep(1);
107
108 // End reset
109 getNode("csr.ctrl.i2c_sw_rst").write(0x1);
110 getNode("csr.ctrl.i2c_exten_rst").write(0x1);
111 getNode("csr.ctrl.clk_gen_rst").write(0x1);
112 getClient().dispatch();
113
114 CarrierType carrier_type = convert_value_to_carrier_type(read_carrier_type());
115
116 // enclustra i2c switch stuff
117 if (carrier_type == kCarrierEnclustraA35) {
118 try {
119 getNode<I2CMasterNode>(m_uid_i2c_bus).get_slave("AX3_Switch").write_i2c(0x01, 0x7f);
120 } catch (const std::exception& e) {
121 ers::warning(EnclustraSwitchFailure(ERS_HERE, e));
122 }
123 }
124
125 getNode("csr.ctrl.gps_clk_en").write(0x0);
126
127 // Set filter to full bandwidth mode A = B = 0x0
128 getNode("csr.ctrl.gps_clk_fltr_a").write(0x0);
129 getNode("csr.ctrl.gps_clk_fltr_b").write(0x0);
130 getClient().dispatch();
131
132 // Upload config file to PLL
133 configure_pll(clock_config_file);
134
135 getNode("csr.ctrl.rst").write(0x1);
136 getNode("csr.ctrl.rst").write(0x0);
137 getClient().dispatch();
138
139 auto sfp_expander_0 = get_i2c_device<I2CExpanderSlave>(m_uid_i2c_bus, "SFPExpander0");
140 auto sfp_expander_1 = get_i2c_device<I2CExpanderSlave>(m_uid_i2c_bus, "SFPExpander1");
141
142 // Set invert registers to default for both (0,1) banks
143 sfp_expander_0->set_inversion(0, 0x00);
144 sfp_expander_0->set_inversion(1, 0x00);
145 sfp_expander_1->set_inversion(0, 0x00);
146 sfp_expander_1->set_inversion(1, 0x00);
147
148 // 0: pin set as output, 1: pin set as input
149 sfp_expander_0->set_io(0, 0xff); // set all pins of bank 0 as inputs
150 sfp_expander_0->set_io(1, 0xff); // set all pins of bank 1 as inputs
151
152 sfp_expander_1->set_io(0, 0xff); // set all pins of bank 0 as inputs
153 sfp_expander_1->set_io(1, 0x00); // set all pins of bank 1 as outputs
154
155 // Set SFP disable
156 // Set pins 1-6 low, i.e. enable SFP 1-6 (pins 7,8 unused)
157 sfp_expander_1->set_outputs(1, 0xC0);
158
159 TLOG() << "Reset done";
160}

165GIBIONode::get_sfp_status(uint32_t sfp_id, bool print_out) const { // NOLINT(build/unsigned)
166 std::stringstream status;
167
168 validate_sfp_id(sfp_id);
169
170 uint8_t i2c_mux_bitmask = 1UL << (sfp_id+1);
171
172 set_i2c_mux_channels(i2c_mux_bitmask);
173
174 auto sfp = get_i2c_device<I2CSFPSlave>(m_sfp_i2c_buses.at(sfp_id), "SFP_EEProm");
175
176 status << "SFP " << sfp_id << ":" << std::endl;
177 status << sfp->get_status();
178
179 if (print_out)
180 TLOG() << status.str();
181
182 return status.str();
