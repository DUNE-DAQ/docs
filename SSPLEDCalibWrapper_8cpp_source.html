41{

43
44 m_device_interface = new dunedaq::sspmodules::DeviceInterface();
45
46 m_number_channels = conf->get_number_channels();
47 m_channel_mask = conf->get_channel_mask();
48 m_burst_count = conf->get_burst_count();
49 m_double_pulse_delay_ticks = conf->get_double_pulse_delay_ticks();

56 m_module_id = conf->get_module_id();

61 if (m_timing_address > 0xff) {
62 std::stringstream ss;
63 ss << "Error: Incorrect timing address set (" << m_timing_address << ")!" << std::endl;

74 m_device_interface->SetTimingAddress(m_timing_address);
75 m_device_interface->ConfigureLEDCalib(conf); //This sets up the ethernet interface and make sure that the pdts is synched

77 //m_device_interface->SetRegisterByName("eventDataInterfaceSelect", m_cfg.interface_type);

79 if ( conf->get_pulse_mode() == "single") {
80 m_single_pulse = true;
81 TLOG(TLVL_FULL_DEBUG) << "SSPLEDCalibWrapper: I think that you want SSP LED Calib module to be in single pulse..." << std::endl;
82 } else if ( conf->get_pulse_mode() == "burst") {

98 TLOG(TLVL_FULL_DEBUG) << "SSPLEDCalibWrapper: Configuring for single pulse mode..." << std::endl;

115void
116SSPLEDCalibWrapper::start(const data_t& /*args*/)
117{
118 TLOG_DEBUG(TLVL_ENTER_EXIT_METHODS) << "Start pulsing SSPLEDCalibWrapper of card " << m_board_id << "...";
119
120 if (m_run_marker.load()) {
121 TLOG_DEBUG(TLVL_ENTER_EXIT_METHODS) << "Run Marker says that SSPLEDCalibWrapper card " << m_board_id << " is already pulsing...";
122 return;
123 }
124
125 unsigned int base_bias_regAddress = 0x40000340;
126 unsigned int base_timing_regAddress = 0x800003C0;
127
128 if (m_number_channels == 5) {
129 base_bias_regAddress = 0x4000035C;
130 base_timing_regAddress = 0x800003DC;
131 TLOG_DEBUG(TLVL_ENTER_EXIT_METHODS) << "Configuring board with 5 channels..." ;
132 } else if (m_number_channels == 12) {
133 TLOG_DEBUG(TLVL_ENTER_EXIT_METHODS) << "Configuring board with 12 channels..." ;
134 } else {
135 std::stringstream ss;
136 ss << "ERROR: SOMEHOW ENDED UP WITHOUT NUMBER OF CHANNELS SET!!!!" << std::endl;
137 TLOG() << ss.str();
138 throw ConfigurationError(ERS_HERE, ss.str());
139 }
140
141 unsigned int pulse_bias_setting_270nm = (4095 * m_pulse_bias_percent_270nm)/100;
142 unsigned int pulse_bias_setting_367nm = (4095 * m_pulse_bias_percent_367nm)/100;
143
144 for (unsigned int counter = 0; counter < m_number_channels ; counter++) {
145 unsigned int bias_regAddress = base_bias_regAddress + 0x4*(counter); //0x40000340 - 0x4000036C
146 unsigned int bias_regVal = 0x00040000;
147 unsigned int timing_regAddress = base_timing_regAddress + 0x4*(counter); //0x80003C0 - 0x800003EC
148 unsigned int timing_regVal = 0x0; //the highest byte sets 8 (pdts trigger) + 1 (953.5 Hz) for burst mode, but 8 (pdts trigger) + 7 (single shot) for single
149 if (m_single_pulse) {
150 timing_regVal = 0xF0000000;
151 } else if (m_burst_mode) {
152 timing_regVal = 0x90000000;
153 }
154
155 TLOG(TLVL_FULL_DEBUG) << "Channel map is 0x" << std::hex << m_channel_mask << " and the comparison is 0x" << (1 << counter ) << std::endl;
156 if ( (m_channel_mask & ( (unsigned int)1 << counter)) == ((unsigned int)1 << counter) ) {
157 if ( counter < 6) {
158 bias_regVal = bias_regVal + pulse_bias_setting_270nm;
159 } else {
160 bias_regVal = bias_regVal + pulse_bias_setting_367nm;
161 }
162 timing_regVal = timing_regVal + m_pulse1_width_ticks;
163 timing_regVal = timing_regVal + (m_pulse2_width_ticks << 8);
164 timing_regVal = timing_regVal + (m_double_pulse_delay_ticks << 16);
165 TLOG(TLVL_FULL_DEBUG) << "Will turn on " << std::dec << counter << " channel at bias register 0x" << std::hex << bias_regAddress << " with bias value 0x" << bias_regVal << std::endl;
166 TLOG(TLVL_FULL_DEBUG) << " and set the width regsiter 0x" << std::hex << timing_regAddress << " to value of 0x" << timing_regVal << std::dec << std::endl;
167 m_device_interface->SetRegister(bias_regAddress, bias_regVal); //BIAS_DAC_CONFIG_N
168 if ( (counter == 7) && ! ( (m_channel_mask & ( (unsigned int)1 << counter)) == ((unsigned int)1 << (counter-1) ) ) ) {
169 //this is a really convoluted situation where for the very specific 12 channel board that arrived at CERN in June 2022
170 //the bias for channel 7 was not working and so the bias is taken from channel 6
171 //which means that if channel 7 is to be turned on while channel 6 is masked off
172 //you still have to bias channel 6 in order for the led on channel 7 to emmit light
173 m_device_interface->SetRegister((bias_regAddress - 0x4), bias_regVal); //BIAS_DAC_CONFIG_N
174 }
175
176 m_device_interface->SetRegister(timing_regAddress, timing_regVal); //cal_CONFIG_N
177 } else {
178 TLOG(TLVL_FULL_DEBUG) << "Will turn off channel " << std::dec << counter << " at timing register 0x" << std::hex << timing_regAddress << std::dec << std::endl;
179 m_device_interface->SetRegister(bias_regAddress, bias_regVal); //BIAS_DAC_CONFIG_N
180 m_device_interface->SetRegister(timing_regAddress, timing_regVal); //cal_CONFIG_N
181 }
182 }
183
184 m_device_interface->SetRegister(0x40000300, 0x1); //writing 0x1 to this register applies the bias voltage settings
185
186 m_run_marker = true;
187 TLOG_DEBUG(TLVL_ENTER_EXIT_METHODS) << "Start pulsing SSPLEDCalibWrapper of card " << m_board_id << " complete.";
188}

190void
191SSPLEDCalibWrapper::stop(const data_t& /*args*/)
192{
193
194 TLOG_DEBUG(TLVL_ENTER_EXIT_METHODS) << "Stop pulsing SSPLEDCalibWrapper of card " << m_board_id << "...";
195 if (!m_run_marker.load()) {
196 TLOG_DEBUG(TLVL_ENTER_EXIT_METHODS) << "The run_marker says that SSPLEDCalibWrapper card " << m_board_id << " is already stopped, but stopping anyways...";
197 }
198
199 for (unsigned int counter = 0; counter < 5; counter++) { //switch this to 12 for a 12 channel SSP
200 unsigned int bias_regAddress = 0x4000035C + 0x4*(counter);
201 unsigned int timing_regAddress = 0x800003DC + 0x4*(counter);
202 m_device_interface->SetRegister(bias_regAddress, 0x00040000); //BIAS_DAC_CONFIG_N
203 m_device_interface->SetRegister(timing_regAddress, 0x00000000); //cal_CONFIG_N
204 }
205
206 m_device_interface->SetRegister(0x40000300, 0x1); //writing 0x1 to this register applies the bias voltage settings
207
208 m_run_marker = false;
209 TLOG_DEBUG(TLVL_ENTER_EXIT_METHODS) << "Stop pulsing SSPLEDCalibWrapper of card " << m_board_id << " complete.";
210}
