92void dbse::SchemaMethodEditor::ParseToSave()
93{
94 bool changed = false;
95 std::string MethodName = ui->MethodName->text().toStdString();
96 std::string MethodDescription = ui->DescriptionTextBox->toPlainText().toStdString();
97
98 if ( MethodName != SchemaMethod->get_name() )
99 {
100 KernelWrapper::GetInstance().PushSetNameMethodCommand ( SchemaClass, SchemaMethod, MethodName );
101 changed = true;
102 }
103
104 if ( MethodDescription != SchemaMethod->get_description() )
105 {
106 KernelWrapper::GetInstance().PushSetDescriptionMethodCommand ( SchemaClass,
107 SchemaMethod,
108 MethodDescription );
109 changed = true;
110 }
111
112 if ( changed )
113 {
114 emit RebuildModel();
115 }
116
117 close();
118}

154bool dbse::SchemaMethodEditor::ShouldOpenMethodImplementationEditor ( QString Name )
155{
156 bool WidgetFound = false;
157
158 for ( QWidget * Editor : QApplication::allWidgets() )
159 {
160 SchemaMethodImplementationEditor * Widget =
161 dynamic_cast<SchemaMethodImplementationEditor *> ( Editor );
162
163 if ( Widget != nullptr )
164 {
165 if ( ( Widget->objectName() ).compare ( Name ) == 0 )
166 {
167 Widget->raise();
168 Widget->setVisible ( true );
169 Widget->activateWindow();
170 WidgetFound = true;
171 }
172 }
173 }
174
175 return !WidgetFound;
176}
