40// Special record command that writes to files from memory aligned LBs
41template<class ReadoutType, class LatencyBufferType>
42void
43ZeroCopyRecordingRequestHandlerModel<ReadoutType, LatencyBufferType>::record(const nlohmann::json& /*args*/)
44{
45 if (inherited::m_recording.load()) {
46 ers::error(
47 CommandError(ERS_HERE, inherited::m_sourceid, "A recording is still running, no new recording was started!"));
48 return;
49 }
50// FIXME: parameters to commands to be clarified.... hardcode for now
51 int recording_time_sec = 1;
52
53 inherited::m_recording_thread.set_work(
54 [&](int duration) {
55 size_t chunk_size = inherited::m_stream_buffer_size;
56 size_t alignment_size = inherited::m_latency_buffer->get_alignment_size();
57 TLOG() << "Start recording for " << duration << " second(s)" << std::endl;
58 inherited::m_recording.exchange(true);
59 auto start_of_recording = std::chrono::high_resolution_clock::now();
60 auto current_time = start_of_recording;
61 inherited::m_next_timestamp_to_record = 0;
62
63 const char* current_write_pointer = nullptr;
64 const char* start_of_buffer_pointer =
65 reinterpret_cast<const char*>(inherited::m_latency_buffer->start_of_buffer()); // NOLINT
66 const char* current_end_pointer;
67 const char* end_of_buffer_pointer = reinterpret_cast<const char*>(inherited::m_latency_buffer->end_of_buffer()); // NOLINT
68
69 size_t bytes_written = 0;
70
71 while (std::chrono::duration_cast<std::chrono::seconds>(current_time - start_of_recording).count() < duration) {
72 if (!inherited::m_cleanup_requested || (inherited::m_next_timestamp_to_record == 0)) {
73 size_t considered_chunks_in_loop = 0;
74
75 // Wait for potential running cleanup to finish first
76 {
77 std::unique_lock<std::mutex> lock(inherited::m_cv_mutex);
78 inherited::m_cv.wait(lock, [&] { return !inherited::m_cleanup_requested; });
79 }
80 inherited::m_cv.notify_all();
81
82 // Some frames have to be skipped to start copying from an aligned piece of memory
83 // These frames cannot be written without O_DIRECT as this would mess up the alignment of the write pointer
84 // into the target file
85 if (inherited::m_next_timestamp_to_record == 0) {
86 auto begin = inherited::m_latency_buffer->begin();
87 if (begin == inherited::m_latency_buffer->end()) {
88 // There are no elements in the buffer, update time and try again
89 current_time = std::chrono::high_resolution_clock::now();
90 continue;
91 }
92 inherited::m_next_timestamp_to_record = begin->get_timestamp();
93 size_t skipped_frames = 0;
94 while (reinterpret_cast<std::uintptr_t>(&(*begin)) % alignment_size) { // NOLINT
95 ++begin;
96 skipped_frames++;
97 if (!begin.good()) {
98 // We reached the end of the buffer without finding an aligned element
99 // Reset the next timestamp to record and try again
100 current_time = std::chrono::high_resolution_clock::now();
101 inherited::m_next_timestamp_to_record = 0;
102 continue;
103 }
104 }
105 TLOG() << "Skipped " << skipped_frames << " frames";
106 current_write_pointer = reinterpret_cast<const char*>(&(*begin)); // NOLINT
107 }
108
109 current_end_pointer = reinterpret_cast<const char*>(inherited::m_latency_buffer->back()); // NOLINT
110
111 // Break the loop from time to time to update the timestamp and check if we should stop recording
112 while (considered_chunks_in_loop < 100) {
113 auto iptr = reinterpret_cast<std::uintptr_t>(current_write_pointer); // NOLINT
114 if (iptr % alignment_size) {
115 // This should never happen
116 TLOG() << "Error: Write pointer is not aligned";
117 }
118 bool failed_write = false;
119 if (current_write_pointer + chunk_size < current_end_pointer) {
120 // We can write a whole chunk to file
121 failed_write |= !::write(m_fd, current_write_pointer, chunk_size);
122 if (!failed_write) {
123 bytes_written += chunk_size;
124 }
125 current_write_pointer += chunk_size;
126 } else if (current_end_pointer < current_write_pointer) {
127 if (current_write_pointer + chunk_size < end_of_buffer_pointer) {
128 // Write whole chunk to file
129 failed_write |= !::write(m_fd, current_write_pointer, chunk_size);
130 if (!failed_write) {
131 bytes_written += chunk_size;
132 }
133 current_write_pointer += chunk_size;
134 } else {
135 // Write the last bit of the buffer without using O_DIRECT as it possibly doesn't fulfill the
136 // alignment requirement
137 fcntl(m_fd, F_SETFL, O_CREAT | O_WRONLY);
138 failed_write |= !::write(m_fd, current_write_pointer, end_of_buffer_pointer - current_write_pointer);
139 fcntl(m_fd, F_SETFL, m_oflag);
140 if (!failed_write) {
141 bytes_written += end_of_buffer_pointer - current_write_pointer;
142 }
143 current_write_pointer = start_of_buffer_pointer;
144 }
145 }
146
147 if (current_write_pointer == end_of_buffer_pointer) {
148 current_write_pointer = start_of_buffer_pointer;
149 }
150
151 if (failed_write) {
152 TLOG() << "Failed write!";
153 ers::warning(CannotWriteToFile(ERS_HERE, inherited::m_output_file));
154 }
155 considered_chunks_in_loop++;
156 // This expression is "a bit" complicated as it finds the last frame that was written to file completely
157 inherited::m_next_timestamp_to_record =
158 reinterpret_cast<const ReadoutType*>( // NOLINT
159 start_of_buffer_pointer +
160 (((current_write_pointer - start_of_buffer_pointer) / ReadoutType::fixed_payload_size) *
161 ReadoutType::fixed_payload_size))
162 ->get_timestamp();
163 }
164 }
165 current_time = std::chrono::high_resolution_clock::now();
166 }
167
168 // Complete writing the last frame to file
169 if (current_write_pointer != nullptr) {
170 const char* last_started_frame =
171 start_of_buffer_pointer +
172 (((current_write_pointer - start_of_buffer_pointer) / ReadoutType::fixed_payload_size) *
173 ReadoutType::fixed_payload_size);
174 if (last_started_frame != current_write_pointer) {
175 fcntl(m_fd, F_SETFL, O_CREAT | O_WRONLY);
176 if (!::write(m_fd,
177 current_write_pointer,
178 (last_started_frame + ReadoutType::fixed_payload_size) - current_write_pointer)) {
179 ers::warning(CannotWriteToFile(ERS_HERE, inherited::m_output_file));
180 } else {
181 bytes_written += (last_started_frame + ReadoutType::fixed_payload_size) - current_write_pointer;
182 }
183 }
184 }
185 ::close(m_fd);
186
187 inherited::m_next_timestamp_to_record = std::numeric_limits<uint64_t>::max(); // NOLINT (build/unsigned)
188
189 TLOG() << "Stopped recording, wrote " << bytes_written << " bytes";
190 inherited::m_recording.exchange(false);
191 }, recording_time_sec);
192}
