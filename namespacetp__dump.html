148def write_summary_stats(data: np.ndarray, filename: str, title: str) -> None:
149 """
150 Writes the given summary statistics to :filename:.
151
152 Parameters:
153 data (np.ndarray): Array of a TP data member.
154 filename (str): File to append outputs to.
155 title (str): Title of the TP data member.
156
157 Appends statistics to the given file.
158 """
159 # Algorithm, Det ID, etc. are not expected to vary.
160 # Check first that they don't vary, and move on if so.
161 if np.all(data == data[0]):
162 print(f"{title} data member is the same for all TPs. Skipping summary statistics.")
163 return None
164
165 summary = stats.describe(data)
166 std = np.sqrt(summary.variance)
167 with open(filename, 'a') as out:
168 out.write(f"{title}\n")
169 out.write(f"Reference Statistics:\n"
170 f"\tTotal # TPs = {summary.nobs},\n"
171 f"\tMean = {summary.mean:.2f},\n"
172 f"\tStd = {std:.2f},\n"
173 f"\tMin = {summary.minmax[0]},\n"
174 f"\tMax = {summary.minmax[1]}.\n")
175 std3_count = np.sum(data > summary.mean + 3*std) + np.sum(data < summary.mean - 3*std)
176 std2_count = np.sum(data > summary.mean + 2*std) + np.sum(data < summary.mean - 2*std)
177 out.write(f"Anomalies:\n"
178 f"\t# of >3 Sigma TPs = {std3_count},\n"
179 f"\t# of >2 Sigma TPs = {std2_count}.\n")
180 out.write("\n\n")
181
