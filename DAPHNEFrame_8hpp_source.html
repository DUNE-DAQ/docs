101 uint16_t get_adc(int i) const // NOLINT
102 {
103 if (i < 0 || i >= s_num_adcs)
104 throw std::out_of_range("ADC index out of range");
105
106 // The index of the first (and sometimes only) word containing the required ADC value
107 int word_index = s_bits_per_adc * i / s_bits_per_word;
108 assert(word_index < s_num_adc_words);
109 // Where in the word the lowest bit of our ADC value is located
110 int first_bit_position = (s_bits_per_adc * i) % s_bits_per_word;
111 // How many bits of our desired ADC are located in the `word_index`th word
112 int bits_from_first_word = std::min(s_bits_per_adc, s_bits_per_word - first_bit_position);
113 uint16_t adc = adc_words[word_index] >> first_bit_position; // NOLINT
114 // If we didn't get the full 14 bits from this word, we need the rest from the next word
115 if (bits_from_first_word < s_bits_per_adc) {
116 assert(word_index + 1 < s_num_adc_words);
117 adc |= adc_words[word_index + 1] << bits_from_first_word;
118 }
119 // Mask out all but the lowest 14 bits;
120 return adc & 0x3FFFu;
121 }
122
126 void set_adc(int i, uint16_t val) // NOLINT
127 {
128 if (i < 0 || i >= s_num_adcs)
129 throw std::out_of_range("ADC index out of range");
130 if (val >= (1 << s_bits_per_adc))
131 throw std::out_of_range("ADC value out of range");
132
133 // The index of the first (and sometimes only) word containing the required ADC value
134 int word_index = s_bits_per_adc * i / s_bits_per_word;
135 assert(word_index < s_num_adc_words);
136 // Where in the word the lowest bit of our ADC value is located
137 int first_bit_position = (s_bits_per_adc * i) % s_bits_per_word;
138 // How many bits of our desired ADC are located in the `word_index`th word
139 int bits_in_first_word = std::min(s_bits_per_adc, s_bits_per_word - first_bit_position);
140 uint32_t mask = (1 << (first_bit_position)) - 1;
141 adc_words[word_index] = ((val << first_bit_position) & ~mask) | (adc_words[word_index] & mask);
142 // If we didn't put the full 14 bits in this word, we need to put the rest in the next word
143 if (bits_in_first_word < s_bits_per_adc) {
144 assert(word_index + 1 < s_num_adc_words);
145 mask = (1 << (s_bits_per_adc - bits_in_first_word)) - 1;
146 adc_words[word_index + 1] = ((val >> bits_in_first_word) & mask) | (adc_words[word_index + 1] & ~mask);
147 }
148 }
