2026{
2027 Cache<T> *c = static_cast<Cache<T>*>(x);
2028
2029 // rename template object
2030 auto it = c->m_cache.find(old_id);
2031 if (it != c->m_cache.end())
2032 {
2033 TLOG_DEBUG(3) << " * rename \'" << old_id << "\' to \'" << new_id << "\' in class \'" << T::s_class_name << "\')";
2034 c->m_cache[new_id] = it->second;
2035 c->m_cache.erase(it);
2036
2037 std::lock_guard<std::mutex> scoped_lock(it->second->m_mutex);
2038 it->second->p_UID = new_id;
2039 }
2040
2041 // rename generated objects if any
2042 auto range = c->m_t_cache.equal_range(old_id);
2043 for (auto it = range.first; it != range.second;)
2044 {
2045 T * o = it->second;
2046 it = c->m_t_cache.erase(it);
2047 c->m_t_cache.emplace(new_id, o);
2048 }
