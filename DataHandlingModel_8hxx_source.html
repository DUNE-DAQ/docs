8template<class RDT, class RHT, class LBT, class RPT>
9void
10DataHandlingModel<RDT, RHT, LBT, RPT>::init(const appmodel::DataHandlerModule* mcfg)
11{
12 // Setup request queues
13 //setup_request_queues(mcfg);
14 try {
15 for (auto input : mcfg->get_inputs()) {
16 if (input->get_data_type() == "DataRequest") {
17 m_data_request_receiver = get_iom_receiver<dfmessages::DataRequest>(input->UID()) ;
18 }
19 else {
20 m_raw_data_receiver_connection_name = input->UID();
21 // Parse for prefix
22 std::string conn_name = input->UID();
23 const char delim = '_';
24 std::vector<std::string> words;
25 std::size_t start;
26 std::size_t end = 0;
27 while ((start = conn_name.find_first_not_of(delim, end)) != std::string::npos) {
28 end = conn_name.find(delim, start);
29 words.push_back(conn_name.substr(start, end - start));
30 }
31
32 TLOG_DEBUG(TLVL_WORK_STEPS) << "Initialize connection based on uid: "
33 << m_raw_data_receiver_connection_name << " front word: " << words.front();
34
35 std::string cb_prefix("cb");
36 if (words.front() == cb_prefix) {
37 m_callback_mode = true;
38 }
39
40 if (!m_callback_mode) {
41 m_raw_data_receiver = get_iom_receiver<RDT>(m_raw_data_receiver_connection_name);
42 m_raw_receiver_timeout_ms = std::chrono::milliseconds(input->get_recv_timeout_ms());
43 }
44 }
45 }
46 for (auto output : mcfg->get_outputs()) {
47 if (output->get_data_type() == "TimeSync") {
48 m_generate_timesync = true;
49 m_timesync_sender = get_iom_sender<dfmessages::TimeSync>(output->UID()) ;
50 m_timesync_connection_name = output->UID();
51 break;
52 }
53 }
54 } catch (const ers::Issue& excpt) {
55 throw ResourceQueueError(ERS_HERE, "raw_input or frag_output", "DataHandlingModel", excpt);
56 }
57
58 // Raw input connection sensibility check
59 if (!m_callback_mode && m_raw_data_receiver == nullptr) {
60 ers::error(ConfigurationError(ERS_HERE, m_sourceid, "Non callback mode, and receiver is unset!"));
61 }
62
63 // Instantiate functionalities
64 m_error_registry.reset(new FrameErrorRegistry());
65 m_latency_buffer_impl.reset(new LBT());
66 m_raw_processor_impl.reset(new RPT(m_error_registry, mcfg->get_post_processing_enabled()));
67 m_request_handler_impl.reset(new RHT(m_latency_buffer_impl, m_error_registry));
68
69 register_node(mcfg->get_module_configuration()->get_latency_buffer()->UID(), m_latency_buffer_impl);
70 register_node(mcfg->get_module_configuration()->get_data_processor()->UID(), m_raw_processor_impl);
71 register_node(mcfg->get_module_configuration()->get_request_handler()->UID(), m_request_handler_impl);
72
73 //m_request_handler_impl->init(args);
74 //m_raw_processor_impl->init(args);
75 m_request_handler_supports_cutoff_timestamp = m_request_handler_impl->supports_cutoff_timestamp();
76 m_fake_trigger = false;
77 m_raw_receiver_sleep_us = std::chrono::microseconds::zero();
78 m_sourceid.id = mcfg->get_source_id();
79 m_sourceid.subsystem = RDT::subsystem;
80 m_processing_delay_ticks = mcfg->get_module_configuration()->get_post_processing_delay_ticks();
81
82
83 // Configure implementations:
84 m_raw_processor_impl->conf(mcfg);
85 // Configure the latency buffer before the request handler so the request handler can check for alignment
86 // restrictions
87 try {
88 m_latency_buffer_impl->conf(mcfg->get_module_configuration()->get_latency_buffer());
89 } catch (const std::bad_alloc& be) {
90 ers::error(ConfigurationError(ERS_HERE, m_sourceid, "Latency Buffer can't be allocated with size!"));
91 }
92 m_request_handler_impl->conf(mcfg);
93}

166 m_latency_buffer_impl->flush();
167 m_raw_processor_impl->stop(args);

195template<class RDT, class RHT, class LBT, class RPT>
196void
197DataHandlingModel<RDT, RHT, LBT, RPT>::run_consume()
198{
199
200 TLOG_DEBUG(TLVL_WORK_STEPS) << "Consumer thread started...";
201 m_rawq_timeout_count = 0;
202 m_num_payloads = 0;
203 m_sum_payloads = 0;
204 m_stats_packet_count = 0;
205
206 // Variables for book-keeping of delayed post-processing
207 //timestamp_t oldest_ts=0;
208 timestamp_t newest_ts=0;
209 timestamp_t end_win_ts=0;
210 bool first_cycle = true;
211 auto last_post_proc_time = std::chrono::system_clock::now();
212 auto now = last_post_proc_time;
213 std::chrono::milliseconds milliseconds;
214 RDT processed_element;
215
216 while (m_run_marker.load()) {
217 // Try to acquire data
218
219 auto opt_payload = m_raw_data_receiver->try_receive(m_raw_receiver_timeout_ms);
220
221 if (opt_payload) {
222
223 RDT& payload = opt_payload.value();
224
225 m_raw_processor_impl->preprocess_item(&payload);
226 if (m_request_handler_supports_cutoff_timestamp) {
227 int64_t diff1 = payload.get_timestamp() - m_request_handler_impl->get_cutoff_timestamp();
228 if (diff1 <= 0) {
229 //m_request_handler_impl->increment_tardy_tp_count();
230 ers::warning(DataPacketArrivedTooLate(ERS_HERE, m_run_number, payload.get_timestamp(),
231 m_request_handler_impl->get_cutoff_timestamp(), diff1,
232 (static_cast<double>(diff1)/62500.0)));
233 }
234 }
235 if (!m_latency_buffer_impl->write(std::move(payload))) {
236 //TLOG_DEBUG(TLVL_TAKE_NOTE) << "***ERROR: Latency buffer is full and data was overwritten!";
237 m_num_payloads_overwritten++;
238 }
239 if (m_processing_delay_ticks ==0) {
240 m_raw_processor_impl->postprocess_item(m_latency_buffer_impl->back());
241 ++m_num_payloads;
242 ++m_sum_payloads;
243 ++m_stats_packet_count;
244 }
245 } else {
246 ++m_rawq_timeout_count;
247 // Protection against a zero sleep becoming a yield
248 if ( m_raw_receiver_sleep_us != std::chrono::microseconds::zero())
249 std::this_thread::sleep_for(m_raw_receiver_sleep_us);
250 }
251
252 // Add here a possible deferral of the post processing, to allow elements being reordered in the LB
253 // Basically, find data older than a certain timestamp and process all data since the last post-processed element up to that value
254 if (m_processing_delay_ticks !=0 && m_latency_buffer_impl->occupancy() > 0) {
255 now = std::chrono::system_clock::now();
256 milliseconds = std::chrono::duration_cast<std::chrono::milliseconds>(now - last_post_proc_time);
257
258 if (milliseconds.count() > 1) {
259 last_post_proc_time = now;
260 // Get the LB boundaries
261 auto tail = m_latency_buffer_impl->back();
262 newest_ts = tail->get_timestamp();
263
264 if (first_cycle) {
265 auto head = m_latency_buffer_impl->front();
266 processed_element.set_timestamp(head->get_timestamp());
267 first_cycle = false;
268 TLOG() << "***** First pass post processing *****" ;
269 }
270
271 if (newest_ts - processed_element.get_timestamp() > m_processing_delay_ticks) {
272 end_win_ts = newest_ts - m_processing_delay_ticks;
273 auto start_iter=m_latency_buffer_impl->lower_bound(processed_element, false);
274 processed_element.set_timestamp(end_win_ts);
275 auto end_iter=m_latency_buffer_impl->lower_bound(processed_element, false);
276
277 for (auto it = start_iter; it!= end_iter; ++it) {
278 m_raw_processor_impl->postprocess_item(&(*it));
279 ++m_num_payloads;
280 ++m_sum_payloads;
281 ++m_stats_packet_count;
282 }
283 }
284 }
285 }
286 }
287 TLOG_DEBUG(TLVL_WORK_STEPS) << "Consumer thread joins... ";
288}

319template<class RDT, class RHT, class LBT, class RPT>
320void
321DataHandlingModel<RDT, RHT, LBT, RPT>::run_timesync()
322{
323 TLOG_DEBUG(TLVL_WORK_STEPS) << "TimeSync thread started...";
324 m_num_requests = 0;
325 m_sum_requests = 0;
326 uint64_t msg_seqno = 0;
327 timestamp_t prev_timestamp = 0;
328 auto once_per_run = true;
329 size_t zero_timestamp_count = 0;
330 size_t duplicate_timestamp_count = 0;
331 size_t total_timestamp_count = 0;
332 while (m_run_marker.load()) {
333 try {
334 auto timesyncmsg = dfmessages::TimeSync(m_raw_processor_impl->get_last_daq_time());
335 ++total_timestamp_count;
336 // daq_time is zero for the first received timesync, and may
337 // be the same as the previous daq_time if the data has
338 // stopped flowing. In both cases we don't send the TimeSync
339 if (timesyncmsg.daq_time != 0 && timesyncmsg.daq_time != prev_timestamp) {
340 prev_timestamp = timesyncmsg.daq_time;
341 timesyncmsg.run_number = m_run_number;
342 timesyncmsg.sequence_number = ++msg_seqno;
343 timesyncmsg.source_pid = m_pid_of_current_process;
344 TLOG_DEBUG(TLVL_TIME_SYNCS) << "New timesync: daq=" << timesyncmsg.daq_time
345 << " wall=" << timesyncmsg.system_time << " run=" << timesyncmsg.run_number
346 << " seqno=" << timesyncmsg.sequence_number << " pid=" << timesyncmsg.source_pid;
347 try {
348 dfmessages::TimeSync timesyncmsg_copy(timesyncmsg);
349 m_timesync_sender->send(std::move(timesyncmsg_copy), std::chrono::milliseconds(500));
350 } catch (ers::Issue& excpt) {
351 ers::warning(
352 TimeSyncTransmissionFailed(ERS_HERE, m_sourceid, m_timesync_connection_name, excpt));
353 }
354
355 if (m_fake_trigger) {
356 dfmessages::DataRequest dr;
357 ++m_current_fake_trigger_id;
358 dr.trigger_number = m_current_fake_trigger_id;
359 dr.trigger_timestamp = timesyncmsg.daq_time > 500 * us ? timesyncmsg.daq_time - 500 * us : 0;
360 auto width = 300000;
361 uint offset = 100;
362 dr.request_information.window_begin = dr.trigger_timestamp > offset ? dr.trigger_timestamp - offset : 0;
363 dr.request_information.window_end = dr.request_information.window_begin + width;
364 dr.request_information.component = m_sourceid;
365 dr.data_destination = "data_fragments_q";
366 TLOG_DEBUG(TLVL_WORK_STEPS) << "Issuing fake trigger based on timesync. "
367 << " ts=" << dr.trigger_timestamp
368 << " window_begin=" << dr.request_information.window_begin
369 << " window_end=" << dr.request_information.window_end;
370 m_request_handler_impl->issue_request(dr);
371
372 ++m_num_requests;
373 ++m_sum_requests;
374 }
375 } else {
376 if (timesyncmsg.daq_time == 0) {++zero_timestamp_count;}
377 if (timesyncmsg.daq_time == prev_timestamp) {++duplicate_timestamp_count;}
378 if (once_per_run) {
379 TLOG() << "Timesync with DAQ time 0 won't be sent out as it's an invalid sync.";
380 once_per_run = false;
381 }
382 }
383 } catch (const iomanager::TimeoutExpired& excpt) {
384 // ++m_timesyncqueue_timeout;
385 }
386 // Split up the 100ms sleep into 10 sleeps of 10ms, so we respond to "stop" quicker
387 for (size_t i=0; i<10; ++i) {
388 std::this_thread::sleep_for(std::chrono::milliseconds(10));
389 if (!m_run_marker.load()) {
390 break;
391 }
392 }
393 }
394 once_per_run = true;
395 TLOG_DEBUG(TLVL_WORK_STEPS) << "TimeSync thread joins... (timestamp count, zero/same/total = "
396 << zero_timestamp_count << "/" << duplicate_timestamp_count << "/"
397 << total_timestamp_count << ")";
398}
