162dunedaq::sspmodules::DeviceManager::OpenDevice(
163 unsigned int deviceNum,
164 bool slowControlOnly)
165{
166
167 Device* device = 0;
168 if (fEthernetDevices.find(deviceNum) == fEthernetDevices.end()) {
169 fEthernetDevices[deviceNum] = (std::move(
170 std::unique_ptr<dunedaq::sspmodules::EthernetDevice>(new dunedaq::sspmodules::EthernetDevice(deviceNum))));
171 }
172 if (fEthernetDevices[deviceNum]->IsOpen()) {
173 // dune::DAQLogger::LogError("SSP_DeviceManager")<<"Attempt to open already open device!"<<std::endl;
174 throw(EDeviceAlreadyOpen());
175 } else {
176 device = fEthernetDevices[deviceNum].get();
177 device->Open(slowControlOnly);
178 }
179 return device;
180}
