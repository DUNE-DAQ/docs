109void TimingHardwareManagerBase::do_scrap(const nlohmann::json& data)
110{
111 m_hw_command_receiver->remove_callback();
112
113 auto time_of_scrap = std::chrono::high_resolution_clock::now();
114 while(m_command_threads.size())
115 {
116 auto now = std::chrono::high_resolution_clock::now();
117 auto ms_since_scrap = std::chrono::duration_cast<std::chrono::milliseconds>(now - time_of_scrap);
118 TLOG_DEBUG(0) << "Have been waiting for " << ms_since_scrap.count() << " ms for " << m_command_threads.size() << " command threads to finish...";
119 std::this_thread::sleep_for(std::chrono::microseconds(250000));
120 }
121 m_run_endpoint_scan_cleanup_thread.store(false);
122
123 stop_hw_mon_gathering();
124
125 scrap_uhal();
126
127 m_command_threads.clear();
128 m_info_gatherers.clear();
129 m_timing_hw_cmd_map_.clear();
130 m_hw_device_map.clear();
131 m_connection_manager.reset();

134const timing::TimingNode*
135TimingHardwareManagerBase::get_timing_device_plain(const std::string& device_name)
136{
137
138 if (!device_name.compare("")) {
139 std::stringstream message;
140 message << "UHAL device name is an empty string";
141 throw UHALDeviceNameIssue(ERS_HERE, message.str());
142 }
143
144 if (auto hw_device_entry = m_hw_device_map.find(device_name); hw_device_entry != m_hw_device_map.end()) {

159
160 TLOG_DEBUG(0) << get_name() << ": hw device interface for: " << device_name << " successfully created.";
161
162 return dynamic_cast<const timing::TimingNode*>(&m_hw_device_map.find(device_name)->second->getNode(""));

166void
167TimingHardwareManagerBase::gather_monitor_data(InfoGatherer& gatherer)
168{
169 auto device_name = gatherer.get_device_name();
170
171 while (gatherer.run_gathering()) {
172
173 // collect the data from the hardware
174 try {
175 auto design = get_timing_device<const timing::TopDesignInterface*>(device_name);
176
177 gatherer.collect_info_from_device(*design);
178 } catch (const std::exception& excpt) {
179 ers::warning(FailedToCollectOpMonInfo(ERS_HERE, device_name, excpt));
180 }
181
182 auto prev_gather_time = std::chrono::steady_clock::now();
183 auto next_gather_time = prev_gather_time + std::chrono::microseconds(gatherer.get_gather_interval());
184
185 // check running_flag periodically
186 auto slice_period = std::chrono::microseconds(10000);
187 auto next_slice_gather_time = prev_gather_time + slice_period;
188
189 bool break_flag = false;
190 while (next_gather_time > next_slice_gather_time + slice_period) {
191 if (!gatherer.run_gathering()) {
192 TLOG_DEBUG(0) << "while waiting to gather data, negative run gatherer flag detected.";
193 break_flag = true;
194 break;
195 }
196 std::this_thread::sleep_until(next_slice_gather_time);
197 next_slice_gather_time = next_slice_gather_time + slice_period;
198 }
199 if (break_flag == false) {
200 std::this_thread::sleep_until(next_gather_time);
201 }
202 }
203}

280void
281TimingHardwareManagerBase::process_hardware_command(timingcmd::TimingHwCmd& timing_hw_cmd)
282{
283 std::ostringstream starting_stream;
284 starting_stream << ": Executing process_hardware_command() callback.";
285 TLOG_DEBUG(0) << get_name() << starting_stream.str();
286
287 ++m_received_hw_commands_counter;
288
289 TLOG_DEBUG(0) << get_name() << ": Received hardware command #" << m_received_hw_commands_counter.load()
290 << ", it is of type: " << timing_hw_cmd.id << ", targeting device: " << timing_hw_cmd.device << ", with payload: " << timing_hw_cmd.payload.dump();
291
292 std::string hw_cmd_name = timing_hw_cmd.id;
293 if (auto cmd = m_timing_hw_cmd_map_.find(hw_cmd_name); cmd != m_timing_hw_cmd_map_.end()) {
294
295 ++m_accepted_hw_commands_counter;
296
297 TLOG_DEBUG(0) << "Found hw cmd: " << hw_cmd_name;
298 try {
299 std::invoke(cmd->second, timing_hw_cmd);
300 } catch (const std::exception& exception) {
301 ers::error(FailedToExecuteHardwareCommand(ERS_HERE, hw_cmd_name, timing_hw_cmd.device, exception));
302 ++m_failed_hw_commands_counter;
303 }
304 } else {
305 ers::error(InvalidHardwareCommandID(ERS_HERE, hw_cmd_name));
306 ++m_rejected_hw_commands_counter;
307 }
308
309 std::ostringstream exiting_stream;
310 exiting_stream << ": Finished executing process_hardware_command() callback. Received " << m_received_hw_commands_counter.load()
311 << " commands";
312 TLOG_DEBUG(0) << get_name() << exiting_stream.str();
313}

316void
317TimingHardwareManagerBase::io_reset(const timingcmd::TimingHwCmd& hw_cmd)
318{
319 timingcmd::IOResetCmdPayload cmd_payload;
320 timingcmd::from_json(hw_cmd.payload, cmd_payload);
321
322 TLOG_DEBUG(0) << get_name() << ": " << hw_cmd.device << " io reset";
323
324 // io reset disrupts hw mon gathering, so stop if running
325 auto running_hw_gatherers = check_hw_mon_gatherer_is_running(hw_cmd.device);
326 for (auto& gatherer: running_hw_gatherers)
327 {
328 stop_hw_mon_gathering(gatherer);
329 }
330
331 auto design = get_timing_device<const timing::TopDesignInterface*>(hw_cmd.device);
332
333 if (cmd_payload.soft) {
334 TLOG_DEBUG(0) << get_name() << ": " << hw_cmd.device << " soft io reset";
335 design->soft_reset_io();
336 } else if (!cmd_payload.clock_config.empty()) {
337 TLOG_DEBUG(0) << get_name() << ": " << hw_cmd.device
338 << " io reset, with supplied clk file: " << cmd_payload.clock_config;
339 design->reset_io(cmd_payload.clock_config);
340 }
341 else
342 {
343 TLOG_DEBUG(0) << get_name() << ": " << hw_cmd.device
344 << " io reset, with supplied clk source: " << cmd_payload.clock_source;
345 design->reset_io(static_cast<timing::ClockSource>(cmd_payload.clock_source));
346 }
347
348 // if hw mon gathering was running previously, start it again
349 for (auto& gatherer: running_hw_gatherers)
350 {
351 start_hw_mon_gathering(gatherer);
352 }
353}

403void TimingHardwareManagerBase::perform_endpoint_scan(const timingcmd::TimingHwCmd& hw_cmd)
404{
405 timingcmd::TimingMasterEndpointScanPayload cmd_payload;
406 timingcmd::from_json(hw_cmd.payload, cmd_payload);
407
408 for (auto& endpoint_location : cmd_payload.endpoints)
409 {
410 auto endpoint_address = endpoint_location.address;
411 auto fanout_slot = endpoint_location.fanout_slot;
412 auto sfp_slot = endpoint_location.sfp_slot;
413
414 std::unique_lock<std::mutex> master_sfp_lock(master_sfp_mutex);
415
416 TLOG_DEBUG(1) << get_name() << ": " << hw_cmd.device << " master_endpoint_scan starting: ept adr: " << endpoint_address << ", ept sfp: " << sfp_slot << ", fanout slot: " << fanout_slot;
417
418 auto master_design = get_timing_device<const timing::MasterDesignInterface*>(hw_cmd.device);
419
420 try
421 {
422 master_design->get_master_node_plain()->switch_endpoint_sfp(endpoint_address, true);
423
424 if (sfp_slot >= 0)
425 {
426 if (fanout_slot > 0)
427 {
428 // configure fanout/FIB
429 get_timing_device<const timing::CDRMuxDesignInterface*>(m_monitored_device_names_fanout.at(fanout_slot-1))->switch_cdr_mux(sfp_slot);
430
431 // configure MIB
432 dynamic_cast<const timing::CDRMuxDesignInterface*>(master_design)->switch_cdr_mux(fanout_slot-1);
433 }
434 else
435 {
436 dynamic_cast<const timing::MasterMuxDesign*>(master_design)->switch_downstream_mux_channel(sfp_slot, false);
437 }
438 }
439 // configure any master mux, possibly
440 master_design->get_master_node_plain()->scan_endpoint(endpoint_address, false);
441 master_design->get_master_node_plain()->switch_endpoint_sfp(endpoint_address, false);
442 }
443 catch(std::exception& e)
444 {
445 ers::error(EndpointScanFailure(ERS_HERE,e));
446 master_design->get_master_node_plain()->switch_endpoint_sfp(endpoint_address, false);
447 }
448 }

451void TimingHardwareManagerBase::clean_endpoint_scan_threads()
452{
453 TLOG_DEBUG(0) << "Entering clean_endpoint_scan_threads()";
454 bool break_flag = false;
455 while (!break_flag)
456 {
457 for (auto& thread : m_command_threads)
458 {
459 if (thread.second->joinable())
460 {
461 std::unique_lock map_lock(m_command_threads_map_mutex);
462 TLOG_DEBUG(2) << thread.first << " thread ready. Cleaning up.";
463 thread.second->join();
464 m_command_threads.erase(thread.first);
465 }
466 }
467
468 auto prev_clean_time = std::chrono::steady_clock::now();
469 auto next_clean_time = prev_clean_time + std::chrono::milliseconds(30);
470
471 // check running_flag periodically
472 auto flag_check_period = std::chrono::milliseconds(1);
473 auto next_flag_check_time = prev_clean_time + flag_check_period;
474
475 while (next_clean_time > next_flag_check_time + flag_check_period) {
476 if (!m_run_endpoint_scan_cleanup_thread.load()) {
477 TLOG_DEBUG(2) << "while waiting to clean up endpoint scan threads, negative run gatherer flag detected.";
478 break_flag = true;
479 break;
480 }
481 std::this_thread::sleep_until(next_flag_check_time);
482 next_flag_check_time = next_flag_check_time + flag_check_period;
483 }
484 if (break_flag == false) {
485 std::this_thread::sleep_until(next_clean_time);
486 }
487 }
488 TLOG_DEBUG(0) << "Exiting clean_endpoint_scan_threads()";
489}
