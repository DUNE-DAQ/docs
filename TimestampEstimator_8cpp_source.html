60TimestampEstimator::add_timestamp_datapoint(uint64_t daq_time, uint64_t system_time)
61{
62 using namespace std::chrono;
63
64 std::scoped_lock<std::mutex> lk(m_datapoint_mutex);
65
66 // First, update the latest timestamp
67 TimeSyncPoint estimate = m_current_timestamp_estimate.load();
68 int64_t diff = estimate.daq_time - daq_time;
69 TLOG_DEBUG(TLVL_TIME_SYNC_PROPERTIES) << "Got a TimeSync timestamp = " << daq_time
70 << ", system time = " << system_time
71 << " when current timestamp estimate was " << estimate.daq_time << ". diff=" << diff;
72
73 if (m_most_recent_daq_time == std::numeric_limits<uint64_t>::max() ||
74 daq_time > m_most_recent_daq_time) {
75 m_most_recent_daq_time = daq_time;
76 m_most_recent_system_time = system_time;
77 }
78
79 if (m_most_recent_daq_time != std::numeric_limits<uint64_t>::max()) {
80 // Update the current timestamp estimate, based on the most recently-read TimeSync
81 using namespace std::chrono;
82
83 auto time_now =
84 static_cast<uint64_t>(duration_cast<microseconds>(system_clock::now().time_since_epoch()).count()); // NOLINT
85 auto steady_time_now = steady_clock::now();
86
87 // (PAR 2021-07-22) We only want to _increase_ our timestamp
88 // estimate, not _decrease_ it, so we only attempt the update if
89 // our system time is later than the latest time sync's system
90 // time. We can get TimeSync messages from the "future" if
91 // they're coming from another host whose clock is not exactly
92 // synchronized with ours: that's fine, but if the discrepancy
93 // is large, then badness could happen, so emit a warning
94
95 if (time_now < m_most_recent_system_time - 10000) {
96 ers::warning(EarlyTimeSync(ERS_HERE, m_most_recent_system_time - time_now));
97 }
98
99 if (time_now > m_most_recent_system_time) {
100
101 auto delta_time = time_now - m_most_recent_system_time;
102 TLOG_DEBUG(TLVL_TIME_SYNC_PROPERTIES)
103 << "Time diff between current system and latest TimeSync system time [us]: " << delta_time;
104
105 // Warn user if current system time is more than 1s ahead of latest TimeSync system time. This could be a sign of
106 // an issue, e.g. machine times out of sync
107 if (delta_time > 1e6)
108 ers::warning(LateTimeSync(ERS_HERE, delta_time));
109
110 const uint64_t new_timestamp =
111 m_most_recent_daq_time + delta_time * m_clock_frequency_hz / 1000000;
112
113 // Don't ever decrease the timestamp; just wait until enough
114 // time passes that we want to increase it
115 if (estimate.daq_time == std::numeric_limits<uint64_t>::max() ||
116 new_timestamp >= estimate.daq_time) {
117 TLOG_DEBUG(TLVL_TIME_SYNC_NEW_ESTIMATE)
118 << "Storing new timestamp estimate of " << new_timestamp << " ticks (..." << std::fixed
119 << std::setprecision(8)
120 << (static_cast<double>(new_timestamp % (m_clock_frequency_hz * 1000)) /
121 static_cast<double>(m_clock_frequency_hz))
122 << " sec), mrt.daq_time is " << m_most_recent_daq_time << " ticks (..."
123 << (static_cast<double>(m_most_recent_daq_time % (m_clock_frequency_hz * 1000)) /
124 static_cast<double>(m_clock_frequency_hz))
125 << " sec), delta_time is " << delta_time << " usec, clock_freq is " << m_clock_frequency_hz << " Hz";
126 m_current_timestamp_estimate.store(TimeSyncPoint{new_timestamp, steady_time_now});
127 } else {
128 TLOG_DEBUG(TLVL_TIME_SYNC_NOTES) << "Not updating timestamp estimate backwards from "
129 << m_current_timestamp_estimate.load().daq_time << " to " << new_timestamp;
130 }
131 }
132 }
