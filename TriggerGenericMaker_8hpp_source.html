172 void do_work(std::atomic<bool>& m_running_flag)
173 {
174 // Loop until a stop is received
175 while (m_running_flag.load()) {
176 // While there are items in the input queue, continue draining even if
177 // the running_flag is false, but stop _immediately_ when input is empty
178 IN in;
179 while (receive(in)) {
180 if (m_running_flag.load()) {
181 worker.process(in);
182 }
183 }
184 }
185 // P. Rodrigues 2022-06-01. The argument here is whether to drop
186 // buffered outputs. We choose 'true' because some significant
187 // time can pass between the last input sent by readout and when
188 // we receive a stop. (This happens because stop is sent serially
189 // to readout units before trigger, and each RU takes ~1s to
190 // stop). So by the time we receive a stop command, our buffered
191 // outputs are stale and will cause tardy warnings from the zipper
192 // downstream
193 worker.drain(true);
194 TLOG() << get_name() << ": Exiting do_work() method for run " << m_run_number << ", received " << m_received_count
195 << " inputs (" << worker.get_low_level_input_count() << " sub-inputs) and successfully sent " << m_sent_count
196 << " outputs. ";
197 worker.reset();
198 }

200 bool receive(IN& in)
201 {
202 try {
203 in = m_input_queue->receive(m_queue_timeout);
204 } catch (const dunedaq::iomanager::TimeoutExpired& excpt) {
205 // it is perfectly reasonable that there might be no data in the queue
206 // some fraction of the times that we check, so we just continue on and try again
207 return false;
208 }
209 ++m_received_count;
210 return true;
211 }

213 bool send(OUT&& out)
214 {
215 try {
216 m_output_queue->send(std::move(out), m_queue_timeout);
217 } catch (const dunedaq::iomanager::TimeoutExpired& excpt) {
218 ers::warning(excpt);
219 return false;
220 }
221 ++m_sent_count;
222 return true;
223 }

310 void process_slice(const std::vector<A>& time_slice, std::vector<B>& out_vec)
311 {
312 // time_slice is a full slice (all Set<A> combined), time ordered, vector of A
313 // call operator for each of the objects in the vector
314 for (const A& x : time_slice) {
315 try {
316 m_parent.m_maker->operator()(x, out_vec);
317 } catch (...) { // NOLINT TODO Benjamin Land <BenLand100@github.com> May 28-2021 can we restrict the possible
318 // exceptions triggeralgs might raise?
319 ers::fatal(AlgorithmFatalError(ERS_HERE, m_parent.get_name(), m_parent.m_algorithm_name));
320 return;
321 }

325 void process(Set<A>& in)
326 {
327 std::vector<B> elems; // Bs to buffer for the next window
328 switch (in.type) {
329 case Set<A>::Type::kPayload: {
330 if (m_prev_start_time != 0 && in.start_time < m_prev_start_time) {
331 ers::warning(OutOfOrderSets(ERS_HERE, m_parent.get_name(), m_prev_start_time, in.start_time));
332 }
333 m_prev_start_time = in.start_time;
334 std::vector<A> time_slice;
335 daqdataformats::timestamp_t start_time, end_time;
336 if (!m_in_buffer.buffer(in, time_slice, start_time, end_time)) {
337 return; // no complete time slice yet (`in` was part of buffered slice)
338 }
339 m_low_level_input_count += time_slice.size();
340 process_slice(time_slice, elems);
341 } break;
342 case Set<A>::Type::kHeartbeat: {
343 // PAR 2022-04-27 We've got a heartbeat for time T, so we know
344 // we won't receive any more inputs for times t < T. Therefore
345 // we can flush all items in the input buffer, which have
346 // times t < T, because the input is time-ordered. We put the
347 // heartbeat in the output buffer, which will handle it
348 // appropriately
349
350 std::vector<A> time_slice;
351 daqdataformats::timestamp_t start_time, end_time;
352 if (m_in_buffer.flush(time_slice, start_time, end_time)) {
353 if (end_time > in.start_time) {
354 // This should never happen, but we check here so we at least get some output if it did
355 ers::fatal(OutOfOrderSets(ERS_HERE, m_parent.get_name(), end_time, in.start_time));
356 }
357 m_low_level_input_count += time_slice.size();
358 process_slice(time_slice, elems);
359 }
360
361 Set<B> heartbeat;
362 heartbeat.type = Set<B>::Type::kHeartbeat;
363 heartbeat.start_time = in.start_time;
364 heartbeat.end_time = in.end_time;
365 heartbeat.origin = daqdataformats::SourceID(
366 daqdataformats::SourceID::Subsystem::kTrigger, m_parent.m_sourceid);
367
368 TLOG_DEBUG(4) << "Buffering heartbeat with start time " << heartbeat.start_time;
369 m_out_buffer.buffer_heartbeat(heartbeat);
370
371 // flush the maker
372 try {
373 // TODO Benjamin Land <BenLand100@github.com> July-14-2021 flushed events go into the buffer... until a window
374 // is ready?
375 m_parent.m_maker->flush(in.end_time, elems);
376 } catch (...) { // NOLINT TODO Benjamin Land <BenLand100@github.com> May-28-2021 can we restrict the possible
377 // exceptions triggeralgs might raise?
378 ers::fatal(AlgorithmFatalError(ERS_HERE, m_parent.get_name(), m_parent.m_algorithm_name));
379 return;
380 }
381 } break;
382 case Set<A>::Type::kUnknown:
383 ers::error(UnknownSetError(ERS_HERE, m_parent.get_name(), m_parent.m_algorithm_name));
384 break;
385 }
386
387 // add new elements to output buffer
388 if (elems.size() > 0) {
389 m_out_buffer.buffer(elems);
390 }
391
392 size_t n_output_windows=0;
393 // emit completed windows
394 while (m_out_buffer.ready()) {
395 ++n_output_windows;
396 Set<B> out;
397 m_out_buffer.flush(out);
398 out.seqno = m_parent.m_sent_count;
399 out.origin = daqdataformats::SourceID(
400 daqdataformats::SourceID::Subsystem::kTrigger, m_parent.m_sourceid);
401
402 if (out.type == Set<B>::Type::kHeartbeat) {
403 TLOG_DEBUG(4) << "Sending heartbeat with start time " << out.start_time;
404 if (!m_parent.send(std::move(out))) {
405 ers::error(AlgorithmFailedToSend(ERS_HERE, m_parent.get_name(), m_parent.m_algorithm_name));
406 // out is dropped
407 }
408 }
409 // Only form and send Set<B> if it has a nonzero number of objects
410 else if (out.type == Set<B>::Type::kPayload && out.objects.size() != 0) {
411 TLOG_DEBUG(4) << "Output set window ready with start time " << out.start_time << " end time " << out.end_time
412 << " and " << out.objects.size() << " members";
413 if (!m_parent.send(std::move(out))) {
414 ers::error(AlgorithmFailedToSend(ERS_HERE, m_parent.get_name(), m_parent.m_algorithm_name));
415 // out is dropped
416 }
417 }
418 }
419 TLOG_DEBUG(4) << "process() done. Advanced output buffer by " << n_output_windows << " output windows";
420 }

422 void drain(bool drop)
423 {
424 // First, send anything in the input buffer to the algorithm, and add any
425 // results to output buffer
426 std::vector<A> time_slice;
427 daqdataformats::timestamp_t start_time, end_time;
428 if (m_in_buffer.flush(time_slice, start_time, end_time)) {
429 std::vector<B> elems;
430 m_low_level_input_count += time_slice.size();
431 process_slice(time_slice, elems);
432 if (elems.size() > 0) {
433 m_out_buffer.buffer(elems);
434 }
435 }
436 // Second, drain the output buffer onto the queue. These may not be "fully
437 // formed" windows, but at this point we're getting no more data anyway.
438 while (!m_out_buffer.empty()) {
439 Set<B> out;
440 m_out_buffer.flush(out);
441 out.seqno = m_parent.m_sent_count;
442 out.origin = daqdataformats::SourceID(
443 daqdataformats::SourceID::Subsystem::kTrigger, m_parent.m_sourceid);
444
445 if (out.type == Set<B>::Type::kHeartbeat) {
446 if(!drop) {
447 if (!m_parent.send(std::move(out))) {
448 ers::error(AlgorithmFailedToSend(ERS_HERE, m_parent.get_name(), m_parent.m_algorithm_name));
449 // out is dropped
450 }
451 }
452 }
453 // Only form and send Set<B> if it has a nonzero number of objects
454 else if (out.type == Set<B>::Type::kPayload && out.objects.size() != 0) {
455 TLOG_DEBUG(1) << "Output set window ready with start time " << out.start_time << " end time " << out.end_time
456 << " and " << out.objects.size() << " members";
457 if (!drop) {
458 if (!m_parent.send(std::move(out))) {
459 ers::error(AlgorithmFailedToSend(ERS_HERE, m_parent.get_name(), m_parent.m_algorithm_name));
460 // out is dropped
461 }
462 }
463 }
464 }
465 }
