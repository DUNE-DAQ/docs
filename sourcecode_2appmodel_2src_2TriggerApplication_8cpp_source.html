82std::vector<const confmodel::DaqModule*>
83TriggerApplication::generate_modules(conffwk::Configuration* confdb,
84 const std::string& dbfile,
85 const confmodel::Session* /*session*/) const
86{
87 std::vector<const confmodel::DaqModule*> modules;
88
89 auto ti_conf = get_trigger_inputs_handler();
90 auto ti_class = ti_conf->get_template_for();
91 std::string handler_name("");
92 // Process the queue rules looking for inputs to our trigger handler modules
93 const QueueDescriptor* ti_inputq_desc = nullptr;
94
95 for (auto rule : get_queue_rules()) {
96 auto destination_class = rule->get_destination_class();
97 auto data_type = rule->get_descriptor()->get_data_type();
98 if (destination_class == "DataHandlerModule" || destination_class == ti_class) {
99 ti_inputq_desc = rule->get_descriptor();
100 }
101 }
102 // Process the network rules looking for the Fragment Aggregator and TP handler data reuest inputs
103 const NetworkConnectionDescriptor* req_net_desc = nullptr;
104 const NetworkConnectionDescriptor* tin_net_desc = nullptr;
105 const NetworkConnectionDescriptor* tout_net_desc = nullptr;
106 const NetworkConnectionDescriptor* tset_out_net_desc = nullptr;
107 for (auto rule : get_network_rules()) {
108 auto endpoint_class = rule->get_endpoint_class();
109 auto data_type = rule->get_descriptor()->get_data_type();
110
111 if (data_type == "DataRequest") {
112 req_net_desc = rule->get_descriptor();
113 }
114 else if (data_type == "TASet" || data_type == "TCSet"){
115 tset_out_net_desc = rule->get_descriptor();
116 }
117 else if (endpoint_class == "DataSubscriberModule") {
118 if (!tin_net_desc) {
119 tin_net_desc = rule->get_descriptor();
120 }
121 else if (rule->get_descriptor()->get_data_type() == tin_net_desc->get_data_type()) {
122 // For now endpoint_class of DataSubscriberModule for both input and output
123 // with the same data type is not possible.
124 throw (BadConf(ERS_HERE, "Have two network connections of the same data_type and the same endpoint_class"));
125 }
126 else if (tin_net_desc->get_data_type() == "TriggerActivity" &&
127 rule->get_descriptor()->get_data_type() == "TriggerCandidate") {
128 // For TA->TC
129 tout_net_desc = rule->get_descriptor();
130 handler_name = "tahandler";
131 }
132 else if (tin_net_desc->get_data_type() == "TriggerCandidate" &&
133 rule->get_descriptor()->get_data_type() == "TriggerActivity") {
134 // For TA->TC if we saved TC network connection as input first...
135 tout_net_desc = tin_net_desc;
136 tin_net_desc = rule->get_descriptor();
137 handler_name = "tahandler";
138 }
139 else {
140 throw (BadConf(ERS_HERE, "Unexpected input & output network connection descriptors provided"));
141 }
142 }
143 else if (data_type == "TriggerActivity" || data_type == "TriggerCandidate"){
144 tout_net_desc = rule->get_descriptor();
145 if (data_type == "TriggerActivity")
146 handler_name = "tphandler";
147 else
148 handler_name = "tahandler";
149 }
150 }
151
152 // Now create the Data Handler and its associated queue and network
153 // connections
154 conffwk::ConfigObject input_queue_obj;
155 conffwk::ConfigObject req_net_obj;
156 conffwk::ConfigObject tin_net_obj;
157 conffwk::ConfigObject tout_net_obj;
158 conffwk::ConfigObject tset_out_net_obj;
159 //auto handlerConf = get_trigger_inputs_handler();
160
161 if ( req_net_desc== nullptr) {
162 throw (BadConf(ERS_HERE, "No network descriptor given to receive request and send data was set"));
163 }
164 if ( tin_net_desc== nullptr) {
165 throw (BadConf(ERS_HERE, "No network descriptor given to receive trigger objects"));
166 }
167 if ( tout_net_desc== nullptr) {
168 throw (BadConf(ERS_HERE, "No network descriptor given to publish trigger objects"));
169 }
170 if (ti_inputq_desc == nullptr) {
171 throw (BadConf(ERS_HERE, "No data input queue descriptor given"));
172 }
173
174 std::string queue_uid(ti_inputq_desc->get_uid_base());
175 confdb->create(dbfile, "Queue", queue_uid, input_queue_obj);
176 input_queue_obj.set_by_val<std::string>("data_type", ti_inputq_desc->get_data_type());
177 input_queue_obj.set_by_val<std::string>("queue_type", ti_inputq_desc->get_queue_type());
178 input_queue_obj.set_by_val<uint32_t>("capacity", ti_inputq_desc->get_capacity());
179
180 auto req_service_obj = req_net_desc->get_associated_service()->config_object();
181 std::string req_net_uid(req_net_desc->get_uid_base()+UID());;
182 confdb->create(dbfile, "NetworkConnection", req_net_uid, req_net_obj);
183 req_net_obj.set_by_val<std::string>("connection_type", req_net_desc->get_connection_type());
184 req_net_obj.set_by_val<std::string>("data_type", req_net_desc->get_data_type());
185 req_net_obj.set_obj("associated_service", &req_service_obj);
186
187 auto tin_service_obj = tin_net_desc->get_associated_service()->config_object();
188 std::string t_in_stream_uid(tin_net_desc->get_uid_base()+".*");
189 confdb->create(dbfile, "NetworkConnection", t_in_stream_uid, tin_net_obj);
190 tin_net_obj.set_by_val<std::string>("data_type", tin_net_desc->get_data_type());
191 tin_net_obj.set_by_val<std::string>("connection_type", tin_net_desc->get_connection_type());
192 tin_net_obj.set_obj("associated_service", &tin_service_obj);
193
194 auto tout_service_obj = tout_net_desc->get_associated_service()->config_object();
195 std::string t_stream_uid(tout_net_desc->get_uid_base()+UID());
196 confdb->create(dbfile, "NetworkConnection", t_stream_uid, tout_net_obj);
197 tout_net_obj.set_by_val<std::string>("data_type", tout_net_desc->get_data_type());
198 tout_net_obj.set_by_val<std::string>("connection_type", tout_net_desc->get_connection_type());
199 tout_net_obj.set_obj("associated_service", &tout_service_obj);
200
201
202
203 if (tset_out_net_desc) {
204 auto tset_out_service_obj = tset_out_net_desc->get_associated_service()->config_object();
205 std::string tset_stream_uid(tset_out_net_desc->get_uid_base()+UID());
206 confdb->create(dbfile, "NetworkConnection", tset_stream_uid, tset_out_net_obj);
207 tset_out_net_obj.set_by_val<std::string>("data_type", tset_out_net_desc->get_data_type());
208 tset_out_net_obj.set_by_val<std::string>("connection_type", tset_out_net_desc->get_connection_type());
209 tset_out_net_obj.set_obj("associated_service", &tset_out_service_obj);
210 }
211
212 auto ti_conf_obj = ti_conf->config_object();
213 conffwk::ConfigObject ti_obj;
214 if (get_source_id() == nullptr) {
215 throw(BadConf(ERS_HERE, "No source_id associated with this TriggerApplication!"));
216 }
217 uint32_t source_id = get_source_id()->get_sid();
218 std::string ti_uid(handler_name + "-" + std::to_string(source_id));
219 confdb->create(dbfile, ti_class, ti_uid, ti_obj);
220 ti_obj.set_by_val<uint32_t>("source_id", source_id);
221 ti_obj.set_by_val<uint32_t>("detector_id", 1); // 1 == kDAQ
222 ti_obj.set_obj("module_configuration", &ti_conf_obj);
223 ti_obj.set_objs("inputs", {&input_queue_obj, &req_net_obj});
224 if (tset_out_net_desc!= nullptr) {
225 ti_obj.set_objs("outputs", {&tout_net_obj, &tset_out_net_obj});
226 }
227 else {
228 ti_obj.set_objs("outputs", {&tout_net_obj});
229 }
230 // Add to our list of modules to return
231 modules.push_back(confdb->get<DataHandlerModule>(ti_uid));
232
233
234 // Now create the DataSubscriberModule object
235 auto rdr_conf = get_data_subscriber();
236 if (rdr_conf == nullptr) {
237 throw (BadConf(ERS_HERE, "No DataReaderModule configuration given"));
238 }
239
240 // Create a DataReaderModule
241
242 std::string reader_uid("data-reader-"+UID());
243 std::string reader_class = rdr_conf->get_template_for();
244 conffwk::ConfigObject reader_obj;
245 TLOG_DEBUG(7) << "creating OKS configuration object for Data subscriber class " << reader_class;
246 confdb->create(dbfile, reader_class, reader_uid, reader_obj);
247 reader_obj.set_objs("inputs", {&tin_net_obj} );
248 reader_obj.set_objs("outputs", {&input_queue_obj} );
249 reader_obj.set_obj("configuration", &rdr_conf->config_object());
250
251 modules.push_back(confdb->get<DataSubscriberModule>(reader_uid));
252
253 return modules;
254}
