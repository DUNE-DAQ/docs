294struct ColdataBlock
295{
296 static constexpr int s_num_seg_per_block = 8;
297 static constexpr int s_num_ch_per_adc = 8;
298 static constexpr int s_num_adc_per_block = ColdataSegment::s_num_ch_per_seg * s_num_seg_per_block / s_num_ch_per_adc;
299 static constexpr int s_num_ch_per_block = s_num_seg_per_block * ColdataSegment::s_num_ch_per_seg;
300
301 ColdataHeader head;
302 ColdataSegment segments[s_num_seg_per_block]; // NOLINT
303
304 uint16_t get_channel(const uint8_t adc, const uint8_t ch) const // NOLINT(build/unsigned)
305 {
306 // Each segment houses one half (four channels) of two subsequent ADCs.
307 return segments[get_segment_index_(adc, ch)].get_channel(adc, ch);
308 }
309
310 void set_channel(const uint8_t adc, const uint8_t ch, const uint16_t new_val) // NOLINT(build/unsigned)
311 {
312 segments[get_segment_index_(adc, ch)].set_channel(adc, ch, new_val);
313 }
314
315private:
316 int get_segment_index_(const int adc, const int ch) const
317 {
318 auto segment_id = (adc / 2) * 2 + ch / 4;
319
320 if (segment_id < 0 || segment_id > s_num_seg_per_block - 1) {
321 throw std::out_of_range("ADC index out of range");
322 }
323 return segment_id;
324 }
325};
