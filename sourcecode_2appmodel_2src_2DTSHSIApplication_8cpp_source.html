158 hsiNetObj.set_obj("associated_service", &hsiServiceObj);

161 conffwk::ConfigObject hsiObj;
162 confdb->create(dbfile, "HSIReadout", genuid, hsiObj);

164 hsiObj.set_objs("outputs", { &queueObj, &hsiNetObj });

166 modules.push_back(confdb->get<HSIReadout>(genuid));
