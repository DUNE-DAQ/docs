21void
22TAMakerHorizontalMuonAlgorithm::process(const TriggerPrimitive& input_tp,
23 std::vector<TriggerActivity>& output_ta)
24{
25
26 uint16_t adjacency;
27
28 // Add useful info about recived TPs here for FW and SW TPG guys.
29 if (m_print_tp_info) {
30 TLOG_DEBUG(TLVL_DEBUG_ALL) << "[TAM:HM] TP Start Time: " << input_tp.time_start
31 << ", TP ADC Sum: " << input_tp.adc_integral
32 << ", TP TOT: " << input_tp.time_over_threshold << ", TP ADC Peak: " << input_tp.adc_peak
33 << ", TP Offline Channel ID: " << input_tp.channel;
34 TLOG_DEBUG(TLVL_DEBUG_ALL) << "[TAM:HM] Adjacency of current window is: " << check_adjacency();
35 }
36
37 // 0) FIRST TP =====================================================================
38 // The first time process() is called, reset the window object.
39 if (m_current_window.is_empty()) {
40 m_current_window.reset(input_tp);
41 return;
42 }
43
44 // If the difference between the current TP's start time and the start of the window
45 // is less than the specified window size, add the TP to the window.
46 if ((input_tp.time_start - m_current_window.time_start) < m_window_length) {
47 m_current_window.add(input_tp);
48 }
49
50 // 1) ADC THRESHOLD EXCEEDED =======================================================
51 // If the addition of the current TP to the window would make it longer specified
52 // window length, don't add it but check whether the ADC integral if the existing
53 // window is above the configured threshold. If it is, and we are triggering on ADC,
54 // make a TA and start a fresh window with the current TP.
55 else if (m_current_window.adc_integral > m_adc_threshold && m_trigger_on_adc) {
56
57 auto ta = construct_ta();
58
59 TLOG_DEBUG(TLVL_DEBUG_MEDIUM) << "[TA:HM]: Emitting ADC threshold trigger with " << m_current_window.adc_integral
60 << " window ADC integral. ta.time_start=" << ta.time_start
61 << " ta.time_end=" << ta.time_end;
62
63 output_ta.push_back(ta);
64 m_current_window.reset(input_tp);
65 }
66
67 // 2) MULTIPLICITY - N UNQIUE CHANNELS EXCEEDED =====================================
68 // If the addition of the current TP to the window would make it longer than the
69 // specified window length, don't add it but check whether the number of hit channels
70 // in the existing window is above the specified threshold. If it is, and we are triggering
71 // on channel multiplicity, make a TA and start a fresh window with the current TP.
72 else if (m_current_window.n_channels_hit() > m_n_channels_threshold && m_trigger_on_n_channels) {
73
74 TLOG_DEBUG(TLVL_DEBUG_MEDIUM) << "[TAM:HM] Emitting multiplicity trigger with "
75 << m_current_window.n_channels_hit() << " unique channels hit.";
76
77 output_ta.push_back(construct_ta());
78 m_current_window.reset(input_tp);
79 }
80
81 // 3) ADJACENCY THRESHOLD EXCEEDED ==================================================
82 // If the addition of the current TP to the window would make it longer than the
83 // specified window length, don't add it but check whether the adjacency of the
84 // current window exceeds the configured threshold. If it does, and we are triggering
85 // on adjacency, then create a TA and reset the window with the new/current TP.
86 else if ((adjacency = check_adjacency()) > m_adjacency_threshold && m_trigger_on_adjacency) {
87
88 //for (auto tp : m_current_window.inputs){ dump_tp(tp); }
89
90 // Check for a new maximum, display the largest seen adjacency in the log.
91 // uint16_t adjacency = check_adjacency();
92 if (adjacency > m_max_adjacency) {
93 m_max_adjacency = adjacency;
94 }
95 TLOG_DEBUG(TLVL_DEBUG_MEDIUM) << "[TAM:HM] Emitting track and multiplicity TA with adjacency "
96 << check_adjacency() << " and multiplicity " << m_current_window.n_channels_hit()
97 << ". The ADC integral of this TA is " << m_current_window.adc_integral
98 << " and the largest longest track seen so far is " << m_max_adjacency;
99
100 output_ta.push_back(construct_ta());
101 m_current_window.reset(input_tp);
102 }
103
104 // Temporary triggering logic for Adam's large TOT TPs. Trigger on very large TOT TPs.
105 else if (m_trigger_on_tot && input_tp.time_over_threshold > m_tot_threshold) {
106
107 // If the incoming TP has a large time over threshold, we might have a cluster of
108 // interesting physics activity surrounding it. Trigger on that.
109 TLOG_DEBUG(TLVL_DEBUG_MEDIUM) << "[TAM:HM] Emitting a TA due to a TP with a very large time over threshold: "
110 << input_tp.time_over_threshold << " ticks and offline channel: " << input_tp.channel
111 << ", where the ADC integral of that TP is " << input_tp.adc_integral;
112 output_ta.push_back(construct_ta());
113 m_current_window.reset(input_tp);
114 }
115
116 // 4) Otherwise, slide the window along using the current TP.
117 else {
118 m_current_window.move(input_tp, m_window_length);
119 }
120
121 return;
122}

156TriggerActivity
157TAMakerHorizontalMuonAlgorithm::construct_ta() const
158{
159
160 TriggerActivity ta;
161
162 TriggerPrimitive last_tp = m_current_window.inputs.back();
163
164 ta.time_start = last_tp.time_start;
165 ta.time_end = last_tp.time_start + last_tp.time_over_threshold;
166 ta.time_peak = last_tp.time_peak;
167 ta.time_activity = last_tp.time_peak;
168 ta.channel_start = last_tp.channel;
169 ta.channel_end = last_tp.channel;
170 ta.channel_peak = last_tp.channel;
171 ta.adc_integral = m_current_window.adc_integral;
172 ta.adc_peak = last_tp.adc_integral;
173 ta.detid = last_tp.detid;
174 ta.type = TriggerActivity::Type::kTPC;
175 ta.algorithm = TriggerActivity::Algorithm::kHorizontalMuon;
176 ta.inputs = m_current_window.inputs;
177
178 for( const auto& tp : ta.inputs ) {
179 ta.time_start = std::min(ta.time_start, tp.time_start);
180 ta.time_end = std::max(ta.time_end, tp.time_start + tp.time_over_threshold);
181 ta.channel_start = std::min(ta.channel_start, tp.channel);
182 ta.channel_end = std::max(ta.channel_end, tp.channel);
183 if (tp.adc_peak > ta.adc_peak) {
184 ta.time_peak = tp.time_peak;
185 ta.adc_peak = tp.adc_peak;
186 ta.channel_peak = tp.channel;
187 }
188 }
189
190 return ta;
191}

193uint16_t
194TAMakerHorizontalMuonAlgorithm::check_adjacency() const
195{
196 // This function returns the adjacency value for the current window, where adjacency
197 // is defined as the maximum number of consecutive wires containing hits. It accepts
198 // a configurable tolerance paramter, which allows up to adj_tolerance missing hits
199 // on adjacent wires before restarting the adjacency count. The maximum gap is 4 which
200 // comes from tuning on December 2021 coldbox data, and June 2022 coldbox runs.
201
202 uint16_t adj = 1; // Initialise adjacency, 1 for the first wire.
203 uint16_t max = 0; // Maximum adjacency of window, which this function returns
204 unsigned int channel = 0; // Current channel ID
205 unsigned int next_channel = 0; // Next channel ID
206 unsigned int next = 0; // The next position in the hit channels vector
207 unsigned int tol_count = 0; // Tolerance count, should not pass adj_tolerance
208
209 // Generate a channelID ordered list of hit channels for this window
210 std::vector<int> chanList;
211 for (auto tp : m_current_window.inputs) {
212 chanList.push_back(tp.channel);
213 }
214 std::sort(chanList.begin(), chanList.end());
215
216 // ADAJACENCY LOGIC ====================================================================
217 // =====================================================================================
218 // Adjcancency Tolerance = Number of times prepared to skip missed hits before resetting
219 // the adjacency count. This accounts for things like dead channels / missed TPs. The
220 // maximum gap is 4 which comes from tuning on December 2021 coldbox data, and June 2022
221 // coldbox runs.
222 for (int i = 0; i < chanList.size(); ++i) {
223
224 next = (i + 1) % chanList.size(); // Loops back when outside of channel list range
225 channel = chanList.at(i);
226 next_channel = chanList.at(next); // Next channel with a hit
227
228 // End of vector condition.
229 if (next_channel == 0) {
230 next_channel = channel - 1;
231 }
232
233 // Skip same channel hits.
234 if (next_channel == channel) {
235 continue;
236 }
237
238 // If next hit is on next channel, increment the adjacency count.
239 else if (next_channel == channel + 1) {
240 ++adj;
241 }
242
243 // If next channel is not on the next hit, but the 'second next', increase adjacency
244 // but also tally up with the tolerance counter.
245 else if (((next_channel == channel + 2) || (next_channel == channel + 3) || (next_channel == channel + 4) ||
246 (next_channel == channel + 5)) &&
247 (tol_count < m_adj_tolerance)) {
248 ++adj;
249 for (int i = 0; i < next_channel - channel; ++i) {
250 ++tol_count;
251 }
252 }
253
254 // If next hit isn't within reach, end the adjacency count and check for a new max.
255 // Reset variables for next iteration.
256 else {
257 if (adj > max) {
258 max = adj;
259 }
260 adj = 1;
261 tol_count = 0;
262 }
263 }
264
265 return max;
266}
