120 }

146 datahandlinglibs::opmon::FixedRateDataProcessorInfo info;
147
148 info.set_num_seq_id_errors(m_seq_id_error_ctr.load());
149 info.set_min_seq_id_jump(m_seq_id_min_jump.exchange(0));
150 info.set_max_seq_id_jump(m_seq_id_max_jump.exchange(0));
151
152 info.set_num_ts_errors(m_ts_error_ctr.load());
153
154 publish(std::move(info));
155
156 if (m_post_processing_enabled) {
157 auto now = std::chrono::high_resolution_clock::now();
158 int new_hits = m_tpg_hits_count.exchange(0);
159 int new_tps = m_new_tps.exchange(0);
160 int new_tps_suppressed_too_long = m_tps_suppressed_too_long.exchange(0);
161 int new_tps_send_failed = m_tps_send_failed.exchange(0);
162 double seconds = std::chrono::duration_cast<std::chrono::microseconds>(now - m_t0).count() / 1000000.;
163 TLOG_DEBUG(TLVL_BOOKKEEPING) << "Hit rate: " << std::to_string(new_hits / seconds / 1000.) << " [kHz]";
164 TLOG_DEBUG(TLVL_BOOKKEEPING) << "Total new hits: " << new_hits << " new TPs: " << new_tps;
165
166 datahandlinglibs::opmon::HitFindingInfo tp_info;
167 tp_info.set_rate_tp_hits(new_hits / seconds / 1000.);
168
169 tp_info.set_num_tps_sent(new_tps);
170 tp_info.set_num_tps_suppressed_too_long(new_tps_suppressed_too_long);
171 tp_info.set_num_tps_send_failed(new_tps_send_failed);
172
173 publish(std::move(tp_info));
174 // Find the channels with the top TP rates
175 // Create a vector of pairs to store the map elements
176 std::vector<std::pair<uint, int>> channel_tp_rate_vec(m_tp_channel_rate_map.begin(), m_tp_channel_rate_map.end());
177 // Sort the vector in descending order of the value of the pairs
178 sort(channel_tp_rate_vec.begin(), channel_tp_rate_vec.end(), [](std::pair<uint, int>& a, std::pair<uint, int>& b) { return a.second > b.second; });
179 // Add the metrics to opmon
180 // For convenience we are selecting only the top 10 elements
181 if (channel_tp_rate_vec.size() != 0) {
182 int top_highest_values = 10;
183 if (channel_tp_rate_vec.size() < 10) {
184 top_highest_values = channel_tp_rate_vec.size();
185 }
186 //datahandlinglibs::opmon::TPChannelsInfo channels_info;
187 for (int i = 0; i < top_highest_values; i++) {
188 datahandlinglibs::opmon::TPChannelInfo tpc_info;
189 tpc_info.set_number_of_tps(channel_tp_rate_vec[i].second);
190 tpc_info.set_channel_id(channel_tp_rate_vec[i].first);
191 publish(std::move(tpc_info), {{"channel", std::to_string(channel_tp_rate_vec[i].first)}});
192 }
193 }
194
195 // Reset the counter in the channel rate map
196 for (auto& el : m_tp_channel_rate_map) {
197 el.second = 0;
198 }
199 m_t0 = now;
200 }
201 inherited::generate_opmon_data();
202 }

321void
322WIBEthFrameProcessor::find_hits(constframeptr fp)
323{
324 size_t nhits = 0;
325 if (!fp)
326 return;
327 auto wfptr = reinterpret_cast<dunedaq::fddetdataformats::WIBEthFrame*>((uint8_t*)fp); // NOLINT
328
329 // Check that the system is properly configured from the first hit.
330 if (m_first_hit) {
331 if (wfptr->daq_header.crate_id != m_crate_id || wfptr->daq_header.slot_id != m_slot_id || wfptr->daq_header.stream_id != m_stream_id) {
332 ers::error(LinkMisconfiguration(ERS_HERE, wfptr->daq_header.crate_id, wfptr->daq_header.slot_id, wfptr->daq_header.stream_id, m_crate_id, m_slot_id, m_stream_id));
333 }
334
335 m_first_hit = false;
336 }
337
338 std::vector<trgdataformats::TriggerPrimitive> tps = (*m_tp_generator)(wfptr);
339
340 for (auto tp : tps) {
341 // If this TP is on a masked channel, skip it.
342 if (std::binary_search(m_channel_mask_set.begin(), m_channel_mask_set.end(), tp.channel))
343 continue;
344 // Need to move into a type adapter.
345 trigger::TriggerPrimitiveTypeAdapter tpa;
346 tpa.tp = tp;
347 tpa.tp.detid = m_det_id; // Last missing piece.
348 m_tp_channel_rate_map[tp.channel]++;
349 if(!m_tp_sink[m_channel_map->get_plane_from_offline_channel(tp.channel)]->try_send(std::move(tpa), iomanager::Sender::s_no_block)) {
350 ers::warning(FailedToSendTP(ERS_HERE, tp.time_start, tp.channel));
351 m_tps_send_failed++;
352 } else {
353 m_new_tps++;
354 ++nhits;
355 }
356 }
357 m_tpg_hits_count += nhits;
358 return;
359}
