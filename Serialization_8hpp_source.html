202template<class T>
203std::vector<uint8_t> // NOLINT(build/unsigned)
204serialize(const T& obj, SerializationType stype)
205{
206 switch (stype) {
207 case kJSON: {
208 nlohmann::json j = obj;
209 nlohmann::json::string_t s = j.dump();
210 std::vector<uint8_t> ret(s.size() + 1); // NOLINT(build/unsigned)
211 ret[0] = serialization_type_byte(stype);
212 std::copy(s.begin(), s.end(), ret.begin() + 1); // NOLINT
213 return ret;
214 }
215 case kMsgPack: {
216 // Serialize into the sbuffer and then copy to a
217 // std::vector. Seems like it would be more efficient to
218 // write directly to the vector (by creating a class that
219 // implements `void write(char* buf, size_t len)`), but my
220 // tests aren't any faster than this
221 msgpack::sbuffer buf;
222 msgpack::pack(buf, obj);
223 std::vector<uint8_t> ret(buf.size() + 1); // NOLINT(build/unsigned)
224 ret[0] = serialization_type_byte(stype);
225 std::copy(buf.data(), buf.data() + buf.size(), ret.begin() + 1); // NOLINT
226 return ret;
227 }
228 default:
229 throw UnknownSerializationTypeEnum(ERS_HERE);
230 }
231}

236template<class T, typename CharType = unsigned char>
237T
238deserialize(const std::vector<CharType>& v)
239{
240 using json = nlohmann::json;
241
242 // The first byte in the array indicates the serialization format;
243 // the rest is the actual message
244 switch (v[0]) {
245 case serialization_type_byte(kJSON): {
246 try {
247 json j = json::parse(v.begin() + 1, v.end());
248 return j.get<T>();
249 } catch (json::exception& e) {
250 throw CannotDeserializeMessage(ERS_HERE, e);
251 }
252 }
253 case serialization_type_byte(kMsgPack): {
254 try {
255 // The lambda function here is of type `unpack_reference_func`
256 // as described at
257 // https://github.com/msgpack/msgpack-c/wiki/v2_0_cpp_unpacker#memory-management
258 // . It is called for every STR, BIN and EXT field in the
259 // MsgPack data. If the function returns false, the object is
260 // copied into MsgPack's "zone", otherwise a pointer to the
261 // original buffer is stored. Our input buffer is going to exist
262 // at least until the end of this function, so it's safe to
263 // return true (ie, store a pointer in the MsgPack object; no
264 // copy) everywhere. Doing so results in a factor ~2 speedup in
265 // deserializing Fragment, which is just a large BIN field
266 msgpack::object_handle oh = msgpack::unpack(
267 const_cast<char*>(reinterpret_cast<const char*>(v.data() + 1)),
268 v.size() - 1,
269 [](msgpack::type::object_type /*typ*/, std::size_t /*length*/, void* /*user_data*/) -> bool { return true; });
270 msgpack::object obj = oh.get();
271 return obj.as<T>();
272 } catch (msgpack::type_error& e) {
273 throw CannotDeserializeMessage(ERS_HERE, e);
274 } catch (msgpack::unpack_error& e) {
275 throw CannotDeserializeMessage(ERS_HERE, e);
276 }
277 }
278 default:
279 throw UnknownSerializationTypeByte(ERS_HERE, (char)v[0]); // NOLINT
280 }
