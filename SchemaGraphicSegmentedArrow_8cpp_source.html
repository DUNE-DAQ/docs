275QPolygonF SchemaGraphicSegmentedArrow::make_arrow_head(qreal rotation) const {
276
277 float opening_angle = atan2(1, 0.5);
278 qreal size = m_arrow_size;
279 QPointF a1 = QPointF ( sin ( rotation + opening_angle ) * size, cos ( rotation + opening_angle ) * size );
280 QPointF a2 = QPointF ( sin ( rotation + M_PI - opening_angle ) * size, cos ( rotation + M_PI - opening_angle ) * size );
281
282 QPolygonF arrow;
283 arrow << QPointF(0,0) << a1 << a2;
284 return arrow;
285}

287QPolygonF SchemaGraphicSegmentedArrow::make_rhombus(qreal rotation) const {
288
289 float opening_angle = M_PI / 3;
290 qreal size = 0.6*m_arrow_size;
291
292 QPointF a1 = QPointF ( sin ( rotation + opening_angle ) * size, cos ( rotation + opening_angle ) * size );
293 QPointF a2 = QPointF ( sin ( rotation + M_PI - opening_angle ) * size, cos ( rotation + M_PI - opening_angle ) * size );
294
295 QPointF m = QPointF ( ( a1.x() + a2.x() ) / 2, ( a1.y() + a2.y() ) / 2 );
296 QPointF a3 = QPointF ( 2 * m.x(), 2 * m.y() );
297
298 QPolygonF rhombus;
299 rhombus << QPointF(0,0) << a1 << a3 << a2;
300 return rhombus;
301}
