101template<>
102struct adl_serializer<dunedaq::daqdataformats::Fragment>
103{
104 // note: the return type is no longer 'void', and the method only takes
105 // one argument
106 static dunedaq::daqdataformats::Fragment from_json(const json& j)
107 {
108 std::vector<uint8_t> tmp; // NOLINT(build/unsigned)
109 for (auto const& it : j.items()) {
110 if (!it.value().is_number_integer()) {
111 throw dunedaq::dfmessages::CannotDeserializeFragment(ERS_HERE);
112 }
113 tmp.push_back(it.value().get<uint8_t>()); // NOLINT(build/unsigned)
114 }
115 return dunedaq::daqdataformats::Fragment(tmp.data(),
116 dunedaq::daqdataformats::Fragment::BufferAdoptionMode::kCopyFromBuffer);
117 }
118

120 {
121 const uint8_t* storage = static_cast<const uint8_t*>(frag.get_storage_location()); // NOLINT(build/unsigned)
122 std::vector<uint8_t> bytes(storage, storage + frag.get_size()); // NOLINT(build/unsigned)
123 j = bytes;
124 }
