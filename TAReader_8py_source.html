96 def read_fragment(self, fragment_path: str) -> NDArray:
97 """
98 Read from the given data fragment path.
99
100 Returns a np.ndarray of the first TA that was read and appends all TAs in the fragment to :self.ta_data:.
101 """
102 if self._verbosity >= 2:
103 print("="*60)
104 print(f"INFO: Reading from the path\n{fragment_path}")
105
106 fragment = self._h5_file.get_frag(fragment_path)
107 fragment_data_size = fragment.get_data_size()
108
109 if fragment_data_size == 0:
110 self._num_empty += 1
111 if self._verbosity >= 1:
112 print(
113 self._FAIL_TEXT_COLOR

118 print("="*60)

121 ta_idx = 0 # Debugging output.
122 byte_idx = 0 # Variable TA sizing, must do while loop.
123 while byte_idx < fragment_data_size:
124 if self._verbosity >= 2:
125 print(f"INFO: Fragment Index: {ta_idx}.")
126 ta_idx += 1
127 print(f"INFO: Byte Index / Frag Size: {byte_idx} / {fragment_data_size}")

129 # Read TA data
130 ta_datum = trgdataformats.TriggerActivity(fragment.get_data(byte_idx))
131 np_ta_datum = np.array([(
132 ta_datum.data.adc_integral,
133 ta_datum.data.adc_peak,
134 ta_datum.data.algorithm,
135 ta_datum.data.channel_end,
136 ta_datum.data.channel_peak,

142 ta_datum.data.time_peak,

144 ta_datum.data.type,
145 np.uint16(ta_datum.data.version))],
146 dtype=self.ta_dt)
147
148 self.ta_data = np.hstack((self.ta_data, np_ta_datum))

150 byte_idx += ta_datum.sizeof()
151 if self._verbosity >= 2:

153

162 tp.detid,

164 tp.time_over_threshold,
165 tp.time_peak,
166 tp.time_start,
167 tp.type,
168 tp.version)],
169 dtype=self.tp_dt)
