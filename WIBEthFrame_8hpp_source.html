94 uint16_t get_adc(int i, int sample=0) const // NOLINT(build/unsigned)
95 {
96 if (i < 0 || i >= s_num_channels)
97 throw std::out_of_range("ADC index out of range");
98
99 // The index of the first (and sometimes only) word containing the required ADC value
100 int word_index = s_bits_per_adc * i / s_bits_per_word;
101 assert(word_index < s_num_adc_words_per_ts);
102 // Where in the word the lowest bit of our ADC value is located
103 int first_bit_position = (s_bits_per_adc * i) % s_bits_per_word;
104 // How many bits of our desired ADC are located in the `word_index`th word
105 int bits_from_first_word = std::min(s_bits_per_adc, s_bits_per_word - first_bit_position);
106 // uint16_t adc = adc_words[word_index][sample] >> first_bit_position; // NOLINT(build/unsigned)
107 uint16_t adc = adc_words[sample][word_index] >> first_bit_position; // NOLINT(build/unsigned)
108 // If we didn't get the full 14 bits from this word, we need the rest from the next word
109 if (bits_from_first_word < s_bits_per_adc) {
110 assert(word_index + 1 < s_num_adc_words_per_ts);
111 // adc |= adc_words[word_index + 1][sample] << bits_from_first_word;
112 adc |= adc_words[sample][word_index + 1] << bits_from_first_word;
113 }
114 // Mask out all but the lowest 14 bits;
115 return adc & 0x3FFFu;
116 }

121 void set_adc(int i, int sample, uint16_t val) // NOLINT(build/unsigned)
122 {
123 if (i < 0 || i >= s_num_channels)
124 throw std::out_of_range("ADC index out of range");
125 if (val >= (1 << s_bits_per_adc))
126 throw std::out_of_range("ADC value out of range");
127
128 // The index of the first (and sometimes only) word containing the required ADC value
129 int word_index = s_bits_per_adc * i / s_bits_per_word;
130 assert(word_index < s_num_adc_words);
131 // Where in the word the lowest bit of our ADC value is located
132 int first_bit_position = (s_bits_per_adc * i) % s_bits_per_word;
133 // How many bits of our desired ADC are located in the `word_index`th word
134 int bits_in_first_word = std::min(s_bits_per_adc, s_bits_per_word - first_bit_position);
135 uint64_t mask = (static_cast<uint64_t>(1) << first_bit_position) - 1;
136 adc_words[sample][word_index] = ((static_cast<uint64_t>(val) << first_bit_position) & ~mask) | (adc_words[sample][word_index] & mask);
137 // If we didn't put the full 14 bits in this word, we need to put the rest in the next word
138 if (bits_in_first_word < s_bits_per_adc) {
139 assert(word_index + 1 < s_num_adc_words);
140 mask = (1 << (s_bits_per_adc - bits_in_first_word)) - 1;
141 adc_words[sample][word_index + 1] = ((val >> bits_in_first_word) & mask) | (adc_words[sample][word_index + 1] & ~mask);
142 }
143 }
