172NetworkReceiverModel<Datatype>::add_callback_impl(std::function<void(MessageType&)> callback)
173{
174 remove_callback();

176 std::lock_guard<std::mutex> lk(m_callback_mutex);

178 TLOG() << "Registering callback.";
