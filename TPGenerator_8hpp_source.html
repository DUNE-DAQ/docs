51 template <typename T>
52 std::vector<dunedaq::trgdataformats::TriggerPrimitive> operator()(const T* frame) {
53 // Max number of TPs for a channel: number of time samples / 2.
54 std::vector<dunedaq::trgdataformats::TriggerPrimitive> tp_aggr;
55 tp_aggr.reserve(T::s_num_channels * T::s_time_samples_per_frame / 2);
56
57 const typename T::word_t (*words_ptr)[T::s_bits_per_adc] = frame->adc_words;
58 const uint64_t timestamp = frame->get_timestamp();
59
60 const int register_alignment = T::s_bits_per_adc * m_num_channels_per_pipeline;
61 // Loop in time.
62 for (int t = 0; t < T::s_time_samples_per_frame; t++) {
63 const typename T::word_t *time_sample = *(words_ptr + t);
64 char* cursor = (char*) time_sample; // Need to walk in terms of bytes/bits.
65
66 // Loop in pipelines.
67 for (int p = 0; p < m_num_pipelines; p++) {
68 if (p == m_num_pipelines - 1)
69 cursor -= 4; // Take a step of 32 bit backwards for the last sub-frame.
70
71 __m256i regi = _mm256_lddqu_si256((__m256i*)cursor);
72
73 if (p == m_num_pipelines - 1) // Permute the row order to use the same operation.
74 regi = _mm256_permutevar8x32_epi32(regi, _mm256_setr_epi32(1, 2, 3, 4, 5, 6, 7, 0));
75
76 __m256i expanded_subframe = expand_frame(regi);
77 std::vector<dunedaq::trgdataformats::TriggerPrimitive> tps = m_tpg_pipelines[p].process(expanded_subframe);
78
79 // Need to insert all the TPs while scaling and shifting time_start and time_peak.
80 for (auto tp : tps) {
81 tp.time_start = (t - tp.time_over_threshold) * m_sample_tick_difference + timestamp;
82 tp.time_peak = tp.time_peak * m_sample_tick_difference + tp.time_start;
83 tp.time_over_threshold *= m_sample_tick_difference; // Convert to the same "time" units.
84 tp_aggr.push_back(tp);
85 }
86 cursor += register_alignment / 8; // Numerator is in bits. Need bytes.
87 }
88 }
89
90 return tp_aggr;
91 }
