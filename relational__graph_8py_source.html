1'''
2Extended explanation of the RelationalGraph class:
3
4The RelationalGraph class is used to generate a graph of the relationships between the DALs in the configuration file.
5The class is used to generate a topological ordering of the DALs, and to calculate the longest path in the graph.
6The reasoning is that the configuration should naturally group similar objects together based on how far they are from the session object
7
8Current this is only really used to find "top level" objects
9
10'''

30 def generate_graph(self):

46 def __generate_adjacency_matrix(self):
47 """Generates adjacency matrix from configuration handler object i.e. finds connected DALs
48 """
49 for i, dal in enumerate(self._handler.conf_obj_list):
50 for connection_category in self._handler.get_relationships_for_conf_object(dal):
51 # Allows for multiply connected nodes
52 for connection in list(connection_category.values())[0]:
53 # Loop over just conf objects
54 self._adjacency_matrix[i][self._handler.conf_obj_list.index(connection)] += 1

57 def __compute_degree(self):
58 """Get number of incoming nodes for each node"""
59 return np.sum(self._adjacency_matrix !=0, axis=0)

61 def __get_topological_order(self):
62 """
63 Topological sort of the adjacency graph
64
65 Algorithm implementation roughly based on: https://en.wikipedia.org/wiki/Topological_sorting#Kahn's_algorithm
66
67 """
68 in_degree = self.__compute_degree()
69 queue = deque(np.where(in_degree == 0)[0])
70 topological_ordering = []
71
72 while queue:
73 node = queue.popleft()
74 # Add node to topological ordering
75 topological_ordering.append(node)
76 #Â Check out going edges
77 outgoing_edges = np.where(self._adjacency_matrix != 0)[0]
78 # Reduce the number of incoming edges for each outgoing edge
79 in_degree[outgoing_edges] -= 1
80 # Change the number of nodes with no outgoing edges
81 zero_in_degree_nodes = outgoing_edges[in_degree[outgoing_edges] == 0]
82 # Add to the queue
83 queue.extend(zero_in_degree_nodes)
84
85 return topological_ordering

87 def __update_distances(self, distance: NDArray, node_id: int):
88 """Update maximum distance to each node
89
90 Arguments:
91 distance -- List of distancezs to each node
92 node_id -- ID of a node
93 """
94 outgoing_edges = np.where(self._adjacency_matrix[node_id] != 0)[0]
95 distance[outgoing_edges] = np.maximum(distance[outgoing_edges], distance[node_id] + self._adjacency_matrix[node_id][outgoing_edges])
96
97 def __longest_path(self, start_id: int):
98 """Calculate the longest path in a DAG from the start node."""
99 dist = np.full(self._handler.n_dals, -np.inf)
100 dist[start_id] = 0
101
102 for u in self._topological_ordered_matrix:
103 if dist[u] != -np.inf:
104 self.__update_distances(dist, u)
105 return dist

107 def __calculate_longest_paths(self)->None:
108 '''
109 Idea is to find shortest paths on -G for each top level node where G is the connection graph.
110 Layer each item lives on is then simply max(longest_path) for each top level item
111 '''
112
113 self._topological_ordered_matrix = self.__get_topological_order()
114 for node_id in range(self._handler.n_dals):
115 self._max_distance = np.maximum(self._max_distance, self.__longest_path(node_id))
116
117 self._max_distance = self._max_distance.astype(int)
