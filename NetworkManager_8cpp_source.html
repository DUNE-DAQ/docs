39void
40NetworkManager::configure(const std::string& session_name,
41 const std::vector<const confmodel::NetworkConnection*>& connections,
42 const confmodel::ConnectivityService* conn_svc,
43 dunedaq::opmonlib::OpMonManager& opmgr)
44{
45 if (!m_preconfigured_connections.empty()) {
46 throw AlreadyConfigured(ERS_HERE);
47 }
48
49 for (auto& connection : connections) {
50 auto name = connection->UID();
51 TLOG_DEBUG(15) << "Adding connection " << name << " to connection map";
52 ConnectionId id(connection);
53 if (m_preconfigured_connections.count(id)) {
54 TLOG_DEBUG(15) << "Name collision for connection " << name << ", DT " << connection->get_data_type()
55 << " connection_map.count: " << m_preconfigured_connections.count(id);
56 reset();
57 throw NameCollision(ERS_HERE, connection->UID());
58 }
59 m_preconfigured_connections[id] = connection;
60 }
61
62 if (conn_svc != nullptr) {
63
64 auto service = conn_svc->get_service();
65
66 auto connectionServer = conn_svc->get_host();
67 auto connectionPort = service->get_port();
68 auto config_client_interval = std::chrono::milliseconds(conn_svc->get_interval_ms());
69
70 TLOG_DEBUG(17) << "ConnectionServer host and port are " << connectionServer << ":" << connectionPort;
71 if (m_config_client == nullptr) {
72 m_config_client =
73 std::make_unique<ConfigClient>(connectionServer, std::to_string(connectionPort),session_name, config_client_interval);
74 }
75 m_config_client_interval = config_client_interval;
76 }
77
78 opmgr.register_node("senders", m_sender_opmon_link);
79 opmgr.register_node("receivers", m_receiver_opmon_link);
80}

210ConnectionResponse
211NetworkManager::get_connections(ConnectionId const& conn_id, bool restrict_single) const
212{
213 auto response = get_preconfigured_connections(conn_id);
214 if (restrict_single && response.connections.size() > 1) {
215 throw NameCollision(ERS_HERE, conn_id.uid);
216 }
217 if (m_config_client != nullptr) {
218 auto start_time = std::chrono::steady_clock::now();
219 while (
220 std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - start_time).count() <
221 1000) {
222 try {
223 auto client_response = m_config_client->resolveConnection(conn_id, conn_id.session);
224 if (restrict_single && client_response.connections.size() > 1) {
225 throw NameCollision(ERS_HERE, conn_id.uid);
226 }
227
228 if (client_response.connections.size() > 0) {
229 response = client_response;
230 }
231 break;
232 } catch (FailedLookup const& lf) {
233 if (m_config_client->is_connected()) {
234 throw ConnectionNotFound(ERS_HERE, conn_id.uid, conn_id.data_type, lf);
235 }
236 usleep(1000);
237 }
238 }
239 }
240
241 if (response.connections.size() == 0) {
242 throw ConnectionNotFound(ERS_HERE, conn_id.uid, conn_id.data_type);
243 }
244
245 return response;
246}

273std::shared_ptr<ipm::Receiver>
274NetworkManager::create_receiver(std::vector<ConnectionInfo> connections, ConnectionId const& conn_id)
275{
276 TLOG_DEBUG(12) << "START";
277 if (connections.size() == 0) {
278 return nullptr;
279 }
280
281 bool is_pubsub = connections[0].connection_type == ConnectionType::kPubSub;
282 if (connections.size() > 1 && !is_pubsub) {
283 throw OperationFailed(ERS_HERE,
284 "Trying to configure a kSendRecv receiver with multiple Connections is not allowed!");
285 }
286
287 auto plugin_type =
288 ipm::get_recommended_plugin_name(is_pubsub ? ipm::IpmPluginType::Subscriber : ipm::IpmPluginType::Receiver);
289
290 TLOG_DEBUG(12) << "Creating plugin of type " << plugin_type;
291 auto plugin = dunedaq::ipm::make_ipm_receiver(plugin_type);
292
293 nlohmann::json config_json;
294 if (is_pubsub) {
295 std::vector<std::string> uris;
296 for (auto& conn : connections) {
297 // Check for case where both ends are in app and ConnectivityService hasn't received other end yet
298 if (conn.uri.find("*") != std::string::npos || conn.uri.find("0.0.0.0") != std::string::npos) {
299 continue;
300 }
301 uris.push_back(conn.uri);
302 }
303 if (uris.size() == 0) {
304 return nullptr;
305 }
306 config_json["connection_strings"] = uris;
307 } else {
308 config_json["connection_string"] = connections[0].uri;
309 }
310 auto newCs = plugin->connect_for_receives(config_json);
311 TLOG_DEBUG(12) << "Receiver reports connected to URI " << newCs;
312
313 // Replace with resolved if there are wildcards (host and/or port)
314 if (connections[0].uri.find("*") != std::string::npos || connections[0].uri.find("0.0.0.0") != std::string::npos) {
315 TLOG_DEBUG(14) << "Wildcard found in connection URI " << connections[0].uri << ", adjusting before publish";
316 auto newUri = utilities::parse_connection_string(newCs);
317 auto oldUri = utilities::parse_connection_string(connections[0].uri);
318
319 if (oldUri.port == "*")
320 oldUri.port = newUri.port;
321 if (oldUri.host == "*" || oldUri.host == "0.0.0.0")
322 oldUri.host = newUri.host;
323
324 connections[0].uri = oldUri.to_string();
325 TLOG_DEBUG(14) << "Connection URI is now " << connections[0].uri;
326 }
327
328 if (is_pubsub) {
329 TLOG_DEBUG(12) << "Subscribing to topic " << connections[0].data_type << " after connect_for_receives";
330 auto subscriber = std::dynamic_pointer_cast<ipm::Subscriber>(plugin);
331 subscriber->subscribe(connections[0].data_type);
332 std::lock_guard<std::mutex> lkk(m_subscriber_plugin_map_mutex);
333 m_subscriber_plugins[conn_id] = subscriber;
334 if (!m_subscriber_update_thread_running && m_config_client != nullptr) {
335 m_subscriber_update_thread_running = true;
336 m_subscriber_update_thread.reset(new std::thread(&NetworkManager::update_subscribers, this));
337 }
338 }
339
340 if (m_config_client != nullptr && !is_pubsub) {
341 m_config_client->publish(connections[0]);
342 }
343
344 register_monitorable_node(plugin, m_receiver_opmon_link, conn_id.uid, is_pubsub);
345
346 TLOG_DEBUG(12) << "END";
347 return plugin;
348}

350std::shared_ptr<ipm::Sender>
351NetworkManager::create_sender(ConnectionInfo connection)
352{
353 auto is_pubsub = connection.connection_type == ConnectionType::kPubSub;
354 auto plugin_type =
355 ipm::get_recommended_plugin_name(is_pubsub ? ipm::IpmPluginType::Publisher : ipm::IpmPluginType::Sender);
356
357 // Check for case where both ends are in app and ConnectivityService hasn't received other end yet
358 if (!is_pubsub &&
359 (connection.uri.find("*") != std::string::npos || connection.uri.find("0.0.0.0") != std::string::npos)) {
360 return nullptr;
361 }
362
363 TLOG_DEBUG(11) << "Creating sender plugin of type " << plugin_type;
364 auto plugin = dunedaq::ipm::make_ipm_sender(plugin_type);
365 TLOG_DEBUG(11) << "Connecting sender plugin to " << connection.uri;
366 auto newCs = plugin->connect_for_sends({ { "connection_string", connection.uri } });
367 TLOG_DEBUG(11) << "Sender Plugin connected, reports URI " << newCs;
368
369 // Replace with resolved if there are wildcards (host and/or port)
370 if (connection.uri.find("*") != std::string::npos || connection.uri.find("0.0.0.0") != std::string::npos) {
371 TLOG_DEBUG(13) << "Wildcard found in connection URI " << connection.uri << ", adjusting before publish";
372 auto newUri = utilities::parse_connection_string(newCs);
373 auto oldUri = utilities::parse_connection_string(connection.uri);
374
375 if (oldUri.port == "*")
376 oldUri.port = newUri.port;
377 if (oldUri.host == "*" || oldUri.host == "0.0.0.0")
378 oldUri.host = newUri.host;
379
380 connection.uri = oldUri.to_string();
381 TLOG_DEBUG(13) << "Connection URI is now " << connection.uri;
382 }
383
384 if (m_config_client != nullptr && is_pubsub) {
385 m_config_client->publish(connection);
386 }
387
388 register_monitorable_node(plugin, m_sender_opmon_link, connection.uid, is_pubsub);
389
390 return plugin;

393void
394NetworkManager::update_subscribers()
395{
396 while (m_subscriber_update_thread_running.load()) {
397 {
398 TLOG_DEBUG(14) << "Updating registered subscribers";
399 std::lock_guard<std::mutex> lk(m_subscriber_plugin_map_mutex);
400 for (auto& subscriber_pair : m_subscriber_plugins) {
401 try {
402 auto response = get_connections(subscriber_pair.first, false);
403
404 nlohmann::json config_json;
405 std::vector<std::string> uris;
406 for (auto& conn : response.connections)
407 uris.push_back(conn.uri);
408 config_json["connection_strings"] = uris;
409
410 subscriber_pair.second->connect_for_receives(config_json);
411 } catch (ers::Issue&) {
412 }
413 }
414 }
415 std::this_thread::sleep_for(m_config_client_interval);
416 }
417}
