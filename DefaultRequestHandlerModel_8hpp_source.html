63// This function takes the type returned by the begin() and end()
64// functions in a ReadoutType object and returns the timestamp that
65// should be used to determine whether the item pointed to by the
66// iterator is within the readout window. The "frame" type
67// ReadoutTypes all have a get_timestamp() function on the type
68// pointed to by the iterator (eg, WIBFrame), so we make this
69// "default" function return that. For other types, eg, those coming
70// from DS, there isn't a get_timestamp() function, so a different
71// template specialization is used in the appropriate package (eg,
72// trigger)
73
74template<class T>
75uint64_t
76get_frame_iterator_timestamp(T iter)
77{
78 return iter->get_timestamp();
79}

183 // An inline helper function that merges a set of byte arrays into a destination array
184 inline
185 void dump_to_buffer(const void* data, std::size_t size,
186 void* buffer, uint32_t buffer_pos, // NOLINT(build/unsigned)
187 const std::size_t& buffer_size)
188 {
189 auto bytes_to_copy = size;
190 while (bytes_to_copy > 0) {
191 auto n = std::min(bytes_to_copy, buffer_size - buffer_pos);
192 std::memcpy(static_cast<char*>(buffer) + buffer_pos, static_cast<const char*>(data), n);
193 buffer_pos += n;
194 bytes_to_copy -= n;
195 if (buffer_pos == buffer_size) {
196 buffer_pos = 0;
197 }
198 }
199 }
