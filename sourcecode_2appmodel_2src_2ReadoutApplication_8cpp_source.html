165 // Process the queue rules looking for inputs to our DL/TP handler modules
166 //
167 const QueueDescriptor* dlh_input_qdesc = nullptr;
168 const QueueDescriptor* dlh_reqinput_qdesc = nullptr;
169 const QueueDescriptor* tp_input_qdesc = nullptr;
170 // const QueueDescriptor* tpReqInputQDesc = nullptr;
171 const QueueDescriptor* fa_output_qdesc = nullptr;
172
173 for (auto rule : get_queue_rules()) {
174 auto destination_class = rule->get_destination_class();
175 auto data_type = rule->get_descriptor()->get_data_type();
176 // Why datahander here?
177 if (destination_class == "DataHandlerModule" || destination_class == dlh_class || destination_class == tph_class) {
178 if (data_type == "DataRequest") {
179 dlh_reqinput_qdesc = rule->get_descriptor();
180 } else if (data_type == "TriggerPrimitive" && get_tp_generation_enabled()) {
181 tp_input_qdesc = rule->get_descriptor();
182 } else {
183 dlh_input_qdesc = rule->get_descriptor();
184 }
185 } else if (destination_class == "FragmentAggregatorModule") {
186 fa_output_qdesc = rule->get_descriptor();
187 }
188 }
189
190 //
191 // Process the network rules looking for the Fragment Aggregator and TP handler data reuest inputs
192 //
193 const NetworkConnectionDescriptor* fa_net_desc = nullptr;
194 const NetworkConnectionDescriptor* tp_net_desc = nullptr;
195 const NetworkConnectionDescriptor* ta_net_desc = nullptr;
196 const NetworkConnectionDescriptor* ts_net_desc = nullptr;
197 for (auto rule : get_network_rules()) {
198 auto endpoint_class = rule->get_endpoint_class();
199 auto data_type = rule->get_descriptor()->get_data_type();
200
201 if (endpoint_class == "FragmentAggregatorModule") {
202 fa_net_desc = rule->get_descriptor();
203 } else if (data_type == "TPSet") {
204 tp_net_desc = rule->get_descriptor();
205 } else if (data_type == "TriggerActivity") {
206 ta_net_desc = rule->get_descriptor();
207 } else if (data_type == "TimeSync") {
208 ts_net_desc = rule->get_descriptor();
209 }
210 }
211
212 // Create here the Queue on which all data fragments are forwarded to the fragment aggregator
213 // and a container for the queues of data request to TP handler and DLH
214 if (fa_output_qdesc == nullptr) {
215 throw(BadConf(ERS_HERE, "No fragment output queue descriptor given"));
216 }
217 std::vector<const confmodel::Connection*> req_queues;
218 conffwk::ConfigObject frag_queue_obj = obj_fac.create_queue_obj(fa_output_qdesc);
219
220 //
221 // Scan Detector 2 DAQ connections to extract sender, receiver and stream information
222 //
223
224 std::vector<const confmodel::DaqModule*> modules;
225
226 // Loop over the detector to daq connections and generate one data reader per connection
227 // and the cooresponding datalink handlers
228
229 // Collect all streams
230 std::vector<const confmodel::DetectorStream*> det_streams;
231 std::vector<const conffwk::ConfigObject*> d2d_conn_objs;
232
233 for (auto d2d_conn_res : get_contains()) {
234
235 // Are we sure?
236 if (d2d_conn_res->disabled(*session)) {
237 TLOG_DEBUG(7) << "Ignoring disabled DetectorToDaqConnection " << d2d_conn_res->UID();
238 continue;
239 }
240
241 d2d_conn_objs.push_back(&d2d_conn_res->config_object());
242
243 TLOG_DEBUG(6) << "Processing DetectorToDaqConnection " << d2d_conn_res->UID();
244 // get the readout groups and the interfaces and streams therein; 1 reaout group corresponds to 1 data reader module
245 auto d2d_conn = d2d_conn_res->cast<confmodel::DetectorToDaqConnection>();
246
247 if (!d2d_conn) {
248 throw(BadConf(ERS_HERE, "ReadoutApplication contains something other than DetectorToDaqConnection"));
249 }
250
251 if (d2d_conn->get_contains().empty()) {
252 throw(BadConf(ERS_HERE, "DetectorToDaqConnection does not contain sebders or receivers"));
253 }
254
255 // Loop over detector 2 daq connections to find senders and receivers
256 auto det_senders = d2d_conn->get_senders();
257 auto det_receiver = d2d_conn->get_receiver();
258
259 // Loop over senders
260 for (auto stream : d2d_conn->get_streams()) {
261
262 // Are we sure?
263 if (stream->disabled(*session)) {
264 TLOG_DEBUG(7) << "Ignoring disabled DetectorStream " << stream->UID();
265 continue;
266 }
267
268 // loop over streams
269 det_streams.push_back(stream);
270 }
271
272
273
274 // Here I want to resolve the type of connection (network, felix, or?)
275 // Rules of engagement: if the receiver interface is network or felix, the receivers should be castable to the counterpart
276 if (reader_class == "DPDKReaderModule") {
277 if (!det_receiver->cast<appmodel::DPDKReceiver>()) {
278 throw(BadConf(ERS_HERE, fmt::format("DPDKReaderModule requires NWDetDataReceiver, found {} of class {}", det_receiver->UID(), det_receiver->class_name())));
279 }
280
281 bool all_nw_senders = true;
282 for (auto s : det_senders) {
283 all_nw_senders &= (s->cast<appmodel::NWDetDataSender>() != nullptr);
284 }
285
286 // Ensure that all senders are compatible with receiver
287 if (!all_nw_senders) {
288 throw(BadConf(ERS_HERE, "Non-network DetDataSener found with NWreceiver"));
289 }
290 }
291 }

295 // Create DataReaderModule object

297
298 //
299 // Instantiate DataReaderModule of type DPDKReaderModule

301
302 // Create the DPDKReaderModule object
303
304 uint16_t conn_idx = 0;
305 std::string reader_uid(fmt::format("datareader-{}-{}", this->UID(), std::to_string(conn_idx++)));
306 conffwk::ConfigObject reader_obj;
307 TLOG_DEBUG(6) << fmt::format("creating OKS configuration object for Data reader class {} with id {}", reader_class, reader_uid);
308 config->create(dbfile, reader_class, reader_uid, reader_obj);
309
310 // Populate configuration and interfaces (leave output queues for later)
311 reader_obj.set_obj("configuration", &reader_conf->config_object());
312 reader_obj.set_objs("connections", d2d_conn_objs);
313
314 // Create the raw data queues
315 std::vector<const conffwk::ConfigObject*> data_queue_objs;
316 // keep a map for convenience
317 std::map<uint32_t, const confmodel::Connection*> data_queues_by_sid;
318
319 // Create data queues
320 for (auto ds : det_streams) {
321 conffwk::ConfigObject queue_obj = obj_fac.create_queue_sid_obj(dlh_input_qdesc, ds);
322 const auto* connection = config->get<confmodel::Connection>(queue_obj.UID());
323 data_queue_objs.push_back(&connection->config_object());
324 data_queues_by_sid[ds->get_source_id()] = connection;
325 }
326
327 reader_obj.set_objs("outputs", data_queue_objs);
328
329 modules.push_back(config->get<confmodel::DaqModule>(reader_uid));

331 //-----------------------------------------------------------------
332 //
333 // Prepare the tp handlers and related queues
334 //
335 std::vector<const confmodel::Connection*> tp_queues;
336 if (get_tp_generation_enabled()) {
337
338 // Create TP handler object
339 auto tph_conf_obj = tph_conf->config_object();
340 auto tpsrc_ids = get_tp_source_ids();
341
342 for (auto sid : tpsrc_ids) {
343 conffwk::ConfigObject tp_queue_obj;
344 conffwk::ConfigObject tpreq_queue_obj;
345 conffwk::ConfigObject tph_obj;
346 std::string tp_uid("tphandler-" + std::to_string(sid->get_sid()));
347 config->create(dbfile, tph_class, tp_uid, tph_obj);
348 tph_obj.set_by_val<uint32_t>("source_id", sid->get_sid());
349 tph_obj.set_by_val<uint32_t>("detector_id", 1); // 1 == kDAQ
350 tph_obj.set_by_val<bool>("post_processing_enabled", get_ta_generation_enabled());
351 tph_obj.set_obj("module_configuration", &tph_conf_obj);
352
353 // Create the TPs aggregator queue (from RawData Handlers to TP handlers)
354 tp_queue_obj = obj_fac.create_queue_sid_obj(tp_input_qdesc, sid->get_sid());
355 tp_queue_obj.set_by_val<uint32_t>("recv_timeout_ms", 1);
356 tp_queue_obj.set_by_val<uint32_t>("send_timeout_ms", 1);
357
358 tp_queues.push_back(config->get<confmodel::Connection>(tp_queue_obj.UID()));
359 // Create tp data requests queue from Fragment Aggregator
360 tpreq_queue_obj = obj_fac.create_queue_sid_obj(dlh_reqinput_qdesc, sid->get_sid());
361 req_queues.push_back(config->get<confmodel::Connection>(tpreq_queue_obj.UID()));
362
363 // Create the tp(set) publishing service
364 conffwk::ConfigObject tp_net_obj = obj_fac.create_net_obj(tp_net_desc, tp_uid);
365
366 // Create the ta(set) publishing service
367 conffwk::ConfigObject ta_net_obj = obj_fac.create_net_obj(ta_net_desc, tp_uid);
368
369 // Register queues with tp hankder
370 tph_obj.set_objs("inputs", { &tp_queue_obj, &tpreq_queue_obj });
371 tph_obj.set_objs("outputs", { &tp_net_obj, &ta_net_obj, &frag_queue_obj });
372 modules.push_back(config->get<confmodel::DaqModule>(tp_uid));
373 }
374 }
375
376 // Add output queueus of tps
377 std::vector<const conffwk::ConfigObject*> tp_queue_objs;
378 for (auto q : tp_queues) {
379 tp_queue_objs.push_back(&q->config_object());
380 }

384 // Create datalink handlers

386 // Recover the emulation flag

388 for (auto ds : det_streams) {
389
390 uint32_t sid = ds->get_source_id();
391 TLOG_DEBUG(6) << fmt::format("Processing stream {}, id {}, det id {}", ds->UID(), ds->get_source_id(), ds->get_geo_id()->get_detector_id());
392 std::string uid(fmt::format("DLH-{}", sid));
393 conffwk::ConfigObject dlh_obj;

395 config->create(dbfile, dlh_class, uid, dlh_obj);
396 dlh_obj.set_by_val<uint32_t>("source_id", sid);
397 dlh_obj.set_by_val<uint32_t>("detector_id", ds->get_geo_id()->get_detector_id());
398 dlh_obj.set_by_val<bool>("post_processing_enabled", get_tp_generation_enabled());
399 dlh_obj.set_by_val<bool>("emulation_mode", emulation_mode);
400 dlh_obj.set_obj("geo_id", &ds->get_geo_id()->config_object());
401 dlh_obj.set_obj("module_configuration", &dlh_conf->config_object());
402
403 std::vector<const conffwk::ConfigObject*> dlh_ins, dlh_outs;
404
405 // Add datalink-handler queue to the inputs
406 dlh_ins.push_back(&data_queues_by_sid.at(sid)->config_object());
407
408 // Create request queue
409 conffwk::ConfigObject req_queue_obj = obj_fac.create_queue_sid_obj(dlh_reqinput_qdesc, ds);
410
411
412 // Add the requessts queue dal pointer to the outputs of the FragmentAggregatorModule
413 req_queues.push_back(config->get<confmodel::Connection>(req_queue_obj.UID()));
414 dlh_ins.push_back(&req_queue_obj);
415 dlh_outs.push_back(&frag_queue_obj);
416
417
418 // Time Sync network connection
419 if (dlh_conf->get_generate_timesync()) {
420 // Add timestamp endpoint
421 conffwk::ConfigObject ts_net_obj = obj_fac.create_net_obj(ts_net_desc, std::to_string(sid));
422 dlh_outs.push_back(&ts_net_obj);
423 }
424

426 dlh_outs.push_back(tpq);
427 }
428 dlh_obj.set_objs("inputs", dlh_ins);
429 dlh_obj.set_objs("outputs", dlh_outs);

431 modules.push_back(config->get<confmodel::DaqModule>(uid));
432 }
433
434
435 // Finally create Fragment Aggregator
436 std::string faUid("fragmentaggregator-" + UID());
437 conffwk::ConfigObject frag_aggr;
438 TLOG_DEBUG(7) << "creating OKS configuration object for Fragment Aggregator class ";

440
441 // Add network connection from TRBs
442 conffwk::ConfigObject fa_net_obj = obj_fac.create_net_obj(fa_net_desc);
443
444 // Process special Network rules!
445 // Looking for Fragment rules from DFAppplications in current Session
446 auto sessionApps = session->get_enabled_applications();
447 std::vector<conffwk::ConfigObject> fragOutObjs;
448 for (auto app : sessionApps) {
449 auto dfapp = app->cast<appmodel::DFApplication>();

469 } // Loop over Apps network rules
470 } // loop over Session specific Apps
471
472 // Add output queueus of data requests and Fragments
473 std::vector<const conffwk::ConfigObject*> fa_output_objs;

475 fa_output_objs.push_back(&fNet);

479 fa_output_objs.push_back(&q->config_object());

483 frag_aggr.set_objs("outputs", fa_output_objs);

485 modules.push_back(config->get<confmodel::DaqModule>(frag_aggr.UID()));
486
