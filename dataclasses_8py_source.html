18def sparsify_array_diff_locs_and_vals(arr):
19 # Find indices where the value changes compared to the previous value, adjusting the first index to start from 0
20 change_locations = np.insert(np.where(arr[1:] != arr[:-1])[0], 0, -1) + 1
21 # Check if the array is not empty
22 if len(arr) > 0:
23 # Return locations of changes, values at these locations, and the length of the array
24 return change_locations, arr[change_locations], len(arr)
25 else:
26 # Return empty results for an empty array
27 return [], [], 0

29def desparsify_array_diff_locs_and_vals(change_locations, change_values, arr_size):
30 # Create an empty array of the original size
31 reconstructed_arr = np.empty(arr_size, dtype=np.uint)
32 # Loop through each change location
33 for i in range(len(change_locations)):
34 # Apply the change value from the current change location to the end or next change location
35 if (i + 1) == len(change_locations):
36 reconstructed_arr[change_locations[i]:] = change_values[i]
37 else:
38 reconstructed_arr[change_locations[i]:change_locations[i + 1]] = change_values[i]
39 return reconstructed_arr

41def sparsify_array_diff_of_diff_locs_and_vals(arr):
42 # Store the first value of the array for later reconstruction
43 arr_first = arr[0]
44 # Compute the difference of consecutive elements
45 arr_diff = np.diff(arr)
46 # Use sparsify function to find locations and values of changes in the diff array
47 arr_diff_locs, arr_diff_vals, _ = sparsify_array_diff_locs_and_vals(arr_diff)
48 # Return the first value, change locations, change values, and array size for reconstruction
49 return arr_first, arr_diff_locs, arr_diff_vals, len(arr)

51def desparsify_array_diff_of_diff_locs_and_vals(arr_first, change_locations, change_values, arr_size):
52 # Reconstruct the differential array from sparse representation
53 arr_diff = desparsify_array_diff_locs_and_vals(change_locations, change_values, arr_size - 1)
54 # Reconstruct the original array by cumulatively summing the differences and adding the first value
55 arr = np.concatenate((np.array([0], dtype=np.uint), arr_diff)).cumsum() + arr_first
56 return arr
