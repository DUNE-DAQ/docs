56 void init(const confmodel::DaqModule* cfg) override
57 {
58 if (cfg->get_outputs().size() != 1) {
59 throw datahandlinglibs::InitializationError(ERS_HERE, "Only 1 output supported for subscribers");
60 }
61 m_data_sender = get_iom_sender<triggeralgs::TriggerCandidate>(cfg->get_outputs()[0]->UID());
62
63 if (cfg->get_inputs().size() != 1) {
64 throw datahandlinglibs::InitializationError(ERS_HERE, "Only 1 input supported for subscribers");
65 }
66 m_data_receiver = get_iom_receiver<dfmessages::HSIEvent>(cfg->get_inputs()[0]->UID());
67 auto data_reader = cfg->cast<appmodel::DataSubscriberModule>();
68 if (data_reader == nullptr) {
69 throw datahandlinglibs::InitializationError(ERS_HERE, "DAQ module is not a DataReader");
70 }
71 auto hsi_conf = data_reader->get_configuration()->cast<appmodel::HSI2TCTranslatorConf>();
72 if (hsi_conf == nullptr) {
73 throw datahandlinglibs::InitializationError(ERS_HERE, "Missing HSI2TCTranslatorConf");
74 }
75
76 // Get the HSI-signal to TC-output map
77 for (auto win : hsi_conf->get_signals()) {
78 triggeralgs::TriggerCandidate::Type tc_type;
79 tc_type = static_cast<triggeralgs::TriggerCandidate::Type>(
80 dunedaq::trgdataformats::string_to_fragment_type_value(win->get_tc_type_name()));
81
82 // Throw error if unknown TC type
83 if (tc_type == triggeralgs::TriggerCandidate::Type::kUnknown) {
84 throw datahandlinglibs::InitializationError(ERS_HERE, "Provided an unknown TC type output to HSISourceModel");
85 }
86
87 // Throw error if already exists
88 uint32_t signal = win->get_signal_type();
89 if (m_signals.count(signal)) {
90 throw datahandlinglibs::InitializationError(ERS_HERE, "Provided more than one of the same HSI signal ID input to HSISourceModel");
91 }
92
93 // Fill the signal-tctype map
94 m_signals[signal] = { tc_type,
95 win->get_time_before(),
96 win->get_time_after() };
97
98 TLOG() << "Will cover HSI signal id: " << signal << " to TC type: " << win->get_tc_type_name()
99 << " window before: " << win->get_time_before() << " window after: " << win->get_time_after();
100 }
101
102 m_prescale = hsi_conf->get_prescale();
103 m_latency_monitoring.store( hsi_conf->get_latency_monitoring() );
104
105 }

124 bool handle_payload(dfmessages::HSIEvent& data) // NOLINT(build/unsigned)
125 {
126 m_received_events_count++;
127 if (m_latency_monitoring.load()) m_latency_instance.update_latency_in( data.timestamp );
128
129 // Prescale after n-hsi received
130 if (m_received_events_count % m_prescale != 0) {
131 return true;
132 }
133
134 TLOG_DEBUG(1) << "Received HSIEvent with signal map " << data.signal_map << " and timestamp " << data.timestamp;
135
136 // Iterate over all the signals
137 uint32_t signal_map = data.signal_map;
138 while (signal_map) {
139 // Get the index of the least significant bit
140 int bit_index = __builtin_ctzll(signal_map);
141 uint32_t signal = 1 << bit_index;
142
143 // Throw an error if we don't have this signal bit configured
144 if (!m_signals.count(signal)) {
145 throw dunedaq::trigger::SignalTypeError(ERS_HERE, "HSI subscriber" , data.signal_map);
146 }
147
148 // Create the trigger candidate
149 triggeralgs::TriggerCandidate candidate;
150 candidate.time_start = data.timestamp - m_signals[signal].time_before;
151 candidate.time_end = data.timestamp + m_signals[signal].time_after;
152 candidate.time_candidate = data.timestamp;
153 // throw away bits 31-16 of header, that's OK for now
154 candidate.detid = (uint)detdataformats::DetID::Subdetector::kDAQ ; // NOLINT(build/unsigned)
155 candidate.type = m_signals[signal].type;
156 candidate.algorithm = triggeralgs::TriggerCandidate::Algorithm::kHSIEventToTriggerCandidate;
157 candidate.inputs = {};
158 m_tcs_made_count++;
159
160 if (m_latency_monitoring.load()) m_latency_instance.update_latency_out( candidate.time_candidate );
161 // Send the TC
162 if (!m_data_sender->try_send(std::move(candidate), iomanager::Sender::s_no_block)) {
163 m_tcs_dropped_count++;
164 }
165 else {
166 m_tcs_sent_count++;
167 }
168
169 // Clear the least significant bit
170 signal_map &= signal_map - 1;
171 }
172
173 return true;
174 }
