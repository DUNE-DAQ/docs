69TPProcessor::conf(const appmodel::DataHandlerModule* conf)

71 for (auto output : conf->get_outputs()) {

77 ers::error(datahandlinglibs::ResourceQueueError(ERS_HERE, "ta", "DefaultRequestHandlerModel", excpt));

80
81 m_sourceid.id = conf->get_source_id();
82 m_sourceid.subsystem = TriggerPrimitiveTypeAdapter::subsystem;
83
84 std::vector<const appmodel::TAAlgorithm*> ta_algorithms;
85 auto dp = conf->get_module_configuration()->get_data_processor();
86 auto proc_conf = dp->cast<appmodel::TPDataProcessor>();
87 if (proc_conf != nullptr && m_post_processing_enabled) {
88 ta_algorithms = proc_conf->get_algorithms();
89 }
90
91 for (auto algo : ta_algorithms) {
92 TLOG() << "Selected TA algorithm: " << algo->UID() << " from class " << algo->class_name();
93 std::shared_ptr<triggeralgs::TriggerActivityMaker> maker = make_ta_maker(algo->class_name());
94 nlohmann::json algo_json = algo->to_json(true);
95
96 TLOG() << "Algo config:\n" << algo_json.dump();
97
98 maker->configure(algo_json[algo->UID()]);
99 inherited::add_postprocess_task(std::bind(&TPProcessor::find_ta, this, std::placeholders::_1, maker));
100 m_tams.push_back(maker);
101 }
102 m_latency_monitoring.store( dp->get_latency_monitoring() );
103 inherited::conf(conf);

132void
133TPProcessor::find_ta(const TriggerPrimitiveTypeAdapter* tp, std::shared_ptr<triggeralgs::TriggerActivityMaker> taa)
134{
135 if (m_latency_monitoring.load()) m_latency_instance.update_latency_in( tp->tp.time_start ); // time_start or time_peak ?
136 m_tp_received_count++;
137 std::vector<triggeralgs::TriggerActivity> tas;
138 taa->operator()(tp->tp, tas);
139
140 while (tas.size()) {
141 m_ta_made_count++;
142 if (m_latency_monitoring.load()) m_latency_instance.update_latency_out( tas.back().time_start );
143 if (!m_ta_sink->try_send(std::move(tas.back()), iomanager::Sender::s_no_block)) {
144 ers::warning(TADropped(ERS_HERE, tp->tp.time_start, m_sourceid.id));
145 m_ta_failed_sent_count++;
146 } else {
147 m_ta_sent_count++;
148 }
149 tas.pop_back();
150 }
151 return;
152}
